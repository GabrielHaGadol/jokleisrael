    א(יג) וַיֹּאמֶר אֵלָיו בָּלָק לְךָ-נָּא אִתִּי אֶל מָקוֹם אַחֵר אֲשֶׁר תִּרְאֶנּוּ מִשָּׁם אֶפֶס קָצֵהוּ תִרְאֶה וְכֻלּוֹ לֹא תִרְאֶה וְקָבְנוֹ לִי מִשָּׁם: וַאֲמַר לֵיהּ בָּלָק אִיתָא כְעַן עִמִי לַאֲתַר אוֹחֲרָן דְתֶחֱזִנֵיהּ מִתַּמָן לְחוֹד קְצָתֵיהּ תֶּחֱזֵי וְכוּלֵיהּ לָא תֶחֱזֵי וּתְלוּטֵיהּ לִי מִתַּמָן: (יד) וַיִּקָּחֵהוּ שְׂדֵה צֹפִים אֶל רֹאשׁ הַפִּסְגָּה וַיִּבֶן שִׁבְעָה מִזְבְּחֹת וַיַּעַל פָּר וָאַיִל בַּמִּזְבֵּחַ: וְדַבְּרֵיהּ לַחֲקַל סְכוּתָא לְרֵישׁ רָמָתָא וּבְנָא שַׁבְעָא מַדְבְּחִין וְאַסֵיק תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא: (טו) וַיֹּאמֶר אֶל בָּלָק הִתְיַצֵּב כֹּה עַל עֹלָתֶךָ וְאָנֹכִי אִקָּרֶה כֹּה: וַאֲמַר לְבָלָק אִתְעַתֵּד הָכָא עַל עֲלָתָךְ וַאֲנָא אִתְמְטֵי עַד כָּא: (טז) וַיִּקָּר יְהוָה אֶל בִּלְעָם וַיָּשֶׂם דָּבָר בְּפִיו וַיֹּאמֶר שׁוּב אֶל בָּלָק וְכֹה תְדַבֵּר: וְעָרַע מֵימַר מִן קֳדָם יְיָ לְבִלְעָם וְשַׁוִי פִתְגָמָא בְּפוּמֵיהּ וַאֲמַר תּוּב לְוָת בָּלָק וּכְדֵין תְּמַלֵיל: (יז) וַיָּבֹא אֵלָיו וְהִנּוֹ נִצָּב עַל עֹלָתוֹ וְשָׂרֵי מוֹאָב אִתּוֹ וַיֹּאמֶר לוֹ בָּלָק מַה דִּבֶּר יְהוָה: וַאֲתָא לְוָתֵיהּ וְהוּא מְעַתַּד עַל עֲלָתֵיהּ וְרַבְרְבֵי מוֹאָב עִמֵיהּ וַאֲמַר לֵיהּ בָּלָק מָה מַלִיל יְיָ: (יח) וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר קוּם בָּלָק וּשֲׁמָע הַאֲזִינָה עָדַי בְּנוֹ צִפֹּר: וּנְטַל מַתְלֵיהּ וַאֲמַר קוּם בָּלָק וּשְׁמַע אָצִית לְמֵימְרִי בַּר צִפֹּר: (יט) לֹא אִישׁ אֵל וִיכַזֵּב וּבֶן אָדָם וְיִתְנֶחָם הַהוּא אָמַר וְלֹא יַעֲשֶׂה וְדִבֶּר וְלֹא יְקִימֶנָּה: לָא כְמִלֵי בְּנֵי אֱנָשָׁא מֵימַר אֱלָהָא בְּנֵי אֱנָשָׁא אָמְרִין וּמְכַדְבִין וְאַף לָא כְעוֹבָדֵי בְּנֵי בִּשְׂרָא דְאִינוּן גָזְרִין לְמֶעְבַּד וְתַיְבִין וּמִתְמַלְכִין דְהוּא אָמַר וְעָבֵיד וְכָל מֵימְרֵהּ מִתְקַיָם: (כ) הִנֵּה בָרֵךְ לָקָחְתִּי וּבֵרֵךְ וְלֹא אֲשִׁיבֶנָּה: הָא בִּרְכָן קַבֵּילִית וַאֲבָרֵכִינֵיהּ לְיִשְׂרָאֵל וְלָא אָתֵב בִּרְכְּתִי מִינֵיה: (כא) לֹא הִבִּיט אָוֶן בְּיַעֲקֹב וְלֹא רָאָה עָמָל בְּיִשְׂרָאֵל יְהוָה אֱלֹהָיו עִמּוֹ וּתְרוּעַת מֶלֶךְ בּוֹ: אִסְתַּכָּלִית לֵית פָּלְחֵי גִלוּלִין בִּדְבֵית יַעֲקֹב וְאַף לָא עָבְדֵי לֵיאוּת שְׁקַר בְּיִשְׂרָאֵל מֵימְרָא דַיְיָ אֱלָהָהוֹן בְּסַעְדְהוֹן וּשְׁכִינַת מַלְכְּהוֹן בֵּינֵיהוֹן: (כב) אֵל מוֹצִיאָם מִמִּצְרָיִם כְּתוֹעֲפֹת רְאֵם לוֹ: אֱלָהָא דְאַפֵּקִנוּן מִמִצְרָיִם תּוּקְפָא וְרוּמָא דִילֵיהּ: (כג) כִּי לֹא נַחַשׁ בְּיַעֲקֹב וְלֹא קֶסֶם בְּיִשְׂרָאֵל כָּעֵת יֵאָמֵר לְיַעֲקֹב וּלְיִשְׂרָאֵל מַה פָּעַל אֵל: אֲרֵי לָא נַחֲשַׁיָא צָבָן דְיִיטַב לִדְבֵית יַעֲקֹב וְאַף לָא קְסָמַיָא רְעַן בִּרְבוּת בֵּית יִשְׂרָאֵל כְּעִדַן יִתְאַמַר לְיַעֲקֹב וּלְיִשְׂרָאֵל מָה עֲבַד אֱלָהָא: (כד) הֶן עָם כְּלָבִיא יָקוּם וְכַאֲרִי יִתְנַשָּׂא לֹא יִשְׁכַּב עַד יֹאכַל טֶרֶף וְדַם חֲלָלִים יִשְׁתֶּה: הָא עַמָא כְּלֵיתָא שָׁרֵי וּכְאַרְיָא יִתְנַטֵל לָא יִשְׁרֵי בְאַרְעֵיהּ עַד דְיִקְטוֹל קְטוֹל וְנִכְסֵי עֲמְמַיָא יֵירָת: (כה) וַיֹּאמֶר בָּלָק אֶל בִּלְעָם גַּם קֹב לֹא תִקֳּבֶנּוּ גַּם בָּרֵךְ לֹא תְבָרֲכֶנּוּ: וַאֲמַר בָּלָק לְבִלְעָם אַף מֵילַט לָא תְלוּטִּינוּן אַף בְּרָכָא לָא תְבָרֵכִינוּן: (כו) וַיַּעַן בִּלְעָם וַיֹּאמֶר אֶל בָּלָק הֲלֹא דִּבַּרְתִּי אֵלֶיךָ לֵאמֹר כֹּל אֲשֶׁר יְדַבֵּר יְהוָה אֹתוֹ אֶעֱשֶׂה: וַאֲתֵיב בִּלְעָם וַאֲמַר לְבָלָק הֲלָא מַלֵילִית עִמָךְ לְמֵימַר כֹּל דִי יְמַלֵל יְיָ יָתֵיהּ אַעֲבֵיד: שישי - במחוברות שביעי (כז) וַיֹּאמֶר בָּלָק אֶל בִּלְעָם לְכָה נָּא אֶקָּחֲךָ אֶל מָקוֹם אַחֵר אוּלַי יִישַׁר בְּעֵינֵי הָאֱלֹהִים וְקַבֹּתוֹ לִי מִשָּׁם: וַאֲמַר בָּלָק לְבִלְעָם אִיתָא כְּעַן אֲדַבְּרִינָךְ לַאֲתַר אָחֳרָן מָאִים יְהֵי רַעֲוָא מִן קֳדָם יְיָ וּתְלוּטֵיהּ לִי מִתַּמָן: (כח) וַיִּקַּח בָּלָק אֶת בִּלְעָם רֹאשׁ הַפְּעוֹר הַנִּשְׁקָף עַל פְּנֵי הַיְשִׁימֹן: וּדְבַר בָּלָק יַת בִּלְעָם רֵישׁ רָמָתָא דְמִסְתַּכְּיָא עַל אַפֵּי בֵּית יְשִׁימוֹן: (כט) וַיֹּאמֶר בִּלְעָם אֶל בָּלָק בְּנֵה לִי בָזֶה שִׁבְעָה מִזְבְּחֹת וְהָכֵן לִי בָּזֶה שִׁבְעָה פָרִים וְשִׁבְעָה אֵילִם: וַאֲמַר בִּלְעָם לְבָלָק בְּנֵה לִי הָכָא שַׁבְעָא מַדְבְּחִין וְאַתְקִין לִי הָכָא שַׁבְעָא תוֹרִין וְשַׁבְעָא דִכְרִין: (ל) וַיַּעַשׂ בָּלָק כַּאֲשֶׁר אָמַר בִּלְעָם וַיַּעַל פָּר וָאַיִל בַּמִּזְבֵּחַ: וַעֲבַד בָּלָק כְּמָא דִי אֲמַר בִּלְעָם וְאַסֵיק תּוֹר וּדְכַר עַל כָּל מַדְבְּחָא: כד (א) וַיַּרְא בִּלְעָם כִּי טוֹב בְּעֵינֵי יְהוָה לְבָרֵךְ אֶת יִשְׂרָאֵל וְלֹא הָלַךְ כְּפַעַם בְּפַעַם לִקְרַאת נְחָשִׁים וַיָּשֶׁת אֶל הַמִּדְבָּר פָּנָיו: וַחֲזָא בִלְעָם אֲרֵי תַקִין קֳדָם יְיָ לְבָרָכָא יַת יִשְׂרָאֵל וְלָא הַלַךְ כִּזְמַן בִּזְמַן אֱלָהֵין לָקֳדָמוּת נְחָשַׁיָא וְשַׁוִי לָקָבֵיל עֶגְלָא דַעֲבָדוּ יִשְׂרָאֵל בְּמַדְבְּרָא אַפּוֹהִי: (ב) וַיִּשָּׂא בִלְעָם אֶת עֵינָיו וַיַּרְא אֶת יִשְׂרָאֵל שֹׁכֵן לִשְׁבָטָיו וַתְּהִי עָלָיו רוּחַ אֱלֹהִים: וּזְקֵף בִּלְעָם יָת עֵינוֹהִי וַחֲזָא יָת יִשְׂרָאֵל שָׁרָן לְשִׁבְטוֹהִי וּשְׁרַת עֲלוֹהִי רוּחַ נְבוּאָה מִן קֳדָם יְיָ: (ג) וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר נְאֻם בִּלְעָם בְּנוֹ בְעֹר וּנְאֻם הַגֶּבֶר שְׁתֻם הָעָיִן: וּנְטַל מַתְלֵיהּ וַאֲמַר אֵימַר בִּלְעָם בְּרֵיהּ בְעוֹר וְאֵימַר גַבְרָא דְשַׁפִּיר חָזֵי: (ד) נְאֻם שֹׁמֵעַ אִמְרֵי אֵל אֲשֶׁר מַחֲזֵה שַׁדַּי יֶחֱזֶה נֹפֵל וּגְלוּי עֵינָיִם: אֵימַר דִשְׁמַע מֵימַר מִן קֳדָם אֵל וְחֵזוּ מִן קֳדָם שַׁדַי חָזֵי שְׁכִיב וּמִתְגְלֵי לֵיהּ: (ה) מַה טֹּבוּ אֹהָלֶיךָ יַעֲקֹב מִשְׁכְּנֹתֶיךָ יִשְׂרָאֵל: מָה טָבָא מַשְׁכְּנָךְ יַעֲקֹב בֵּית מֵישְׁרָךְ יִשְׂרָאֵל: (ו) כִּנְחָלִים נִטָּיוּ כְּגַנֹּת עֲלֵי נָהָר כַּאֲהָלִים נָטַע יְהוָה כַּאֲרָזִים עֲלֵי מָיִם: כְּנַחֲלִין דִמְדַבְּרִין כְּגִינַת שַׁקְיָא דְעַל פְּרָת כְּבוּסְמַיָא דִנְצִיב יְיָ כְּאַרְזִין דִנְצִבִין עַל מַיָא: (ז) יִזַּל מַיִם מִדָּלְיָו וְזַרְעוֹ בְּמַיִם רַבִּים וְיָרֹם מֵאֲגַג מַלְכּוֹ וְתִנַּשֵּׂא מַלְכֻתוֹ: יִסְגֵי מַלְכָּא דְיִתְרַבָּא מִבְּנוֹהִי וְיִשְׁלוֹט בְּעַמְמִין סַגִיאִין וְיִתְקוֹף מֵאֲגַג מַלְכֵּיהּ וְתִתְנַטֵל יֵיכְלוּן בְּנֵי יִשְׂרָאֵל נִכְסֵי עַמְמַיָא סַנְאֵיהוֹן וּבְבִזַת מַלְכֵּיהוֹן יִתְפַּנְקוּן וְאַרְעֲתְהוֹן יַחְסְנוּן: (ט) כָּרַע שָׁכַב כַּאֲרִי וּכְלָבִיא מִי יְקִימֶנּוּ מְבָרֲכֶיךָ בָרוּךְ אַרְבָּעָה וְעֶשְׂרִים אָלֶף: וַהֲווֹ דְמִיתוּ בְּמוֹתָנָא עַשְׂרִין וְאַרְבְּעָה אַלְפִין: פפפ:     רש"י

    אוקבנו לי. לשון צווי, קללהו לי: שדה צופים. מקום גבוה היה ששם הצופה עומד לשמור אם יבא חיל על העיר: ראש הפסגה. בלעם לא היה קוסם כבלק, ראה בלק שעתידה פרצה להפרץ בישראל משם ששם מת משה. כסבור ששם תחול עליהם הקללה, וזו היא הפרצה שאני רואה: אקרה כה. מאת הקדוש ברוך הוא: אקרה. לשון אתפעל: וישם דבר בפיו. ומה היא השימה הזאת, ומה חסר המקרא באמרו שוב אל בלק וכה תדבר, אלא כשהיה שומע שאינו נרשה לקלל, אמר מה אני חוזר אצל בלק לצערו. ונתן לו הקדוש ברוך הוא רסן וחכה בפיו כאדם הפוקס בהמה בחכה להוליכה אל אשר ירצה. אמר לו על כרחך תשוב אל בלק: ושרי מואב אתו. ולמעלה הוא אומר וכל שרי מואב, כיון שראו שאין בו תקוה הלכו להם מקצתם, ולא נשארו אלא מקצתם: מה דבר ה'. לשון צחוק הוא זה, כלומר אינך ברשותך: קום בלק. כיון שראהו מצחק בו, נתכון לצערו עמוד על רגליך, אינך רשאי לישב ואני שלוח אליך בשליחותו של מקום: בנו צפר. לשון מקרא הוא זה, כמו (תהלים קד, כ) חיתו יער, (בראשית א, כד) וחיתו ארץ, (תהלים קיד, ח) למעינו מים: לא איש אל וגו'. כבר נשבע להם להביאם ולהורישם ארץ שבעה אמות, ואתה סבור להמיתם במדבר: ההוא אמר וגו'. בלשון תמה, ותרגומו תיבין ומתמלכין, חוזרים ונמלכין לחזור בהם: הנה ברך לקחתי. אתה שואלני מה דבר ה', קבלתי ממנו לברך אותם: וברך ולא אשיבנה. הוא ברך אותם ואני לא אשיב את ברכתו: וברך. כמו וברך וכן הוא גזרת רי''ש, כמו (תהלים עד, יח) אויב חרף, כמו חרף, וכן (שם י, ג) ובוצע ברך, המהלל ומברך את הגוזל ואומר, אל תירא כי לא תענש, שלום יהיה לך, מרגיז הוא להקדוש ברוך הוא. ואין לומר ברך שם דבר, שאם כן היה נקוד בפתח קטן וטעמו למעלה, אבל לפי שהוא לשון פעל הוא נקוד קמץ, וטעמו למטה: לא הביט און ביעקב וגו'. כתרגומו. דבר אחר אחרי פשוטו הוא נדרש מדרש נאה. לא הביט. הקדוש ברוך הוא און שביעקב, כשהן עוברין על דבריו אינו מדקדק אחריהם להתבונן באוניות [באון] שלהם ובעמלן שהן עוברין על דתו: עמל. לשון עברה, כמו והרה עמל (תהלים ז, טו), כי אתה עמל וכעס תביט (שם י, יד) לפי שהעברה היא עמל לפני המקום: ה' אלהיו עמו. אפלו מכעיסין וממרים לפניו אינו זז מתוכן: ותרועת מלך בו. לשון חבה ורעות כמו רעה דוד (שמואל ב' טו, לז) אוהב דוד, ויתנה למרעהו (שופטים טו, ו) וכן תרגם אנקלוס ושכינת מלכהון ביניהון: אל מוציאם ממצרים. אתה אמרת הנה עם יצא ממצרים, לא יצא מעצמו אלא האלהים הוציאם: כתועפות ראם לו. כתוקף רום וגובה שלו, וכן וכסף תועפות (איוב כב כה), לשון מעוז המה, ואומר אני שהוא לשון ועוף יעופף (בראשית א כ), המעופף ברום וגובה ותקף רב הוא זה ותועפות ראם עפיפת גובה. דבר אחר, תועפות ראם. תקף ראמים, ואמרו רבותינו אלו השדים: כי לא נחש ביעקב. כי ראוים הם לברכה שאין בהם מנחשים וקוסמים: כעת יאמר ליעקב וגו'. עוד עתיד להיות עת כעת הזאת אשר תגלה חבתן לעין כל שהן יושבין לפניו ולומדים תורה מפיו ומחצתן לפנים ממלאכי השרת, והם ישאלו להם מה פעל אל, וזהו שנאמר והיו עיניך רואות את מוריך (ישעיה ל, כ). דבר אחר יאמר ליעקב אינו לשון עתיד, אלא לשון הוה, אינן צריכין למנחש וקוסם כי בכל עת שצריך להאמר ליעקב ולישראל מה פעל הקדוש ברוך הוא ומה גזרותיו במרום, אינן מנחשים וקוסמים אלא נאמר להם על פי נביאיהם מה היא גזרת המקום, או אורים ותמים מגידים להם, ואונקלוס לא תרגם כן: הן עם כלביא יקום וגו'. כשהן עומדים משנתם שחרית, הן מתגברין כלביא וכארי לחטוף את המצות, ללבוש טלית לקרא את שמע ולהניח תפלין: לא ישכב. בלילה על מטתו עד שהוא אוכל ומחבל כל מזיק הבא לטרפו. כיצד, קורא את שמע על מטתו ומפקיד רוחו ביד המקום, בא מחנה וגיס להזיקם, הקדוש ברוך הוא שומרם ונלחם מלחמותם ומפילם חללים. דבר אחר הן עם כלביא יקום וגו'. כתרגומו: ודם חללים ישתה. נתנבא שאין משה מת עד שיפיל מלכי מדין חללים ויהרג הוא עמהם, שנאמר (יהושע יג, כב) ואת בלעם בן בעור הקוסם הרגו בני ישראל בחרב אל חלליהם: גם קב לא תקבנו. גם ראשון מוסיף על גם השני, וגם השני על גם ראשון, וכן גם לי גם לך לא יהיה (מלכים א' ג, כו), וכן גם בחור גם בתולה (דברים לב, כה): וקבותו לי. אין זה לשון צווי, כמו וקבנו, אלא לשון עתיד, אולי יישר בעיניו ותקבנו לי משם, מלדי''רש מי''י בלע''ז [תקלל אותו]: ראש הפעור. קוסם היה בלק וראה שהן עתידין ללקות על ידי פעור ולא היה יודע במה, אמר שמא הקללה תחול עליהם משם, וכן כל החוזים בכוכבים רואים ואינם יודעים מה רואים: וירא בלעם כי טוב וגו'. אמר איני צריך לבדוק בהקדוש ברוך הוא כי לא יחפוץ לקללם: ולא הלך כפעם בפעם. כאשר עשה שתי פעמים: לקראת נחשים. לנחש אולי יקרה ה' לקראתו כרצונו, אמר רוצה ואינו רוצה לקללם, אזכיר עונותיהם, והקללה על הזכרת העברה תחול: וישת אל המדבר פניו. כתרגומו: וישא בלעם את עיניו. בקש להכניס בהם עין רעה, והרי יש לך שלש מדותיו עין רעה ורוח גבוהה ונפש רחבה האמורים למעלה: שוכן לשבטיו. ראה כל שבט ושבט שוכן לעצמו ואינן מעורבין, ראה שאין פתחיהם מכונין זה כנגד זה, שלא יציץ לתוך אהל חברו: ותהי עליו רוח אלהים. עלה בלבו שלא יקללם: בנו בער. כמו למעינו מים (תהלים קיד, ח). ומדרש אגדה שניהם היו גדולים מאבותיהם, בלק בנו צפור, אביו בנו הוא במלכות, ובלעם גדול מאביו בנביאות, מנה בן פרס היה: שתם העין. עינו נקורה ומוצאת לחוץ וחור שלה נראה פתוח, ולשון משנה הוא כדי שישתום ויסתום ויגוב (ע''ז סט.) ורבותינו אמרו לפי שאמר ומספר את רבע ישראל, שהקדוש ברוך הוא יושב ומונה רביעיותיהן של ישראל מתי תבא טפה שנלד הצדיק ממנה, אמר בלבו מי שהוא קדוש ומשרתיו קדושים יסתכל בדברים הללו, ועל דבר זה נסמת עינו של בלעם. ויש מפרשים שתום העין פתוח העין, כמו שתרגם אנקלוס, ועל שאמר שתום העין, ולא אמר שתום העינים, למדנו שסומא באחת מעיניו היה: נפל וגלוי עינים. פשוטו כתרגומו, שאין נראה עליו אלא בלילה כשהוא שוכב. ומדרשו כשהיה נגלה עליו לא היה בו כח לעמוד על רגליו ונופל על פניו, לפי שהיה ערל ומאוס להיות נגלה עליו בקומה זקופה לפניו: מה טבו אהליך. על שראה פתחיהם שאינן מכונין זה מול זה: משכנתיך. חניותיך, כתרגומו. דבר אחר מה טבו אהליך. מה טבו אהל שילה ובית עולמים בישובן שמקריבין בהן קרבנות לכפר עליהם: משכנתיך. אף כשהן חרבין, לפי שהן משכון עליכם, וחרבנן כפרה על הנפשות, שנאמר כלה ה' את חמתו (איכה ד), ובמה כלה, ויצת אש בציון: כנחלים נטיו. שנארכו ונמשכו לנטות למרחוק, אמרו רבותינו מברכותיו של אותו רשע אנו למדים מה היה בלבו לקללם כשאמר (פסוק א) וישת אל המדבר פניו, וכשהפך המקום את פיו ברכם מעין אותן קללות שבקש לומר כו', כדאיתא בחלק (סנהדרין קה, א): כאהלים. כתרגומו, לשון מור ואהלות (תהלים מה, ט): נטע ה'. בגן עדן. לשון אחר כאהלים נטע ה', כשמים המתוחין כאהל: נטע ה'. לשון נטיעה מצינו באהלים, שנאמר ויטע אהלי אפדנו (דניאל יא, מה): מדליו. מבארותיו. ופירושו כתרגומו: וזרעו במים רבים. לשון הצלחה הוא זה, כזרע הזרוע על פני המים: וירם מאגג מלכו. מלך ראשון שלהם יכבוש את אגג מלך עמלק: ותנשא מלכותו. של יעקב יותר ויותר, שיבא אחריו דוד ושלמה: אל מוציאו ממצרים. מי גורם להם הגדלה הזאת, אל המוציאם ממצרים, בתוקף ורום שלו יאכל את הגוים שהם צריו: ועצמותיהם. של צרים: יגרם. מנחם פתר בו לשון שבירה, וכן (צפניה ג, ג) לא גרמו לבקר, וכן (יחזקאל כג, לד) את חרשיה תגרמי, ואני אומר לשון עצם הוא שמגרר הבשר בשניו מסביב והמוח שבפנים, ומעמיד העצם על ערמימותה: וחציו ימחץ. אנקלוס תרגם חציו של צרים, חלוקה שלהם כמו (בראשית מט, כג) בעלי חצים, מרי פלוגתא, וכן ימחץ, לשון (שופטים ה, כו) ומחצה וחלפה רקתו, שיחצו את ארצם. ויש לפתור לשון חצים ממש, חציו של הקדוש ברוך הוא ימחץ בדמם של צרים, יטבול ויצטבע בדמם, כמו (תהלים סח, כד) למען תמחץ רגלך בדם. ואינו זז מלשון מכה, כמו מחצתי, שהצבוע בדם נראה כאלו מחוץ ונגוע: כרע שכב כארי. כתרגומו, יתישבו בארצם בכח וגבורה: ויספק. הכה זו על זו: לעבר את פי ה'. כאן לא נאמר אלהי, כמו שנאמר בראשונה, לפי שידע שנבאש בהקדוש ברוך הוא ונטרד: הולך לעמי. מעתה הריני כשאר עמי, שנסתלק הקדוש ברוך הוא מעליו: לכה איעצך. מה לך לעשות. ומה היא העצה, אלהיהם של אלו שונא זמה הוא כו', כדאיתא בחלק (סנהדרין קו.) תדע שבלעם השיא עצה זו להכשילם בזמה, שהרי נאמר הן הנה היו לבני ישראל בדבר בלעם (לקמן לא, טז): אשר יעשה העם הזה לעמך. מקרא קצר הוא זה, איעצך להכשילם, ואומר לך מה שהן עתידין להרע למואב באחרית הימים: וידע דעת עליון. לכון השעה שכועס בה: אראנו. רואה אני שבחו של יעקב וגדלתן, אך לא עתה היא, אלא לאחר זמן: דרך כוכב. כתרגומו, לשון דרך קשתו (איכה ב, ד), שהכוכב עובר כחץ, ובלע''ז דישטנ''ט [מזנק] כלומר יקום מזל: וקם שבט. מלך רודה ומושל: ומחץ פאתי מואב. זה דוד, שנאמר בו השכב אותם ארצה וימדד שני חבלים להמית וגו' (שמואל ב' ח, ב): וקרקר. לשון קורה כמו אני קרתי (מלכים ב' יט, כד) מקבת בור נקרתם (ישעיה נא, א) יקרוה עורבי נחל (משלי ל, יז) פורי''יר בלע''ז [לנקב]: כל בני שת. כל האמות, שכלם יצאו מן שת בנו של אדם הראשון: והיה ירשה שעיר איביו. ישראל: וירד מיעקב. ועוד יהיה מושל אחר מיעקב: והאביד שריד מעיר. מעיר החשובה של אדום והיא רומי, ועל מלך המשיח אומר כן, שנאמר בו וירד מים עד El verbo וְקָבְנוֹ está en el modo imperativo, queriendo decir: maldícelo por mí.  14. שְׂדֵה צפִים / SEDÉ TZOFIM. Era un sitio elevado donde se paraba un atalaya para observar si una tropa enemiga venía contra la ciudad.  ראשׁ הַפִּסְגָּה / LA PUNTA DE LA CUMBRE. Bilam no era hechicero como lo era Balak. Por medio de sus artes mágicas, Balak vislumbró que en el futuro se abriría una brecha en Israel en ese sitio. Y tenía razón, pues de hecho ahí mismo murió Moshé. Así, pues, pensó que allí recaería sobre ellos la maldición mejor que en otro sitio, y se dijo: “Esa debe ser la brecha que yo veo”. 15. אִקָּרֶה כּה / [Y YO] SERÉ ENCONTRADO FORTUITAMENTE AQUÍ. Por el Santo –bendito es.  אִקָּרֶה / SERÉ ENCONTRADO FORTUITAMENTE. Este verbo está conjugado en el modo pasivo. 16. וַיָּשֶׂם דָּבָר בְּפִיו / Y PUSO UN ENUNCIADO EN SU BOCA. ¿A qué se refiere esta “puesta” en su boca? ¿Y qué le pudo haber faltado decir a la Escritura cuando declaró que Di-s “dijo: Regresa a Balak y hablarás así”? En realidad, la explicación es la siguiente: cuando Bilam escuchó que no se le había concedido permiso para maldecir a Israel, se dijo: “¿Para qué he de regresar con Balak para afligirlo con estas malas noticias? Entonces Di-s le puso un freno y un gancho en la boca, como un hombre que engancha a un animal con un gancho a fin de conducirlo a cualquier lugar que desee. En efecto, Di-s le dijo: “Incluso contra tu voluntad has de regresar a Balak”.  17. וְשָׂרֵי מוֹאָב אִתּוֹ / Y LOS MINISTROS DE MOAB CON ÉL. Pero más arriba la Escritura declara que Balak estaba con “todos los ministros de Moab”. La explicación es la siguiente: cuando ellos vieron que no había esperanza en Bilam de que maldijese a Israel, una parte de ellos se fue, quedando sólo unos pocos.  מַה דִּבֵּר ה' / ¿QUÉ HABLÓ EL ETERNO? Esto lo dijo en un sentido burlón, como diciendo: “De hecho no estás tú bajo tu propia autoridad”. 18. קוּם בָּלָק / ¡LEVÁNTATE, BALAK! Cuando Bilam lo vio burlándose de él, quiso ofenderlo, diciendo: “¡Párate sobre tus pies! ¡No tienes derecho a permanecer sentado mientras que yo soy enviado a ti con una misión de parte del Omnipresente!”  בְּנוֹ צִפר / HIJO DE TZIPOR. Esta forma gramatical es propia del lenguaje hebreo de la Escritura. Es similar a los casos siguientes: “La fiera del [חַיְתוֹ] bosque”; “la fiera de [חַיְתוֹ] la tierra”; “a la fuente de [לְמַעְיְנוֹ] agua”. 19. לא אִישׁ וְגוֹ' / [DI-S] NO ES HOMBRE, ETC. Es decir, Él ya les ha jurado llevarlos a la tierra de las siete naciones de Kenaan y entregársela en posesión; ¿y tú piensas que podrás matarlos en el desierto?  הַהוּא אָמַר וְגוֹ' / ÉL DIJO [¿Y NO HARÁ?] Esta frase tiene sentido interrogativo. Y la traducción del Targum de la palabra וְיִתְנֶחָם es וְתָיְבִין וּמִתְמַלְּכִין, y vuelven a reconsiderar. Quiere decir que los seres humanos reconsideran para retractarse de su decisión. 20. הִנֵּה בָרֵךְ לָקָחְתִּי / HE AQUÍ QUE PARA BENDECIR HE SIDO TOMADO. Como diciendo: Tú me preguntaste en tono de burla qué fue lo que habló el Eterno. Pues bien, recibí de Él el mandato de bendecirlos.  {בָרֵךְ / PARA BENDECIR. Esta forma del verbo es equivalente a לְבָרֵךְ.}  וּבֵרֵךְ וְלא אֲשִׁיבֶנָּה / ÉL HA BENDECIDO, Y YO NO LO REVOCARÉ. Esta frase significa: Él los ha bendecido y yo no puedo revocar Su bendición.  וּבֵרֵךְ / ÉL HA BENDECIDO. Esta forma del verbo es equivalente a וּבִרֵךְ. Y así es la regla gramatical de la letra ר, similar a los siguientes casos: “El enemigo blasfemó [חֵרֵף]”, equivalente a חִרֵף; y lo mismo en: “Que bendice [בֵּרֵךְ] al deshonesto”, que significa: el que alaba y bendice al ladrón, diciéndole: “No tengas miedo, que no serás castigado; paz tendrás”, esa persona hace enojar al Santo –bendito es. Y no se puede decir que וּבֵרֵךְ es un sustantivo, pues de ser así hubiera estado vocalizado en ambas sílabas con pataj katán, y su acento hubiera estado en la primera sílaba. Pero puesto que es un verbo en el modo intensivo pi'el, está vocalizado con kamatz katán en ambas sílabas y su acento está en la última sílaba.  21. לא הִבִּיט אָוֶן בְּיַעֲקֹב וְגוֹ' / ÉL NO HA OBSERVADO INIQUIDAD EN YAACOB, ETC. Esta frase debe ser entendida como la traduce el Targum. Otra explicación: siguiendo su sentido llano, la frase puede ser bellamente interpretada según el Midrash: el Santo –bendito es– “no contempló” la iniquidad que hay en Yaacob. En otras palabras, cuando los israelitas transgreden Sus palabras, Él no es muy puntilloso respecto a ellos para considerar con severidad sus iniquidades y sus violaciones por medio de las que transgreden Su ley.  עָמָל / DELITO. Aquí este término significa delito, lo mismo que en las frases: “Concibió delito [עָמָל];” “pues Tú contemplas el delito [עָמָל] y la injuria”, ya que el delito constituye una pena frente al Omnipresente.  ה' אֱלהָיו עִמּוֹ / EL ETERNO, SU DI-S, ESTÁ CON ÉL. Es decir, incluso cuando le provocan a ira y se rebelan contra Él, Él no se va de en medio de ellos. וּתְרוּעַת מֶלֶךְ בּוֹ / Y LA ESTIMACIÓN DEL REY ESTÁ EN ÉL. La palabra וּתְרוּעַת denota cariño y amistad, lo mismo que en las frases: “El amigo [רֵעֶה] de David”, donde significa “el [amigo] querido de David”; “y se la dio a su amigo [לְמֵרֵעֵהוּ]”. Y en un sentido similar la tradujo el Targum de Onkelós: וּשְׁכִינַת מַלְכְּהוֹן בֵּינֵיהוֹן, y la Presencia de su Rey está entre ellos.  22. אֵל מוֹצִיאָם מִמִּצְרָיִם / ES DI-S [EL QUE] LOS SACÓ DE MITZRÁIM. Tú, Balak, dijiste: “He aquí que un pueblo ha salido de Mitzráim”. Debes saber que no salió por sí solo, sino que fue Di-s el que lo sacó.  כְּתוֹעֲפת רְאֵם לוֹ / CONFORME A LA INTENSIDAD DE SU EXCELENCIA. Esta frase significa: según la intensidad de Su excelencia y Su altura. Y de igual modo, כְּתוֹעֲפת denota “poder” en: “Y plata de poder [תּוֹעֲפוֹת].” Pero yo digo que כְּתוֹעֲפת está relacionado con el término: “Y ave que vuele [יְעוֹפֵף]”, donde designa a un ave que emprende el vuelo hacia los espacios altos y elevados, lo que implica una gran fuerza.  Por tanto, תוֹעֲפוֹת רְאֵם significa: “vuelo en las alturas”. Según otra explicación, תוֹעֲפֹת רְאֵם se refiere a la fuerza de los רְאֵמִים, reemim. Y nuestros Maestros dijeron que esto se refiere a los shedim.  23. כִּי לא נַחַשׁ בְּיַעֲקב / PUES NO HAY ADIVINACIÓN EN YAACOB. Es decir, pues los israelitas son dignos de la bendición divina, puesto que entre ellos no hay adivinadores ni hechiceros.  כָּעֵת יֵאָמֵר לְיַעֲקב וְגוֹ' / COMO EN ESTE MOMENTO SE DIRÁ A YAACOB ETC. En el futuro habrá un momento semejante a este momento en el que será revelado a la vista de todos los seres humanos el cariño que Di-s profesa a los israelitas, en el que ellos estarán sentados y aprenderán Torá de Su boca, estando su lugar “más al interior” que incluso los ángeles mismos. Y éstos preguntarán a los israelitas lo que se dice al final de este versículo: “¿Qué ha hecho Di-s?” A esto se refiere lo que se declara: “Y tus ojos contemplarán a tu Maestro [Di-s].” Según otra explicación, la frase יֵאָמֵר לְיַעֲקב no está en futuro, sino en presente, con lo cual el versículo estaría diciendo que ellos no necesitan de adivinos ni hechiceros, ya que en todo “momento” [עֵת] que sea necesario que “se diga” [יֵאָמֵר] a Yaacob y a Israel qué ha hecho el Santo –bendito es– y cuáles son Sus decretos en las Alturas, ellos no practican la adivinación ni la hechicería para saberlo, sino que por boca de sus profetas se les dice cuál es el decreto promulgado por el Omnipresente, o también, el Urim y Tumim se los declararía. Pero el Targum de Onkelós no lo tradujo así.  24. הֶן עָם כְּלָבִיא יָקוּם וְגוֹ' / HE AQUÍ UN PUEBLO QUE SE LEVANTA COMO LEÓN FUERTE. Cuando ellos se levantan en la mañana de su sueño, se fortalecen como el león fuerte y el león para “arrebatar” los mandamientos y cumplirlos: vestirse con tzitzit, recitar el Shemá Israel y colocarse los tefilín.  לא יִשְׁכַּב / NO SE ACOSTARÁ. De noche, sobre su cama, sino hasta que consume y destruye a todo ser dañino que viene para atacarlo. ¿De qué modo? Recitando el Shemá Israel en su cama y confiando su espíritu en manos del Omnipresente. De este modo, si viene un ejército o una tropa para causar daño a los israelitas, el Santo –bendito es– los protege y pelea sus batallas, haciendo caer muertos a los enemigos. Según otra explicación, todo este pasaje, comenzando con las palabras “He aquí un pueblo que se levanta como un león, etc.” debe ser entendido como lo traduce el Targum.  וְדַם חֲלָלִים יִשְׁתֶּה / Y BEBA LA SANGRE DE LOS MUERTOS. Aquí Bilam profetizó que Moshé no moriría sino hasta que hiciera caer muertos a los reyes de Midián, y que él mismo, Bilam, sería muerto junto con ellos, como se declara: “Y a Bilam hijo de Beor, el hechicero, los Hijos de Israel mataron con la espada entre sus cadáveres”.  25. גַּם קב לא תִקֳּבֶנּוּ / ¡NI LOS MALDIGAS [NI LOS BENDIGAS]! El primer גַּם añade un significado más al segundo, y el segundo גַּם también añade algo más al primer גַּם. Y otro caso análogo a éste lo vemos en: “Ni [גַּם] mío ni [גַּם] tuyo será”. Y asimismo en: “Ni [גַּם] joven ni [גַּם] doncella”.  27. וְקַבּתוֹ לִי / [QUIZÁ]... LO MALDIGAS POR MÍ. Aquí el verbo וְקַבֹּתוֹ no expresa el modo imperativo como el verbo וְקָבְנוֹ, sino que expresa un acto futuro. Balak quería decir: Quizá parezca correcto en los ojos de Di-s que tú los maldigas por mí desde allí. En francés antiguo, maldiras–lui.  28. ראשׁ הַפְּעוֹר / A LA CIMA DE PEOR. Balak era un gran adivino, y vislumbró que en el futuro los israelitas serían castigados a causa del ídolo Peor, sólo que no sabía por medio de qué. Por eso dijo: “Quizá la maldición caiga sobre ellos desde allí”. Y lo mismo que Balak, todos los que contemplan los astros para adivinar el futuro, ven que algo ocurrirá, pero no saben a ciencia cierta qué es lo que ven. 1. וַיַרְא בִּלְעָם כִּי טוֹב וְגוֹ' / BILAM VIO QUE ERA BUENO, ETC. Se dijo: “Ya no necesito averiguar del Santo –bendito es. Es evidente que Él no desea maldecirlos”.  וְלא הָלַךְ כְּפַעַם בְּפַעַם / ASÍ QUE NO MARCHÓ COMO LAS VECES ANTERIORES. Esta frase significa: como había hecho las dos veces anteriores.  לִקְרַאת נְחָשִׁים / AL ENCUENTRO DE ADIVINACIONES. Es decir, para practicar adivinaciones. Quizá el Eterno se le apareciera fortuitamente según era el deseo de Bilam. Él se dijo: “Ya sea que Di-s quiera o no quiera maldecirlos, yo haré mención de sus iniquidades, y la maldición recaerá sobre ellos a la mención misma de sus iniquidades”.  וַיָּשֶׁת אֶל הַמִּדְבָּר פָּנָיו / Y DIRIGIÓ SU ROSTRO HACIA EL DESIERTO. Esto debe ser entendido como lo traduce el Targum.  2. וַיִּשָּׂא בִלְעָם אֶת עֵינָיו / BILAM ALZÓ SUS OJOS. Lo hizo porque intentó meter en ellos el mal de ojo [ain hará]. Con este último acto tienes ya la descripción de las tres características esenciales de Bilam: el ojo maligno descrito aquí, así como el espíritu arrogante y el alma ávida mencionados más arriba.  שׁכֵן לִשְׁבָטָיו / RESIDIENDO SEGÚN SUS TRIBUS. Vio que cada tribu residía por sí misma y no estaba revuelta con las demás. También vio que las entradas de las tiendas de cada uno no estaban dirigidas directamente entre sí a fin de que nadie pudiera atisbar al interior de la tienda de su prójimo.  וַתְּהִי עָלָיו רוּחַ אֱלֹהִים / Y EL ESPÍRITU DIVINO ESTUVO SOBRE ÉL. Esto implica que decidió en su corazón no maldecirlos.  3. בְּנוֹ בְער / HIJO DE BEOR. Esta forma gramatical es análoga a la siguiente: “A la fuente de [לְמַעְיְנוֹ] agua.” Pero según una exégesis agádica, ambos –Balak y Bilam– eran más destacados que sus respectivos padres. En referencia a Balak, la Escritura dice בָּלָק... בְּנוֹ צִפֹּר, que puede ser leído como “Balak... su hijo es Tzipor”, implicando que su padre era su “hijo” en términos de rango real. Y, de igual modo, בִּלְעָם בְּנוֹ בְער también puede ser entendido como “Bilam, su hijo es Beor”, queriendo decir que Bilam era más grande que su padre en términos de don profético. Como dice el refrán, era “un mané hijo de un perás”.  שְׁתֻם הָעָין / PERFORADO EN EL OJO. Se describió de este modo porque su ojo había sido perforado y extraído, y la cavidad de su ojo parecía abierta. El vocablo שְׁתֻם es propio del lenguaje de la Mishná, como vemos en la frase: “[Tiempo suficiente] para que perfore un hoyo [יִּשְׁתּוֹם], lo tape y el tapón se seque.” Y nuestros Maestros dijeron que porque Bilam había dicho “¿Quién ha contado el polvo de Yaacob y el número de la procreación de Israel?” aludiendo al hecho de que, por decirlo así, el Santo –bendito es– se sienta y cuenta las relaciones maritales de Israel, esperando el momento en que llegue la gota de semen del cual nazca un varón justo, ello le asombró tanto que dijo en su corazón: “Aquel que es Santo y cuyos servidores [los ángeles] son santos, ¿es posible que mire estas cosas?” Y a causa de esto el ojo de Bilam se cerró. {Y hay quienes explican que la frase שְׁתֻם הָעָין quiere decir “abierto de ojo”, como la traduce el Targum de Onkelós.} Y puesto que dijo “abierto de ojo” y no “abierto de ojos”, aprendemos que era ciego de uno de sus ojos.  4. נפֵל וּגְלוּי עֵינָיִם / CAÍDO, MAS CON OJOS DESCUBIERTOS. El sentido llano de esta frase es como lo traduce el Targum, queriendo decir que Di-s solamente se le revelaba de noche mientras estaba acostado en su lecho. Y su interpretación midráshica es la siguiente: Cuando Di-s se le revelaba, Bilam no poseía la fuerza suficiente para mantenerse parado, por lo que caía sobre su rostro debido a la intensidad de la revelación, ya que era incircunciso y, por ello, era repugnante que Di-s se le revelase estando él en postura erecta delante de Él.  5. מַה טּבוּ אהָלֶיךָ / ¡QUÉ TAN BUENAS SON TUS TIENDAS! Exclamó esto debido a que vio que las entradas de las tiendas de los israelitas no estaban dirigidas una frente a la otra.  מִשְׁכְּנֹתֶיךָ / TUS RESIDENCIAS. Esto significa “tus campamentos”, como lo traduce el Targum. Según otra explicación, la frase “¡Qué buenas son tus tiendas, Yaacob!” quiere decir: ¡Qué buenas eran la “Tienda de Shiló” y la Casa Eterna cuando existieron! Eran buenas porque en esos sitios ustedes ofrecían ofrendas a Di-s para hacer expiación por ustedes. Y la expresión “tus residencias” significa que son “buenas” incluso ahora que están destruidas, ya que ellas constituyen garantía para ustedes, y el hecho de que estén destruidas es expiación para sus almas, como se declara: “El Eterno ha agotado Su ira”. ¿Y gracias a qué Su ira se ha agotado? Gracias a que “Él ha prendido fuego a Tzión.”  6. כִּנְחָלִים נִטָּיוּ / COMO ARROYOS SE EXTIENDEN. Es decir, que se alargan y extienden prolongándose hacia lo lejos. Respecto a las bendiciones que Bilam pronunció, nuestros Maestros dijeron: A partir de las bendiciones de ese malvado, inferimos la intención que realmente estaba en su corazón maldecirlos, cuando “dirigió su rostro hacia el desierto”. Y cuando Di-s cambió las palabras de su boca , Bilam los bendijo con bendiciones similares a las maldiciones que quería pronunciar, etc., como se declara en el capítulo talmúdico “Jélek”. כַּאֲהָלִים / COMO ÁLOES. Esta palabra debe ser entendida como la traduce el Targum, y está relacionada con la frase: “Mirra y áloes [וַאֲהָלוֹת]”.  נָטַע ה' / QUE EL ETERNO HA PLANTADO. En el Jardín del Edén. Según otra explicación, la frase כַּאֲהָלִים נָטַע ה' significa que son como el cielo, que está extendido como una “tienda” [אהֵל] sobre la tierra, como se declara: “[Di-s] tensó [los cielos] como una tienda para habitar.”  נָטַע ה' / QUE EL ETERNO HA PLANTADO. Hallamos la expresión “plantar” respecto a tiendas, como se declara: “Y plantará las tiendas de su palacio”.  7. מִדָּלְיָו / DE SUS POZOS. Esta palabra significa: de sus fuentes. Y su explicación es tal como la traduce el Targum.  וְזַרְעוֹ בְּמַיִם רַבִּים / Y SU SEMILLA ESTARÁ EN AGUAS ABUNDANTES. Esta frase expresa prosperidad. Quiere decir que será como la semilla sembrada en terreno de aguas. וְיָרם מֵאֲגַג מַלְכּוֹ / SU REY SERÁ EXALTADO MÁS QUE AGAG. Esta profecía quiere decir que el primer rey de los israelitas vencerá a Agag, rey de Amalek.  וְתִנַּשֵּׂא מַלְכֻתוֹ / Y SU REINADO SERÁ ENALTECIDO. De Yaacob, más y más aún, ya que después del rey Shaúl vendrán los reyes David y Shelomó.  8. אֵל מוֹצִיאוֹ מִמִּצְרַיִם / DI-S, [QUE] LOS SACÓ DE MITZRÁIM. ¿Quién fue la causa de esta grandeza? Di-s, que los sacó de Mitzráim por medio de Su poder y de Su excelencia; Él consumirá a los pueblos que sean sus opresores.  וְעַצְמֹתֵיהֶם / SUS HUESOS. De los opresores. יִגָרֵם / ÉL ROERÁ. Menajem ben Saruk explicó este término en el sentido de quebrar. Y asimismo en la frase: “No rompieron [גָרְמוּ] en la mañana”.  Y también en: “Y quebrarás [תְּגָרֵמִי] sus tiestos”.  Pero yo digo que significa hueso. Quiere decir que con los dientes se arranca la carne que está alrededor del hueso, junto con el tuétano que está al interior, dejando el hueso desnudo de carne y de tuétano.  וְחִצָּיו יִמְחָץ / Y SUS SAETAS LOS TRASPASARÁN. El Targum de Onkelós tradujo esta frase en referencia a las חִצָּיו de los opresores, y se refiere a la “repartición de ellos”, de igual modo que tradujo la frase בַּעֲלֵי חִצִים por מָרֵי פַלוּגְתָא, los que se dividen de él. Y en forma análoga, יִמחָץ significa lo mismo que en la frase: “Y ella partió [וּמָחֲצָה] y atravesó sus sienes”.  Para el Targum esta frase quiere decir que los israelitas dividirán su tierra. Pero también puede ser explicado en el sentido literal de חִצִּים, “flechas”. Según esto, la frase significa que el Santo –bendito es– teñirá [יִמְחָץ] sus flechas en la sangre de los opresores.  La palabra יִמְחָץ significa que Él remojará y teñirá las flechas en su sangre.  Similar a esta interpretación tenemos: “Para que tiñas [תִּמְחַץ] tus pies en sangre.” Pero aun así el verbo יִמְחָץ no se aparta del significado de “golpe” o herida. Es similar al verbo: “Yo aplasté [מָחַצְתִּי]”, ya que el que está teñido en sangre da la apariencia de estar herido y golpeado.  9. כָּרַע שָׁכַב כַּאֲרִי / SE AGAZAPÓ, SE ECHÓ COMO UN LEÓN. Esta frase debe ser entendida como la traduce el Targum, queriendo decir que los israelitas se asentarán en su tierra con poder y fuerza.  10. וַיִּסְפּק / BATIÓ [SUS PALMAS]. Esto significa que golpeó sus manos una contra la otra.  13. לַעֲבר אֶת פִּי ה' / TRANSGREDIR LA PALABRA DEL ETERNO. Aquí Bilam no dijo “mi Di-s”, como se declara que dijo la primera vez, ya que se había dado cuenta de que se había vuelto despreciable ante el Santo –bendito es– y había sido expulsado de Su Presencia.  14. הוֹלֵךְ לְעַמִּי / MARCHO A MI PUEBLO. Quería decir: Desde ahora soy como los demás miembros de mi pueblo, puesto que el Santo –bendito es– se ha alejado de mí.  לְכָה אִיעָצְךָ / VEN, TE ACONSEJARÉ. Qué es lo que puedes hacer contra Israel. ¿Y en qué consistió el consejo? Le dijo: “Su Di-s [de Israel] odia la promiscuidad sexual. Haz que pequen por medio de las mujeres de tu pueblo”, como se declara en el capítulo talmúdico “Jélek”. Sabe que, sin lugar a dudas, fue Bilam el que dio este consejo de hacerlos tropezar por medio de promiscuidad, puesto que respecto a las mujeres de Moab se declara: “He aquí que éstas fueron las que provocaron que los Hijos de Israel cometieran traición contra el Eterno a través de la palabra de Bilam”. אֲשֶׁר יַעֲשֶׂה הָעָם הַזֶּה לְעַמְּךָ / LO QUE ESTA NACIÓN HARÁ A TU NACIÓN. Este pasaje es elíptico, y en realidad quiere decir: “Ven, te aconsejaré cómo hacer que tropiecen [pecando]. Y además te diré lo que en el futuro ellos harán de mal a tu pueblo Moab al final de los días: ‘Traspasará a los nobles de Midián’”. Y el Targum explica lo que está implicado en el lenguaje elíptico del hebreo.  16. וְידֵעַ דַּעַת עֶלְיוֹן / Y CONOCE LA MENTE DEL ALTÍSIMO. En el sentido de saber determinar con precisión la hora en que Él se enoja.  17. אֶרְאֶנּוּ / LO VEO. Yo veo la cualidad de Yaacob y su grandeza, pero ésta no se expresa plenamente ahora, sino que lo hará después de un cierto tiempo.  דָּרַךְ כּוֹכָב / UN ASTRO SURGIRÁ.     Tratado Ohalot, capítulo 14

    (1) El voladizo que se encuentra arriba de la entrada, aunque sobresalga lo más mínimo, atrae impureza a la casa. El balcón y el tejado, atraen impureza de su ancho un téfaj como mínimo. ¿Qué es el voladizo? Aquel que su faz se encuentra hacia abajo. ¿Y el balcón? El que su faz se encuentra hacia arriba. Pero de encontrarse más alto a esta medida, atrae impureza a la casa únicamente de tener la medida de un téfaj de ancho. La diadema y los adornos, atraen la impureza de tener la medida de un téfaj de ancho como mínimo. (2) El voladizo que se encuentra arriba de la entrada atrae impureza únicamente de ser su ancho como mínimo un téfaj. El voladizo que se encuentra arriba de la ventana de la medida de dos dedos o de la ventana de la medida de la perforadora, atrae impureza de ser su medida aunque sea lo más mínimo. Rabí Yosé dice: Únicamente de ser su medida como la de la ventana.  (3) Las cañas que se encuentran colocadas arriba de la entrada, aunque estén a cien amá de altura, atraen la impureza aunque resalte de la pared en lo más mínimo, estas son las palabras de Rabí Yehoshúa. Rabí Yojanán dice: No es más estricto este caso que en el voladizo.  (4) El voladizo que rodea toda la casa y en la entrada solamente ocupa la medida de tres dedos, de haber alguna impureza en la casa, todo lo que se encuentra bajo el voladizo es impurificado. De encontrarse alguna impureza debajo del voladizo, la casa conserva su pureza, estas son las palabras de Rabí Yehoshúa. Rabí Eliézer dice: La casa es impurificada. De igual manera se aplica en caso de un patio que se encuentra rodeado por un atrio.  (5) Dos voladizos de la medida de un téfaj de ancho que se encuentran uno arriba del otro y entre ellos el espacio de un téfaj. De encontrarse alguna impureza debajo de ellos, todo lo que se encuentra debajo de ellos es impurificado. De encontrarse alguna impureza entre ellos, todo lo que se encuentra allí es impurificado. De encontrarse arriba de ellos, todo lo que se encuentra arriba de ellos hasta la inmensidad es impurificado. Si sobresalía el voladizo superior del inferior en la medida de un téfaj y se encontraba alguna impureza debajo o entre ellos, todo lo que se encuentra debajo o entre ellos es impurificado. De encontrarse alguna impureza arriba del superior, todo lo que se encuentra arriba de él hasta la inmensidad es impurificado. Si sobresalía el voladizo superior al inferior en menos de la medida de un téfaj y se encuentra alguna impureza debajo del inferior, todo lo que se encuentra debajo y entre ellos es impurificado. De encontrarse alguna impureza debajo de la parte sobresaliente o entre ellos, todo lo que se encuentra debajo o entre ellos es impurificada, estas son las palabras de Rabí Eliézer. Rabí Yehoshúa dice: Solamente es impurificado lo que se encuentra debajo de la parte sobresaliente y entre ellos. Pero lo que se encuentra debajo del voladizo inferior conserva su pureza.  (6) De ser los dos voladizos de la medida de un téfaj de ancho Pero entre ellos no hay la medida de un téfaj y se encuentra alguna impureza debajo de ellos, todo lo que se encuentra abajo y entre ellos es impurificado. De encontrarse alguna impureza arriba o entre ellos, todo lo que se encuentra arriba de ellos hasta la inmensidad, es impurificado.  (7) En caso que los voladizos no tienen la medida de un téfaj de ancho, ya sea se encuentra distanciado uno de otro en la medida de un téfaj o no, ya sea se encuentra alguna impureza debajo, entre o arriba de ellos, la impureza asciende y desciende. De igual manera en el caso de dos cueros que se encuentran elevados del suelo en la medida de un téfaj.

    1. AQUEL QUE SU FAZ SE ENCUENTRA HACIA ABAJO. Es esto como una lámina que sobresale de la pared por encima de la puerta y su punta se inclina hacia el suelo. De encontrarse alguna impureza en la casa se anexa el voladizo con el techo de la casa y es impurificado. Y de encontrarse alguna impureza en él, se considera como si su punta bajara hasta el suelo e impurifica el interior de la casa.  SU FAZ SE ENCUENTRA HACIA ARRIBA. Su punto se inclina hacia arriba y no se anexa con el techo de la casa, por lo tanto impurifica únicamente de sur su medida como mínimo un téfaj.  EL TEJADO. Se encuentra inclinada su punta de un lado y del otro y en el centro es liso.  O SEA, LA MEDIDA DE DOCE TEFAJIM. La medida de cada hilera de ladrillos es de cuatro Tefajim.  LA DIADEMA. Colocaban piedras arriba de la puerta en redondel en similitud a una corona como adorno.  Y LOS ADORNOS. Adornos de piedras que solían colocar arriba de las puertas y ventanas. Y en muchas ocasiones sobresalían de la pared en más de un téfaj. 2. EL VOLADIZO QUE SE ENCUENTRA ARRIBA DE LA ENTRADA. Se refiere aquí cuando se encuentra arriba de una entrada obstruida, por ende únicamente atrae la impureza de ser su medida un téfaj. DE LA MEDIDA DE DOS DEDOS. O sea, el voladizo que se encuentra sobre aquella ventana mencionada en el capitulo anterior en el cual atrae la impureza con ser su medida mínima dos dedos, como la ventana que fue parcialmente obstruida. O se encuentra sobre la ventana que atrae la impureza de ser su medida como la perforadora, impurifica aunque sea su medida lo más mínimo.  ÚNICAMENTE DE SER SU MEDIDA COMO LA DE LA VENTANA. De encontrarse el voladizo arriba de la ventana de dos dedos, se anexa a le ventana de ser su medida mínimamente dos dedos. Y de encontrarse arriba de una ventana como la medida de la perforadora, debe tener esta medida mínimamente. La Halajá no concuerda con Rabí Yosé.  3. AUNQUE ESTÉN A CIEN AMÁ DE ALTURA. A pesar que respecto al voladizo se dijo: “De encontrarse hasta la medida de tres hileras de ladrillos”, en este caso por ser que es transportable, los sabios fueron más rígidos al respecto. Y Rabí Yojanán opina: no se debe ser más meticuloso en este caso que respecto al voladizo, y de encontrarse arriba de los doce tefajim de altura atrae la impureza únicamente de ser su medida un téfaj. La Halajá concuerda con Rabí Yojanán Ben Nurí. 4. EL VOLADIZO QUE RODEA TODA LA CASA. El voladizo que resalta en la pared en la medida de un téfaj y rodea toda la casa, excepto en la entrada que solamente cubre en la medida de tres dedos de largo y no toda la puerta. Porque de ser que cubre toda la entrada, indudablemente atrae impureza al interior de la casa.  TODO LO QUE SE ENCUENTRA BAJO DE ÉL ES IMPURIFICADO. En este caso no discute Rabí Yehoshúa como discute en el siguiente caso, por ser que la naturaleza de la impureza es egresar y no ingresar.  LA CASA ES IMPURIFICADA, ESTAS SON LAS PALABRAS DE RABÍ ELIÉZER. Por ser que la ley respecto al voladizo es más rigurosa, que aun de resaltar de la pared en lo mínimo impurifica, alcanza con tener solamente la medida de tres dedos para anexarse al techo de la casa e impurificar. EN CASO DE UN PATIO QUE SE ENCUENTRA RODEADO POR UN ATRIO. Y el atrio ocupa en la entrada solamente en el largo de tres dedos. De encontrarse alguna impureza en la casa, todo lo que se encuentra en el atrio es impurificado. Y de encontrarse alguna impureza en el atrio, según Rabí Eliezér la casa es impurificada, y según Rabí Yehoshúa la casa conserva su pureza. La Halajá concuerda con Rabí Yehoshúa.  5. A pesar que no sobresale un téfaj, atrae la impureza y la traslada debajo del voladizo inferior. ES IMPURIFICADO LO QUE SE ENCUENTRA DEBAJO DE LA PARTE SOBRESALIENTE ETC. Lo que se encuentra entre los dos voladizos y en el suelo en dirección a la impureza, es impurificado.  LO QUE SE ENCUENTRA DEBAJO DEL VOLADIZO INFERIOR ETC. Por ser que el superior no sobresale del inferior en la medida de un téfaj, no atrae la impureza hacia abajo del inferior.  La Halajá concuerda con Rabí Yehoshúa.  6. PERO ENTRE ELLOS NO HAY LA MEDIDA DE UN TÉFAJ. El voladizo inferior no se encuentra elevado del suelo en la medida de un téfaj y entre el superior e inferior no hay la medida de un téfaj.  TODO LO QUE SE ENCUENTRA ABAJO Y ENTRE ELLOS ES IMPURIFICADO. Por ser que no hay un espacio de un téfaj en el inferior y entre ellos, la impureza se traslada a todos lados, excepto de arriba del superior, ya que este sí se encuentra arriba de un téfaj de altura.  TODO LO QUE SE ENCUENTRA ARRIBA DE ELLOS HASTA LA INMENSIDAD ES IMPURIFICADO. Pero lo que se encuentra debajo del inferior conserva su pureza, a pesar que no hay un espacio entre el superior e inferior, se refiere cuando el inferior se encuentra elevado del suelo en la medida de un téfaj y por ende contiene la impureza.  7. EN CASO QUE LOS VOLADIZOS NO TIENEN LA MEDIDA DE UN TÉFAJ DE ANCHO. Por lo tanto, ya sea se encuentra la impureza arriba o abajo, no contiene la impureza.  DE IGUAL MANERA EN EL CASO DE DOS CUEROS: Se aplica la ley de igual manera que respecto a los voladizos.

    Guemará

    Tratado Nidá 23b

    Los sabios dicen: El feto que no tiene fisonomía humana, no es considerado como tal. Dijo Rabí Yirmiyá Bar Abá en nombre de Rab: Según todas las opiniones, en caso de un feto que su cuerpo tomó la figura de una cabra y su cabeza de humano, es considerado un humano. En caso de ser su cuerpo como el humano y su cabeza como el de una cabra, no es considerado una persona. Y discutieron en caso que su cabeza es como la humana y uno de sus ojos como los del animal. En el cual Rabí Meir opina: es considerado como un humano por ser que parte de él es como tal. Y los sabios opinan: se requiere que sea todo su cuerpo como el humano para considerarse persona.  Le preguntaron a Rabí Yirmiyá Bar Abá: ¡Hemos estudiado todo lo contrario! Rabí Meir dice: requiere que sea todo su cuerpo como el humano para considerarse persona. Y los sabios dicen: es considerado como un humano por ser que parte de él es como tal. Le respondió: Cada cual como lo estudió.

    Zóhar

    Vayakhel 203b

    (Comentario sobre el Zóhar)

    Ven y observa: Hay días seculares, como hemos estudiado que representan a los demás pueblos. Y hay días de Shabat que representan al pueblo de Israel. En cuanto se eleva este punto, todo se esconde y es esto lo llamado “Shabat”. ¿Cuál es el motivo del Nombre Shabat? Si dijeras: porque es un día de descanso como versa lo escrito: “Porque en él descansó de toda la obra” (Génesis 2:3), es real esta explicación, pero si profundizamos en el tema podríamos aprender más. En cuanto sube la luz e ilumina, se corona con los Abot, y es entonces que se une para ser toda una y es llamada Shabat. Todo el reinado junto a los Abot son llamados Shabat. Esta palabra está compuesta por “Shin” y “Bat”. La letra “Shin”, representa a los tres Abot cuando se unen.  Y ella se embellece con ellos (Pues las tres ו”Vav” que se encuentran en la letra “Shin”, representan a los tres Abot). Ella se embellece con ellos y ellos con ella, y esto es Shabat, cuando todos se encuentran unidos. Y si dijeras: el gran Shabat que es superior, ¿porque fue llamado Shabat? Por ser que en el punto donde es el centro del ojo es llamado “Bat”, como versa lo escrito: “Guárdame como a la niña del ojo” (Salmos 17:8). Y por ser lo principal del ojo fue llamado niña. 

    Halajá

    Rambam, Leyes sobre Shabat, capítulo 9

    (1) El acto del cardado en la lana, lino, seda o cosas parecidas, se prohíbe realizar en Shabat, y quien lo hiciera es condenado. Se condena a la persona que lo hiciera en la medida como para hacer de ello un hilo de cuatro tefajim de largo. Asimismo, se prohíbe cardar venas de manera que puedan ser utilizadas como un hilo de lana, y quien lo hiciera es condenado.  (2) El acto de teñir en Shabat se prohíbe, y quien lo hiciera en un hilo de la medida de cuatro tefajim de largo o en la lana en la cantidad suficiente como para hilar un hilo de esta medida, es condenado. Solamente es condenado quien lo hiciera con una tintura permanente, pero de ser una tintura lavable, como ser que le pasó una tintura falsa o limó el metal o cobre y tiño con ello, no es condenado. Por ser que es lavable, y toda cosa que no es permanente respecto a Shabat no es condenado.

    Moral

    Séfer Hajasidim, capítulo 124-125

    Cuídese la persona de presentarse delante de aquellos que enseñan Torá a los pequeños, no vaya a ser que los impida de trabajar. Ven y observa la conducta de Abá Jilkiyá el nieto de Joní Hamaaguel que mientras se encontraba trabajando no contestaba el saludo de las personas, como hemos encontrado en el tratado de Taanit (23b). ¡Cuánto más que la persona debe cuidarse de tratarse de una labor relacionada con la Torá!, y maldito será quien realiza trabajos sagrados con engaño. Quien no requiere de la ayuda de otros e igualmente les solicita, no se irá de este mundo hasta que necesite de las personas. Asimismo, quien finge una molestia en las piernas o pérdida de visión para pedir dinero, no se irá de este mundo hasta que le ocurra.
