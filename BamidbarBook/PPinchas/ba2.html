    א(ד) כִּי לֹא בְחַרְבָּם יָרְשׁוּ אָרֶץ וּזְרוֹעָם לֹא הוֹשִׁיעָה לָּמוֹ כִּי יְמִינְךָ וּזְרוֹעֲךָ וְאוֹר פָּנֶיךָ כִּי רְצִיתָם: אֲרוּם לָא בִּגְבוּרַת סַיְפֵהוֹן יְרִיתוּ אַרְעָא וּתְקוֹף אֶדְרָעֵיהוֹן לָא פַרְקַת לְהוֹן אֲרוּם יְמִינָךְ וְאֶדְרַע גְבוּרְתָּךְ וּנְהוֹר זִיו יְקָרָךְ אֲרוּם כָּל זְמַן דַעֲסִיקוּ בְאוֹרַיְתָא אִתְרְעֵיתָא בְהוֹן: (ה) אַתָּה הוּא מַלְכִּי אֱלֹהִים צַוֵּה יְשׁוּעוֹת יַעֲקֹב: אַנְתְּ הוּא מַלְכִי אֱלָהָא בְּעִדָנָא הָדָא פַּקֵד פּוּרְקָנוּת דְבֵית יַעֲקֹב: (ו) בְּךָ צָרֵינוּ נְנַגֵּחַ בְּשִׁמְךָ נָבוּס קָמֵינוּ: בְּמֵימְרָךְ מְעִיקָנָא נְנַגֵחַ בִּשְׁמָךְ נִכְבּוֹשׁ כָּל דְקָיְמִין עֲלָנָא: (ז) כִּי לֹא בְקַשְׁתִּי אֶבְטָח וְחַרְבִּי לֹא תוֹשִׁיעֵנִי: אֲרוּם לָא בְּקַשְׁתִּי אֲנָא רָחִיץ וְסַיְפִי לָא תִפְרוֹק יָתִי:                                                                                                                                                                                                                                                                                                          ן

    פסחים פרק ז

    אא. כֵּיצַד צוֹלִין אֶת הַפֶּסַח, מְבִיאִין שַׁפּוּד שֶׁל רִמּוֹן, תּוֹחֲבוֹ מִתּוֹךְ פִּיו עַד בֵּית נְקוּבָתוֹ, וְנוֹתֵן אֶת כְּרָעָיו וְאֶת בְּנֵי מֵעָיו לְתוֹכוֹ, דִּבְרֵי רַבִּי יוֹסֵי הַגְּלִילִי. רַבִּי עֲקִיבָא אוֹמֵר, כְּמִין בִּשּׁוּל הוּא זֶה, אֶלָּא תוֹלִין חוּצָה לוֹ: ב. אֵין צוֹלִין אֶת הַפֶּסַח לֹא עַל הַשַּׁפּוּד וְלֹא עַל הָאַסְכְּלָא. אָמַר רַבִּי צָדוֹק, מַעֲשֶׂה בְּרַבָּן גַּמְלִיאֵל שֶׁאָמַר לְטָבִי עַבְדּוֹ, צֵא וּצְלֵה לָנוּ אֶת הַפֶּסַח עַל הָאַסְכְּלָא. נָגַע בְּחַרְסוֹ שֶׁל תַּנּוּר, יִקְלוֹף אֶת מְקוֹמוֹ. נָטַף מֵרָטְבּוֹ עַל הַחֶרֶס וְחָזַר עָלָיו, יִטּוֹל אֶת מְקוֹמוֹ. נָטַף מֵרָטְבּוֹ עַל הַסֹלֶת, יִקְמוֹץ אֶת מְקוֹמוֹ: ג. סָכוֹ בְשֶׁמֶן תְּרוּמָה, אִם חֲבוּרַת כֹּהֲנִים, יֹאכֵלוּ. אִם יִשְׂרָאֵל, אִם חַי הוּא, יְדִיחֶנּוּ. וְאִם צָלִי הוּא, יִקְלוֹף אֶת הַחִיצוֹן. סָכוֹ בְשֶׁמֶן שֶׁל מַעֲשֵׂר שֵׁנִי, לֹא יַעֲשֶׂנּוּ דָמִים עַל בְּנֵי חֲבוּרָה, שֶׁאֵין פּוֹדִין מַעֲשֵׂר שֵׁנִי בִירוּשָׁלַיִם: ד. חֲמִשָּׁה דְבָרִים בָּאִין בְּטֻמְאָה וְאֵינָן נֶאֱכָלִין בְּטֻמְאָה, הָעֹמֶר, וּשְׁתֵּי הַלֶּחֶם, וְלֶחֶם הַפָּנִים, וְזִבְחֵי שַׁלְמֵי צִבּוּר, וּשְׁעִירֵי רָאשֵׁי חֳדָשִׁים. הַפֶּסַח שֶׁבָּא בְטֻמְאָה, נֶאֱכָל בְּטֻמְאָה, שֶׁלֹּא בָא מִתְּחִלָּתוֹ אֶלָּא לַאֲכִילָה: ה. נִטְמָא הַבָּשָׂר וְהַחֵלֶב קַיָם, אֵינוֹ זוֹרֵק אֶת הַדָּם. נִטְמָא הַחֵלֶב וְהַבָּשָׂר קַיָּם, זוֹרֵק אֶת הַדָּם. וּבַמֻּקְדָּשִׁין אֵינוֹ כֵן, אֶלָּא אַף עַל פִּי שֶׁנִּטְמָא הַבָּשָר וְהַחֵלֶב קַיָּם, זוֹרֵק אֶת הַדָּם: ו. נִטְמָא הַקָּהָל אוֹ רֻבּוֹ, אוֹ שֶׁהָיוּ הַכֹּהֲנִים טְמֵאִים וְהַקָּהָל טְהוֹרִים, יֵעָשֶׂה בְטֻמְאָה. נִטְמָא מִעוּט הַקָּהָל, הַטְּהוֹרִין עוֹשִׂין אֶת הָרִאשׁוֹן, וְהַטְּמֵאִין עוֹשִׂין אֶת הַשֵּׁנִי: ז. הַפֶסַח שֶׁנִּזְרַק דָּמוֹ, וְאַחַר כָּךְ נוֹדַע שֶׁהוּא טָמֵא, הַצִיץ מְרַצֶה. נִטְמָא הַגּוּף, אֵין הַצִיץ מְרַצֶה, מִפְּנֵי שֶׁאָמְרוּ, הַנָּזִיר וְעוֹשֶׂה פֶסַח, הַצִיץ מְרַצֶה עַל טֻּמְאַת הַדָּם, וְאֵין הַצִיץ מְרַצֶה עַל טֻּמְאַת הַגּוּף. נִטְמָא טֻמְאַת הַתְּהוֹם, הַצִיץ מְרַצֶה: ח. נִטְמָא שָׁלֵם אוֹ רֻבּוֹ, דִכְתִּיב עֵדּוּת בִיהוֹסֵף שָׂמוֹ אוּף הָכָא י' אִתּוֹסֵף בְּפִנְחָס עַל דְּקַנֵּי בְהָאי:     תרגום הזוהר

    ארַבִּי יְהוּדָה פָּתַח וְאָמַר, שָׁמְרָה נַפְשִׁי כִּי חָסִיד אָנִי הוֹשַׁע עַבְדְּךְ וְגוֹ'. יֵשׁ לְהִסְתַּכֵּל בְּסוֹף הַפָּסוּק, וְאַחַר כָּךְ בְּכָל הַפָּסוּק. בְּסוֹפוֹ שֶׁל הַפָּסוּק כָּתוּב, הַבּוֹטֵחַ אֵלֶיךְ. שׁוֹאֵל, הַבּוֹטֵחַ בְּךְ הָיָה צָרִיךְ לוֹמָר, מַהוּ הַבּוֹטֵחַ אֵלֶיךְ. וּמֵשִׁיב, אֶלָּא כִּבְיָכוֹל דָּוִד הָיָה מַבְטִיחַ שֶׁלֹּא יַעֲבוֹר עָלָיו חֲצוֹת לַיְלָה בְּשֵׁינָה. כְּמ''שׁ חֲצוֹת לַיְלָה אָקוּם לְהוֹדוֹת לְךְ. הָיָה צָרִיךְ לוֹמָר קַמְתִּי. אֶלָּא הַפֵּרוּשׁ הוּא, אָקוּם וְאֶתְקַשֵּׁר בְּךְ לְעוֹלָם. שָׁמְרָה נַפְשִׁי וְכוּ': שׁוֹאֵל, שְׁמוֹר הָיָה צָרִיךְ לוֹמָר. וַהֲרֵי לָמַדְנוּ, שֶׁאֵין אוֹת בַּתּוֹרָה שֶׁלֹּא יִהְיוּ בָּהּ סוֹדוֹת עֶלְיוֹנִים וִיקָרִים. וּמֵשִׁיב, שָׁמְרָה, הַיְנוּ שֶׁאוֹמֵר אֶל הַקָּדוֹשׁ בָּרוּךְ הוּא, שָׁמְרָה, הַיְנוּ, שְׁמוֹר ה', שֶׁהִיא אוֹתוֹ הַחֵלֶק שֶׁהַנֶּפֶשׁ נֶאֱחֶזֶת בּוֹ. כִּי כְּשֶׁהַנֶּפֶשׁ יוֹצֵאת מֵהָעוֹלָם הַזֶּה, הִיא בָּאָה לָרֶשֶׁת הָעוֹלָם הַבָּא. אִם זָכְתָה, כַּמָּה צְבָאוֹת עֶלְיוֹנִים יוֹצְאִים לְקַבְּלָהּ וּלְשָׁמְרָהּ וּלְהַכְנִיסָהּ אֶל הַמָּדוֹר שֶׁלָּהּ בִּמְקוֹמָהּ. וְאוֹתָהּ ה', שֶׁהִיא הַמַּלְכוּת, שׁוֹמֶרֶת אוֹתָהּ, לְהִתְאָחֵד עִמָּהּ בְּרָאשֵׁי חָדֳשִׁים וְשַׁבָּתוֹת. וְאִם לֹא זָכְתָה, כַּמָּה מַלְאֲכֵי חַבָּלָה מוּכָנִים כְּנֶגְדָהּ, שֶׁדּוֹחִים אוֹתָהּ לַחוּץ. אוֹי לְאוֹתָהּ הַנֶּפֶשׁ הַמִּתְגַּלְגֶּלֶת לָרִיק כָּאֶבֶן בְּכַף הַקֶּלַע. זֶ''שׁ וְאֶת נֶפֶשׁ אוֹיִבְךְ יִקְלָעֶנָה בְּתוֹךְ כַּף הַקָּלַע. וְדָוִד בִּקֵּשׁ בַּקָּשָׁתוֹ לִפְנֵי הַקָּדוֹשׁ בָּרוּךְ הוּא. וְאָמַר, שָׁמְרָה נַפְשִׁי, שֶׁלֹּא יִדְחוּ אוֹתָהּ לַחוּץ. וּכְשֶׁיַּגִּיעוּ כְּנֶגְדָהּ, יִפְתְּחוּ לָהּ הַפְּתָחִים וּתְקַבְּלָהּ לְפָנֶיךְ. כִּי חָסִיד אָנִי, שׁוֹאֵל, וְכִי חָסִיד נִקְרָא דָּוִד, הֲלֹא הוּא מֶלֶךְ וּמִדָּתוֹ הִיא מַלְכוּת. אָמַר רַבִּי יְהוּדָה, כֵּן, שֶׁנִּקְרָא חָסִיד, שֶׁכָּתוּב, חַסְדֵּי דָּוִד הַנֶּאֱמָנִים. דְּהַיְנוּ, שֶׁמִּשּׁוּם שֶׁמְּקַבֵּל חֲסָדִים הַנֶּאֱמָנִים, נִקְרָא חָסִיד. וּמִשּׁוּם זֶה שָׁמְרָה נַפְשִׁי שֶׁלֹּא תַּעֲזוֹב אוֹתָהּ לָלֶכֶת לַחוּץ. רַבִּי יִצְחָק אָמַר, כָּל אָדָם שֶׁיֵּשׁ לוֹ חֵלֶק בַּצַדִּיק, דְּהַיְנוּ שֶׁשּׁוֹמֵר בְּרִיתוֹ, יוֹרֵשׁ אֶרֶץ זוֹ, שֶׁהִיא הַמַּלְכוּת, כְּמ''שׁ, וְעַמֵּךְ כֻּלָּם צַדִּיקִים לְעוֹלָם יִירְשׁוּ אָרֶץ. וְהַצַדִּיק הַזֶּה, שֶׁהוּא יְסוֹד, נִקְרָא חָסִיד, מִשּׁוּם שֶׁמַּשְׁפִּיעַ חֲסָדִים. וְעַל כֵּן אָמַר דָּוִד, מֵאַחַר שֶׁאֲנִי אָחוּז בְּאוֹתוֹ הַמָּקוֹם, בַּצַדִּיק, עַל כֵּן חָסִיד אָנִי, וּבִשְׁבִיל זֶה שָׁמְרָה נַפְשִׁי לְהִתְקַשֵּׁר בְּךְ. רַבִּי חִיָּא פָּתַח, עֵדוּת בִּיהוֹסֵף שָׂמוֹ בְּצֵאתוֹ עַל אֶרֶץ מִצְרָיִם שְׂפַת לֹא יָדַעְתִּי אֶשְׁמָע וְגוֹ'. הֲרֵי לָמַדְנוּ, שֶׁהַמַּלְאָךְ לִמְּדוֹ שִׁבְעִים לָשׁוֹן, כְּמוֹ שֶׁיָּדַע פַּרְעֹה, וּבְלָשׁוֹן הַקֹּדֶשׁ הָיָה גָּדוֹל יוֹתֵר מִפַּרְעֹה, שֶׁפַּרְעֹה לֹא הָיָה יוֹדֵעַ לָשׁוֹן הַקֹּדֶשׁ (סוטה לו:). זֶ''שׁ, שְׂפַת לֹא יָדַעְתִּי אֶשְׁמָע, שֶׁלִּמְּדָהוּ שָׂפוֹת, שֶׁלֹּא שָׁמַע אוֹתָן מִקּוֹדֶם. אֲבָל לְפִי זֶה, מַהוּ עֵדוּת. וּמֵשִׁיב, בֹּא וּרְאֵה, בְּשָׁעָה שֶׁאֵשֶׁת פּוֹטִיפַר אֲחָזָתוֹ לְאוֹתוֹ הַדָּבָר עָשָׂה עַצְמוֹ יוֹסֵף כְּמִי שֶׁאֵינוֹ יוֹדֵעַ הַשָּׂפָה שֶׁלָּהּ, וְכֵן בְּכָל יוֹם, עַד הַשָּׁעָה הַהִיא הָאַחֲרוֹנָה שֶׁכָּתוּב, וַתִּתְפְּשֵׂהוּ בְּבִגְדוֹ. מַהוּ וַתִּתְפְּשֵׂהוּ. אֶלָּא, מִשּׁוּם שֶׁעַד אָז, עָשָׂה עַצְמוֹ כְּמִי שֶׁאֵינוֹ יוֹדֵעַ הַלָּשׁוֹן, וְאָז הִכִּירָה בּוֹ, שֶׁהוּא יוֹדֵעַ הַלָּשׁוֹן, כְּלוֹמָר שֶׁמֵּבִין כַּוָּנָתָהּ. וְזֶהוּ וַתִּתְפְּשֵׂהוּ, שֶׁתָּפְשָׂה הָרַמָּאוּת שֶׁלּוֹ. בְּבִגְדוֹ, הוּא לָשׁוֹן בְּגִידָה וְרַמָּאוּת. וְרוּחַ הַקֹּדֶשׁ, שֶׁהוּא הַמַּלְכוּת, צוֹוַחַת כְּנֶגְדוֹ, לְשָׁמְרְךְ מֵאִשָׁה זָרָה מִנָּכְרִיָּה אֲמָרֶיהָ הֶחֱלִיקָה. מַה מַּשְׁמִיעֵנוּ. אֶלָּא כָּל מִי שֶׁשּׁוֹמֵר עַצְמוֹ מִזֶּה, כְּמוֹ יוֹסֵף, הוּא מִתְקַשֵּׁר בַּשְּׁכִינָה, וְאוֹחֵז בָּעֵדוּת הַזּוֹ, וּמָה הִיא, הִיא הַה' שֶׁנִּתּוֹסְפָה בּוֹ, שֶׁכָּתוּב עֵדוּת בִּיהוֹסֵף שָׂמוֹ וְגוֹ'. אַף כָּאן, י' נִתוֹסְפָה בְּפִינְחָס עַל שֶׁקִּנֵּא בָּזֶה.     Tratado Pesajim, capítulo 6

    (1) ¿Cómo se procede a asar el Sacrificio de Pésaj? Se traía una rama de granado y la insertaba desde la boca del animal hasta su orificio anal, y en el interior del estomago se colocaban sus piernas e intestinos, estas son las palabras de Rabí Yosé. Rabí Akibá dice: De proceder de esta manera estaría cocinando sus miembros, por tanto, debe colocarlos fuera del animal.  (2) Se prohíbe asar la ofrenda sobre brochetas o sobre una parrilla. Dijo Rabí Tzadok: Una vez le dijo Rabán Gamliel a su siervo Tabí: Ve y asa nuestro Sacrificio de Pésaj en la parrilla. Si la ofrenda tiene contacto con las paredes del horno, debe seccionar aquella parte. En caso que salpique grasa al horno y retorne nuevamente al animal, debe quitar aquella parte. Si cayó grasa de la ofrenda sobre flor de harina, debe quitar un puñado de ella en el lugar donde cayó.  (3) El Sacrificio de Pésaj que fue ungido con aceite de Terumá, de ser sus dueños Cohanim, se les permite consumir de ella. De ser israelitas, si aún se encuentra el animal crudo, deben lavarlo. En caso que ya fue asado, deben cortar la parte externa del animal. El Sacrificio de Pésaj que fue ungido con aceite del Maaser Shení, no puede luego solicitar su reintegro a los demás socios de la ofrenda, ya que se prohíbe redimir los frutos del Maaser Shení en Jerusalem.  (4) Cinco ofrendas se sacrificaban en estado de impureza, pero no eran consumidas en tal caso. La ofrenda del Omer, los dos Panes de Shabuot, los Panes de los Semblantes, las ofrendas de Paz del público, y los chivos sacrificados en Rosh Jódesh. El Sacrificio de Pésaj era ofrendado y consumido en estado de impureza, porque su finalidad es ser consumido.  (5) El Sacrificio de Pésaj cuya carne se impurificó, aunque no su sebo, su sangre no se rocía en el Altar. En caso que haya sido impurificado su sebo pero no su carne, se rocía su sangre en el Altar. En los demás sacrificios no se actúa así sino que, aunque se impurifique su carne y el sebo permanezca puro, su sangre debe ser rociada.  (6) Si todo el público o su mayor parte se impurifica, o si los Cohanim se impurifican pero el público permanece puro, se llevará a cabo con impureza. Si sólo se impurificó una minoría del público, los puros lo ofrecerán en el primer Pésaj y los impuros en el segundo.  (7) El Sacrificio de Pésaj cuya sangre fue rociada, y después se descubre que era impuro, la Felpa lo hace válido. Si el cuerpo se impurifica, la Felpa no lo convierte en puro. Sobre esto se dijo: Al Nazareno y al que ofrenda el Sacrificio de Pésaj, la Felpa lo libera de la impureza de la sangre, pero no lo libera de su impureza física. Si su impureza es del abismo, la Felpa también lo libera de ella.  (8) El Sacrificio de Pésaj que se impurificó total o parcialmente, debe ser incinerado en el recinto anterior al Santuario con leños del Altar. Si se impurificó sólo una pequeña porción o se impurificaron las sobras, se queman en sus patios particulares o en sus tejados, con los leños del Altar. Los tacaños lo quemaban en el recinto anterior al Santuario, para poder usar los leños del Altar.  (9) El Sacrificio de Pésaj que fue sacado o que se impurificó será incinerado enseguida. Si los propietarios se impurificaron o murieron, esperará hasta que se invalide y lo incinerará el dieciséis. Rabí Yojanán Ben Beroká dice: También este debe ser incinerado enseguida, ya que no hay quien lo pueda consumir.  (10) Los huesos, los tendones y las sobras, serán quemados el dieciséis. Si el dieciséis coincidiera con Shabat, lo quemarán el diecisiete, ya que no se anteponen ni al Shabat ni a Yom Tob.  (11) Todo lo que se come de un toro adulto, se ingiere en un cabrito joven, así como los omoplatos y los cartílagos.  Quien quiebra un hueso del Sacrificio puro de Pésaj, es condenado a cuarenta azotes. Quien deja sobrantes de la ofrenda pura o quiebra huesos de la ofrenda impura, no ese condena a los cuarenta.   (12) El miembro que sobresale de su lugar, debe cortarlo hasta el hueso, quitará la carne que haya hasta la articulación y allí cortará.  En el resto de los sacrificios, cortará con la puntilla, ya que no existe la prohibición de cortar hueso. Desde el marco hacia adentro, es considerado como que se encuentra  en el interior, desde el marco hacia fuera, como si estuviera afuera.  Las ventanas y el grosor de la muralla, como interior.  (13) Dos grupos que comían en una misma casa, unos volverán sus caras hacia un lado y comerán, y los otros volverán sus caras hacia el otro lado y comerán, y el calentador estará en el centro.  Cuando el sirviente va a servir, mantiene su boca cerrada, después volverá su cara hacia su grupo y comerá.  La novia volverá su cara hacia otro lado y comerá.

    Rabí Ovadia de Bartenura

    1. UNA RAMA DE GRANADO. Pero no de otros árboles, ya que es el único árbol que provee madera que no expele savia cuando se quema, pues de hacerlo estaría cocinando la ofrenda en lugar de asarla. De igual manera, se prohíbe utilizar un palo de hierro, porque cuando se calienta una parte, se calienta enteramente, cuando la Torá exigió que sea “Asado por el fuego”, y no por otras cosas.  ESTARÍA COCINANDO. Es como cocinar los miembros dentro de una olla. FUERA DEL ANIMAL. Engancha los miembros en el palo, por el lado de la boca del animal. La Halajá concuerda con Rabí Akibá.  2. DIJO RABÍ TZADOK. La Mishná está incompleta, así se debe estudiar: En caso de ser una parrilla con orificios, se permite asar en ella, dijo Rabí Tzadok… La Halajá concuerda con Rabí Tzadok.  TIENE CONTACTO. El Sacrificio de Pésaj.  DEBE SECCIONAR AQUELLA PARTE. Debe quitar la parte que tuvo contacto con las paredes del horno, pues se cocinó con el calor del horno y no con el fuego.  EN CASO QUE SALPIQUE LA GRASA AL HORNO. Se cocinó la grasa con el calor del horno y luego rebotó en la ofrenda y la absorbió.  DEBE QUITAR AQUELLA PARTE. En este caso no basta con quitar la parte externa del animal, sino que debe sacar un pedazo del grosor de un dedo.  FLOR DE HARINA. Como cuando la sémola está caliente y al caer en ella la grasa se cocina con su calor, por ende debe quitar un puñado de sémola en el lugar donde cayó la grasa y lo quemará.  3. UNGIDO CON ACEITE DE TERUMÁ. Se permite ungir el Sacrificio de Pésaj con jugo de frutas.  DEBEN LAVARLO. Puesto que no absorbe nada.  DEBEN CORTAR LA PARTE EXTERNA DEL ANIMAL. Porque absorbió el aceite de la Terumá.  YA QUE SE PROHÍBE REDIMIR. Se prohíbe venderlos.  LOS FRUTOS DEL MAASER SHENÍ EN JERUSALEM. Aunque los coma con pureza, como versa: “Los darás por dinero que atarás en tu mano e irás con él al lugar que escoja el Eterno tu Di-s” (Deuteronomio 14:25).  4. CINCO OFRENDAS SE SACRIFICABAN. Solamente estas ofrendas públicas se consumían, pues los sacrificios públicos apartan la impureza, tal como está escrito con relación al Sacrificio de Pésaj: “En su tiempo”. Se entiende, de ser solamente algunos de los integrantes del pueblo de Israel que se encuentra impuro deben postergar su ofrenda hasta la segunda pascua, pero de ser todo el público impuro, lo realizan en la primera pascua. Y fue escrito al respecto: “En su tiempo señalado”. Y está escrito referente a las ofrendas publicas: “Todo eso haréis al Eterno en los tiempos señalados” (Números 29:39). De igual manera que lo escrito referente al Sacrificio de Pésaj “En su tiempo señalado”, se refiere a que es sacrificado en estado de impureza, también lo escrito en las ofrendas públicas: “En los tiempos señalados”, nos señala que son sacrificados también con impureza. De todas maneras, respecto a las ofrendas públicas solamente se permitió realizarlas, pero no consumirlas con impureza. El motivo por el cual la Mishná especificó cuál de las ofrendas, pero no generalizó la categoría de las ofrendas públicas, por ser que se excluye de ellas las ofrendas de Jaguigá traídas en las festividades, que son similares a las públicas por ser que vienen las personas en grupos, pero no son traídas con impureza ni en día de Shabat.  LAS OFRENDAS DE PAZ DEL PÚBLICO. Los corderos traídos en la festividad de Atzeret, pues es la única ofrenda de Paz traída por el público.  Y LOS CHIVOS SACRIFICADOS EN ROSH JÓDESH. En caso de los chivos sacrificados en las festividades no requiere enseñarnos la Mishná y se aprende de las ofrendas de Paz traídas por el público. De igual manera que las ofrendas de Paz del público son un sacrificio de sangre y se las sacrifica con impureza, también los chivos sacrificados en las festividades son una ofrenda de sangre y se la sacrifica con impureza. Pero en caso de la ofrenda de Rosh Jódesh, requirió enseñarnos la Mishná, por ser que no fue escrito al respecto “En su tiempo señalado”, y de esta frase se aprende que se sacrifican con impureza.  SU FINALIDAD ES SER CONSUMIDO. La orden de realizar el Sacrificio de Pésaj fue dada con la finalidad de consumirlo, como versa lo escrito: “Calculando la res según lo que cada cual pueda comer” (Éxodo 12:4). Por lo tanto, cuando se permitió hacerlo con impureza, se permitió para comerlo.  5. EL SACRIFICIO DE PÉSAJ CUYA CARNE SE IMPURIFICÓ. Por ser que la ofrenda es traída para utilizar su carne.  NO SE ACTÚA ASÍ. Y mientras que el sebo no se encuentra impuro se debe salpicar su sangre.  6. SI TODO EL PÚBLICO. Impurificados a causa de algún cadáver. Pues es la única impureza que posterga el Sacrificio de Pésaj.  CON IMPUREZA. También en caso que solamente los Cohanim se encuentren impurificados, se permite a los israelitas que fueron impurificados ofrecer el Sacrificio de Pésaj, y siendo que todo el pueblo lo trae con impureza, no se hace diferencia entre ellos.  7. SE DESCUBRE QUE ERA IMPURO. La ofrenda o la sangre.  LA FELPA LO HACE VÁLIDO. Y no se le requiere a su dueño que ofrende otro sacrificio en la segunda pascua.  SI EL CUERPO SE IMPURIFICA. Sus dueños estaban impuros.  LA FELPA NO LO PURIFICA. Y debe ofrecer un sacrifico en la segunda pascua, ya que en el momento de la realización de la ofrenda, la persona se encontraba imposibilitada de hacerla.  AL NAZARENO. Como versa: “Y si accidentalmente algún hombre muere cerca de él…” Pero los días transcurridos antes del hecho que lo impurificó se invalidarán porque los días de su cuenta cayó.  LA FELPA LO LIBERA. Solamente cuando el animal se encuentra impuro. Y se permite luego al nazareno rasurar su cabello, beber vino e impurificarse por los difuntos.  PERO NO DE SU IMPUREZA FÍSICA. Si el nazareno se encuentra impuro en el momento de ofrecer su sacrificio, no es válido y debe volver a contar sus días.  IMPUREZA DEL ABISMO. Toda impureza que la persona no sabe que se encontraba impuro es denominada “Del abismo”.  LA FELPA LO LIBERA. Esta Halajá es un legado de Moshé, el motivo es porque su dueño nunca supo de su impureza, por lo tanto en caso que se enteró luego, no se requiere ofrecer otro sacrificio en la segunda pascua. De igual manera respecto a un nazareno que no sabía que se encontraba impuro, no se requiere volver a contar sus días. Esta Halajá se dijo solamente respecto a la impureza de los cadáveres, pero no las que provienen de otro origen.  8. DEBE SER INCINERADO EN EL RECINTO ANTERIOR. Para hacer pública su negligencia al no cuidar su pureza, y todos observarán y aprenderán. CON LEÑOS DEL ALTAR. Para no avergonzar a quienes no tienen dinero para comprar leños.  UNA PEQUEÑA PORCIÓN. Al igual que la parte del animal puro que quedó sin consumir.  Y LOS TACAÑOS. Quemaban con leños del santuario la parte del animal que fue impurificada y lo que quedó de la ofrenda sin consumir.  9. EL SACRIFICIO DE PÉSAJ QUE FUE SACADO FUERA DE JERUSALEM. Fuera de la muralla de Jerusalem.  SERÁ INCINERADO ENSEGUIDA. En el día catorce, y no se requiere esperar hasta el día quince para que sea considerado una sobra del sacrificio y quemarlo. De todas maneras, no debe quemar en un día festivo.  SE IMPURIFICARON O MURIERON. En este caso el sacrificio es descalificado por un motivo externo al propio cuerpo del animal. ESPERARÁ HASTA QUE SE INVALIDE. Debe esperar que sea considerado sobrante del sacrificio y luego quemarlo. DEBE SER INCINERADO ENSEGUIDA: Rabí Yojanán discute con los sabios en caso que los dueños fueron impurificados o murieron antes de ser salpicada la sangre del animal y siendo que no es apta para consumo de otras personas es considerado como si el propio animal es descalificado. La Halajá no concuerda con Rabí Yojanán.  10. LOS HUESOS. Contienen grasa en su interior y se prohíbe quebrarlos para comer de ella, por lo tanto requieren ser quemados. TENDONES. Como los del muslo que por ley de la Torá se permite su grasa pero los sabios prohibieron. O el tendón externo que se encuentra cercano a la carne, y por ley de la Torá se permite su consumo. Deben ser quemados, porque es considerado un sobrante de la ofrendas.  Y LAS SOBRAS. La carne de la ofrenda que no llegaron a consumirla sus dueños. SERÁN QUEMADOS EL DIECISÉIS. O sea, en Jol Hamoed. A pesar que son inválidas en el primer día de la festividad, se prohíbe quemarlos en la festividad, porque la orden de quemar el sobrante es un precepto activo, y la orden de no quemar en la festividad es un precepto pasivo, y un precepto activo no desplaza una orden que incluye un precepto activo y otro pasivo.  11. TORO ADULTO. Lo trae como ejemplo porque el toro adulto endurece su carne en todo lo posible.  SE INGIERE EN UN CABRITO JOVEN. Lo que no es comestible en el toro adulto, no es válida aquella parte para asociarse en El Sacrificio de Pésaj cuando se ofrece una cabra.  LOS OMOPLATOS Y LOS CARTÍLAGOS. A esto se refiere: Aquello que se dijo “Todos los miembros que son comestibles en el toro adulto, etc.”, se refiere a los omoplatos del animal y a sus cartílagos.  La carne llamada “Trons” que se encuentra en la punta del hombro, y los demás cartílagos, como la oreja, el del pecho y el de las costillas. Y siendo que en el toro adulto aquellas partes son comestibles luego de hervirlas, en la cabra joven, pueden asociarse en ellas y comerlas asadas.  QUIEN DEJA SOBRANTES DE LA OFRENDA PURA. No es castigado, como versa: “Y no haréis sobrar nada de él hasta la mañana, pues si quedara algo, lo quemaréis en el fuego a la mañana” (Éxodo 12:10). La Torá ordenó el precepto activo de quemar lo sobrante para expiar por el precepto pasivo de haber dejado sobras, y por lo tanto no es castigado el trasgresor por ella. Otra explicación: es esto una prohibición pasiva que se trasgrede al no realizar un acto, y la ley indica: “En los preceptos pasivos la persona no es castigada con azotes”.  QUIEBRA HUESOS DE LA OFRENDA IMPURA. La orden escrita en la Torá: “Ni quebraréis hueso alguno” (Éxodo 12:46). Fue escrita respecto a la ofrenda pura, pero no respecto a la impura.  12. QUE SOBRESALE. Se debe cortar la parte que salió fuera de la muralla de Jerusalem, como versa: “No comeréis carne desgarrada en el campo” (Ibíd. 22:30). Aprendieron de aquí los sabios: La carne que sale al campo, o sea la ofrenda que es sacada fuera del lugar adecuado, como las ofrendas sumamente sagradas que fueron sacadas fuera del patio del Santuario o las ofrendas ligeramente sagradas que fueron retiradas de Jerusalem, son como carne desgarrada y se prohíbe su consumo.  DEBE CORTARLO. Corta la carne en rededor hasta llegar al hueso.  QUITARÁ LA CARNE. Debe consumir la parte de la carne que no salió fuera de la muralla y procede a quitarla del hueso hasta el punto de unión y arroja el hueso.  EN EL RESTO DE LOS SACRIFICIOS. En todas las ofrendas excepto la pascual.  CORTARÁ CON LA PUNTILLA. Corta el hueso en la parte donde salió.  Pero en el Sacrificio de Pésaj se prohíbe cortar el hueso.  DESDE EL MARCO HACIA DENTRO. La Mishná utiliza el término “Agaf” interpretado como el choque de las puertas.  Es decir, desde el lugar donde chocan las puertas del lado exterior hacia el interior es el lugar límite donde se permite llevar las ofrendas.  DESDE EL MARCO HACIA FUERA. O sea, desde el lugar donde chocan las puertas hacia afuera.  LAS VENTANAS: De la murallas. Y arriba de la muralla en el grosor, es considerado como el interior de la ciudad.  13. DOS GRUPOS QUE COMÍAN. Una misma ofrenda.  UNOS VOLVERÁN SUS CARAS… Puede cada grupo volver sus caras y comer en lados opuestos, y a pesar que es considerado como comer una misma ofrenda en dos casas, se permite. Se aprende de lo escrito: “De las casas donde lo comerán” (Éxodo 12:7). “De las casas”, se entiende que se permite comer en varias casas una misma ofrenda, como se explica a continuación: “Lo comerán”, escrito en singular. Pero una persona no puede comer una ofrenda en distintas casas.  EL CALENTADOR EN EL CENTRO. El calentador de donde se extraía el agua que se mezclaba con el vino. Y a pesar que separa los dos grupos a ser considerados como dos casas, se permite colocarlo entre ellos y puedan servirse del agua.  CUANDO EL SIRVIENTE VA A SERVIR. El sirviente sirve a los dos grupos, y cuando termina con un grupo sirve al otro.  SU BOCA CERRADA. Cierra su boca y mira en dirección a su grupo para que no piensen que está comiendo con el otro grupo cuando llega a ellos y se prohíbe comer en dos grupos.  LA NOVIA.     (Comentario sobre el Zóhar)

    Rabí Yehudá comenzó y dijo: Está escrito: “Guarda mi alma, porque soy devoto. Salva a Tu siervo”. (Salmos 86:2). Debemos prestar atención a las últimas palabras del versículo para entender el comienzo. Está escrito en el final del versículo: “Que confía hacia ti”. En lugar de esta expresión sería adecuado decir: “Que confía en ti”. Quiere decir: David prometió que no pasaría la mitad de la noche durmiendo. Como versa: “A media noche me levantaré para agradecerte” (Ibíd. 119:62). En lugar de esta expresión sería adecuado decir: “Me levanté para agradecerte”. Su intención fue decir: Me uniré a Ti para agradecerte. “Guarda mi alma”. En lugar de utilizar la expresión “Shomrá”. Debería haber dicho: “Shemor”. Hemos estudiado: No hay palabra en la Torá que no tenga algún sentido. Al decir “Shomrá” es como decir “Que Di-s me resguarde”. Cuando el alma se aparta de este mundo y se dirige a recibir su parte en el mundo venidero, si es meritoria, un ejército sale a recibirla e ingresarla a su lugar, y el Eterno la resguarda para unirse a ella en los días de Shabat y Yom Tob. En caso que el alma no meritó, se presentan a ella muchos ángeles dañinos que la empujan al exterior, y amargo será su destino porque sufrirá desplazamientos de un lugar a otro sin descanso. Esto es lo que está escrito: “Y las almas de tus enemigos, Él las arrojará como con una honda” (Samuel I 25:29). David solicitó al Eterno y dijo: Guarda mi alma, para que no se arrojada, y cuando llegue a Tus Manos, ábrele las puertas y sea recibida delante de Ti, pues yo soy devoto. ¿Acaso David fue denominado devoto? Dijo Rabí Yehudá: Así es, como versa: “Las fieles bondades de David (Isaías 54:3). Por merito de ello, guarda mi alma que no sea expulsada. Dijo Rabí Yitzjak: Todo aquel que tiene parte junto al justo, hereda esta tierra, el reinado, como versa: “Tu pueblo son todos justos, que heredarán la tierra por siempre” (Ibíd. 60:21). Este piadoso fue llamado devoto. Dijo el rey David: Siendo que me encuentro apegado a ese lugar, soy considerado un devoto, por lo tanto guarda mi alma que pueda conectarme a Ti. Rabí Jiyá comenzó y dijo: “Puso un testimonio en Yosef, cuando salió contra la tierra de Egipto. Oí una lengua desconocida” (Salmos 81:6). Hemos estudiado: “El ángel enseñó a Yosef setenta lenguas”, y sobre esto está escrito: “Oí una lengua desconocida”. Pero, ¿a qué testimonio se refiere? Ven y observa: En el momento que la mujer de Potifar sujetó a Yosef por sus partes, se hizo a sí mismo como si no conociera el idioma. Y así todos los días, hasta aquella última vez, como se dijo: ·Ella lo asió por su manto” (Génesis 39:12). Yosef se hacía como quien no conocía el idioma y la Providencia Divina decía por él: “Para cuidarte de una mujer extraña, de una extranjera que habla zalamerías” (Proverbios 7:5). Nos enseña: Todo aquel que se cuida de este pecado como Yosef, se apega a la Providencia Divina, y sostiene este testimonio. ¿A qué se refiere? A la letra “He” que se le agregó, como versa en hebreo: “Bihosef samó”. Aquí igual: A Pinejás se le agregó la letra “Yud”, por haber actuado con celo en este caso. 

    Halajá

    Rambam, Leyes sobre el Rezo, capítulo 11

    (1) Debe ser la persona sumamente cuidadosa en no comportarse con frivolidad den las sinagogas y casas de estudio, como por ejemplo: Jolgorio, burlas y charlas vanas. No comer, beber o realizar actos despreciables. No pasear, no ingresar en ellos en verano para resguardarse del sol, o en invierno, para resguardarse de la lluvia. No obstante, a los estudiosos de la Torá se les permite comer y beber en ellos en momentos de apremio.  (2) No se deben realizar cuentas en ellos, a menos que se requiera para cumplir algún precepto, como por ejemplo, la caja de la Tzedaká, para rescatar cautivos, u otras cosas similares. No se debe despedir allí a los difuntos, a menos que se trate de un gran erudito de aquella ciudad, por quienes toda la ciudad asiste.

    Séfer Hajasidim, capítulos 128-129

    Ocurrió con Rabenu Yaacob, hijo de Ribí Yacar que se arreglaba la barba delante del Arca Sagrada, y cuando el público iba a saludar al rey o al gobernador, él se descalzaba sus zapatos y decía: “Soy pobre”, ellos en cuanto a dinero, y yo en cuanto a rezos y súplicas. Ellos en lo suyo, y yo en lo mío. El público acostumbraba repetir tras él: “Honra al Eterno con lo que posees” (Proverbios 3:9). Uno compra un cofre para almacenar plata y oro, y otros adquieren un hermoso baúl para almacenar libros y oraciones. “Honra al Eterno con lo que posees”, es decir, con lo que te agració. Si tienes una linda voz, reza con ella al Eterno. “Este es mi Di-s, yo Lo embelleceré” (Éxodo 15:2). Se deben embellecer las Mitzvot: Un bonito Etrog, un lindo Lulab, una Sucá agradable, un bonito Talit… Así debes hacer con todos los objetos sagrados.
