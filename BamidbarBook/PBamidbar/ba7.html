    כ(נג) וְהַלְוִיִּם יַחֲנוּ סָבִיב לְמִשְׁכַּן הָעֵדֻת וְלֹא יִהְיֶה קֶצֶף עַל עֲדַת בְּנֵי יִשְׂרָאֵל וְשָׁמְרוּ הַלְוִיִּם אֶת מִשְׁמֶרֶת מִשְׁכַּן הָעֵדוּת: וְלֵוָאֵי יִשְׁרוּן סְחוֹר סְחוֹר לְמַשְׁכְּנָא דְסַהַדוּתָא וְלָא יְהֵי רוּגְזָא עַל כְּנִשְׁתָּא דִבְנֵי יִשְׂרָאֵל וְיִטְרוּן לֵוָאֵי יָת מַטְרַת מַשְׁכְּנָא דְסַהֲדוּתָא: (נד) וַיַּעֲשׂוּ בְּנֵי יִשְׂרָאֵל כְּכֹל אֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה כֵּן עָשׂוּ: וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִּי פַּקֵיד יְיָ יָת משֶׁה כֵּן עֲבָדוּ:  שלישי ב (א) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה וְאֶל אַהֲרֹן לֵאמֹר: וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר: (ב) אִישׁ עַל דִּגְלוֹ בְאֹתֹת לְבֵית אֲבֹתָם יַחֲנוּ בְּנֵי יִשְׂרָאֵל מִנֶּגֶד סָבִיב לְאֹהֶל מוֹעֵד יַחֲנוּ: גְבַר עַל טִּקְסֵהּ בְּאַתְוָן לְבֵית אֲבָהַתְהוֹן יִשְׁרוּן בְּנֵי יִשְׂרָאֵל מִלָקֳבֵל סְחוֹר סְחוֹר לְמַשְׁכַּן זִמְנָא יִשְׁרוּן: (ג) וְהַחֹנִים קֵדְמָה מִזְרָחָה דֶּגֶל מַחֲנֵה יְהוּדָה לְצִבְאֹתָם וְנָשִׂיא לִבְנֵי יְהוּדָה נַחְשׁוֹן בֶּן עַמִּינָדָב: וְדִי שְׁרַן קִדוּמָא מַדִינְחָא טֵקַס מַשְׁרִית יְהוּדָה לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי יְהוּדָה נַחְשׁוֹן בַּר עַמִינָדָב: (ד) וּצְבָאוֹ וּפְקֻדֵיהֶם אַרְבָּעָה וְשִׁבְעִים אֶלֶף וְשֵׁשׁ מֵאוֹת: וְחֵילֵיהּ וּמִנְיָנֵיהוֹן שַׁבְעִין וְאַרְבְּעָא אַלְפִין וְשִׁית מְאָה: (ה) וְהַחֹנִים עָלָיו מַטֵּה יִשָּׂשכָר וְנָשִׂיא לִבְנֵי יִשָּׂשכָר נְתַנְאֵל בֶּן צוּעָר: וְדִי שְׁרַן סְמִיכִין עֲלוֹהִי שִׁבְטָא דְיִשָׂשכָר וְרַבָּא לִבְנֵי יִשָׂשכָר נְתַנְאֵל בַּר צוּעָר: (ו) וּצְבָאוֹ וּפְקֻדָיו אַרְבָּעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת: וְחֵילֵהּ וּמִנְיָנוֹהִי חַמְשִׁין שִׁבְעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת: וְחֵילֵהּ וּמִנְיָנוֹהִי חַמְשִׁין וְשַׁבְעָא אַלְפִין וְאַרְבַּע מְאָה: (ט) כָּל הַפְּקֻדִים לְמַחֲנֵה יְהוּדָה מְאַת אֶלֶף וּשְׁמֹנִים אֶלֶף וְשֵׁשֶׁת אֲלָפִים וְאַרְבַּע מֵאוֹת לְצִבְאֹתָם רִאשֹׁנָה יִסָּעוּ: כָּל מִנְיָנַיָא לְמַשְׁרִית יְהוּדָה מְאָה וּתְמָנָן וְשִׁתָּא אַלְפִין וְאַרְבַּע מְאָה לְחֵילֵיהוֹן בְּקַדְמֵיתָא נַטְלִין: (י) דֶּגֶל מַחֲנֵה רְאוּבֵן תֵּימָנָה לְצִבְאֹתָם וְנָשִׂיא לִבְנֵי רְאוּבֵן אֱלִיצוּר בֶּן שְׁדֵיאוּר: טֵקַס מַשְׁרִית רְאוּבֵן דָרוֹמָא לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי רְאוּבֵן אֱלִיצוּר בַּר שְׁדֵיאוּר: (יא) וּצְבָאוֹ וּפְקֻדָיו שִׁשָּׁה וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת: וְחֵילֵהּ וּמִנְיָנוֹהִי אַרְבְּעִין וְשִׁתָּא אַלְפִין וַחֲמֵשׁ מְאָה: (יב) וְהַחוֹנִם עָלָיו מַטֵּה שִׁמְעוֹן וְנָשִׂיא לִבְנֵי שִׁמְעוֹן שְׁלֻמִיאֵל בֶּן צוּרִי שַׁדָּי: וְדִי שְׁרַן סְמִיכִין עֲלוֹהִי שִׁבְטָא דְשִׁמְעוֹן וְרַבָּא לִבְנֵי שִׁמְעוֹן שְׁלֻמִיאֵל בַּר צוּרִישַׁדָי: (יג) וּצְבָאוֹ וּפְקֻדֵיהֶם תִּשְׁעָה וַחֲמִשִּׁים אֶלֶף וּשְׁלֹשׁ מֵאוֹת: וְחֵילֵהּ וּמִנְיָנֵיהוֹן חַמְשִׁין וּתְשַׁע אַלְפִין וּתְלַת מְאָה: (יד) וְמַטֵּה גָּד וְנָשִׂיא לִבְנֵי גָד אֶלְיָסָף בֶּן רְעוּאֵל: וְשִׁבְטָא דְגָד וְרַבָּא לִבְנֵי גָד אֶלְיָסָף בַּר רְעוּאֵל: (טו) וּצְבָאוֹ וּפְקֻדֵיהֶם חֲמִשָּׁה וְאַרְבָּעִים אֶלֶף וְשֵׁשׁ מֵאוֹת וַחֲמִשִּׁים: וְחֵילֵהּ וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַמְשָׁא אַלְפִין וְשִׁית מְאָה וְחַמְשִׁין: (טז) כָּל הַפְּקֻדִים לְמַחֲנֵה רְאוּבֵן מְאַת אֶלֶף וְאֶחָד וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת וַחֲמִשִּׁים לְצִבְאֹתָם וּשְׁנִיִּם יִסָּעוּ: כָּל מִנְיָנַיָא לְמַשְׁרִית יִסָּעוּ אִישׁ עַל יָדוֹ לְדִגְלֵיהֶם: וְנַטֵל מַשְׁכַּן זִמְנָא מַשְׁרִית לֵוָאֵי בְּגוֹ מַשִּׁרְיָתָא כְּמָא דְשָׁרַן כֵּן נָטְלִין גְבַר עַל אַתְרֵהּ לְטִקְסֵיהוֹן: (יח) דֶּגֶל מַחֲנֵה אֶפְרַיִם לְצִבְאֹתָם יָמָּה וְנָשִׂיא לִבְנֵי אֶפְרַיִם אֱלִישָׁמָע בֶּן עַמִּיהוּד: טֵּקַס מַשְׁרִית אֶפְרַיִם לְחֵילֵיהוֹן מַעַרְבָא וְרַבָּא לִבְנֵי אֶפְרַיִם אֱלִישָׁמָע בַּר עַמִיהוּד: (יט) וּצְבָאוֹ פְּדָהצוּר: וְדִסְמִיכִין עֲלוֹהִי שִׁבְטָא דִמְנַשֶׁה וְרַבָּא לִבְנֵי מְנַשֶׁה גַמְלִיאֵל בַּר פְדָהצוּר: (כא) וּצְבָאוֹ וּפְקֻדֵיהֶם שְׁנַיִם וּשְׁלֹשִׁים אֶלֶף וּמָאתָיִם: וְחֵילֵהּ וּמִנְיָנֵיהוֹן תְּלָתִין וּתְרֵין אַלְפִין וּמָאתָן: (כב) וּמַטֵּה בִּנְיָמִן וְנָשִׂיא לִבְנֵי בִנְיָמִן אֲבִידָן בֶּן גִּדְעֹנִי: וְשִׁבְטָא דְבִנְיָמִן וְרַבָּא לִבְנֵי בִנְיָמִן אֲבִידָן בַּר אֶפְרַיִם מְאַת אֶלֶף וּשְׁמֹנַת אֲלָפִים וּמֵאָה לְצִבְאֹתָם וּשְׁלִשִׁים יִסָּעוּ: כָּל מִנְיָנַיָא לְמַשְׁרִית אֶפְרַיִם מְאָה וְתַמְנֵי אַלְפִין וּמְאָה לְחֵילֵיהוֹן בִּתְלִיתֵתָא נָטְלִין: (כה) דֶּגֶל מַחֲנֵה דָן צָפֹנָה לְצִבְאֹתָם וְנָשִׂיא לִבְנֵי דָן אֲחִיעֶזֶר בֶּן עַמִּישַׁדָּי: טֵקַס מַשְׁרִית דָן צִפּוּנָא לְחֵילֵיהוֹן וְרַבָּא לִבְנֵי דָן אֲחִיעֶזֶר בַּר עַמִישַׁדָי: (כו) וּצְבָאוֹ וּפְקֻדֵיהֶם שְׁנַיִם וְשִׁשִּׁים אֶלֶף וּשְׁבַע מֵאוֹת: וְחֵילֵהּ וּמִנְיָנֵיהוֹן שִׁתִּין וּתְרֵין אַלְפִין וּשְׁבַע מְאָה: (כז) וְהַחֹנִים עָלָיו מַטֵּה אָשֵׁר וְנָשִׂיא לִבְנֵי אָשֵׁר פַּגְעִיאֵל בֶּן עָכְרָן: וְדִי שְׁרַן םְמִכִין עֲלוֹהִי שִׁבְטָא דְאָשֵׁר וְרַבָּא לִבְנֵי אָשֵׁר פַּגְעִיאֵל בַּר עָכְרָן: (כח) וּצְבָאוֹ וּפְקֻדֵיהֶם אֶחָד וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת: וְחֵילֵהּ וּמִנְיָנֵיהוֹן אַרְבְּעִין וְחַד אַלְפִין וַחֲמֵשׁ מְאָה: (כט) וּמַטֵּה נַפְתָּלִי וְנָשִׂיא לִבְנֵי נַפְתָּלִי אֲחִירַע בֶּן עֵינָן: וְשִׁבְטָא דְנַפְתָּלִי וְרַבָּא מְאָה בְּבַתְרֵתָא נָטְלִין לְטִקְסֵיהוֹן: (לב) אֵלֶּה פְּקוּדֵי בְנֵי יִשְׂרָאֵל לְבֵית אֲבֹתָם כָּל פְּקוּדֵי הַמַּחֲנֹת לְצִבְאֹתָם שֵׁשׁ מֵאוֹת אֶלֶף וּשְׁלֹשֶׁת אֲלָפִים וַחֲמֵשׁ מֵאוֹת וַחֲמִשִּׁים: אִלֵין מִנְיָנֵי בְנֵי יִשְׂרָאֵל לְבֵית אֲבָהַתְהוֹן כָּל מִנְיָנֵי מַשִׁרְיָתָא לְחֵילֵיהוֹן שִׁית מְאָהּ וּתְלָתָא אַלְפִין וַחֲמֵשׁ מְאָה וְחַמְשִׁין: (לג) וְהַלְוִיִּם לֹא הָתְפָּקְדוּ בְּתוֹךְ בְּנֵי יִשְׂרָאֵל כַּאֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה: וְלֵוָאֵי לָא אִתְמְנִיאוּ בְּגוֹ בְּנֵי יִשְׂרָאֵל כְּמָא דְפַקֵיד יְיָ יָת משֶׁה: (לד) וַיַּעֲשׂוּ בְּנֵי יִשְׂרָאֵל כְּכֹל אֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה כֵּן חָנוּ לְדִגְלֵיהֶם וְכֵן נָסָעוּ אִישׁ לְמִשְׁפְּחֹתָיו עַל בֵּית אֲבֹתָיו: וַעֲבָדוּ בְּנֵי יִשְׂרָאֵל כְּכֹל דִי פַקֵיד יְיָ יָת משֶׁה כֵּן שָׁרַן לְטִּקְסֵיהוֹן וְכֵן נָטְלִין גְבַר לְזַרְעִיתֵיהּ עַל בֵּית אֲבָהָתוֹהִי:  רביעי ג (א) וְאֵלֶּה תּוֹלְדֹת אַהֲרֹן וּמֹשֶׁה בְּיוֹם דִּבֶּר יְהוָה אֶת מֹשֶׁה בְּהַר סִינָי: וְאִלֵין תּוּלְדַת אַהֲרֹן וּמשֶׁה בְּיוֹמָא דְמַלִיל יְיָ עִם משֶׁה בְּטוּרָא דְסִינָי: (ב) וְאֵלֶּה שְׁמוֹת בְּנֵי אַהֲרֹן הַבְּכוֹר נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר: וְאִלֵין שְׁמָהַת בְּנֵי אַהֲרֹן בּוּכְרָא נָדָב וַאֲבִיהוּא אֶלְעָזָר וְאִיתָמָר: (ג) אֵלֶּה שְׁמוֹת בְּנֵי אַהֲרֹן הַכֹּהֲנִים הַמְּשֻׁחִים אֲשֶׁר מִלֵּא יָדָם לְכַהֵן: אִלֵין שְׁמָהַת בְּנֵי אַהֲרֹן כַּהֲנַיָא דְאִתְרַבִּיאוּ דִי אִתְקָרֵב קֻרְבַּנְהוֹן לְשַׁמָשָׁא: (ד) וַיָּמָת נָדָב וַאֲבִיהוּא לִפְנֵי יְהוָה בְּהַקְרִבָם אֵשׁ זָרָה לִפְנֵי יְהוָה בְּמִדְבַּר סִינַי וּבָנִים לֹא הָיוּ לָהֶם וַיְכַהֵן אֶלְעָזָר וְאִיתָמָר עַל פְּנֵי אַהֲרֹן אֲבִיהֶם: וּמִית נָדָב וַאֲבִיהוּא קֳדָם יְיָ בְּקָרוֹבֵיהוֹן אֶשָׁתָא נוּכְרֵיתָא קֳדָם יְיָ בְּמַדְבְּרָא דְסִינַי וּבְנִין לָא הֲווֹ לְהוֹן וְשַׁמֵּשׁ אֶלְעָזָר וְאִיתָמָר עַל אַפֵּי אַהֲרֹן אֲבוּהוֹן: (ה) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר: וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר: (ו) הַקְרֵב אֶת מַטֵּה לֵוִי וְהַעֲמַדְתָּ אֹתוֹ לִפְנֵי אַהֲרֹן הַכֹּהֵן וְשֵׁרְתוּ אֹתוֹ: קָרֵב יָת שִׁבְטָּא דְלֵוִי וּתְקֵים יָתֵהּ קֳדָם אַהֲרֹן כַּהֲנָא וִישַׁמְשׁוּן יָתֵהּ: (ז) וְשָׁמְרוּ אֶת מִשְׁמַרְתּוֹ וְאֶת מִשְׁמֶרֶת כָּל הָעֵדָה לִפְנֵי אֹהֶל מוֹעֵד לַעֲבֹד אֶת עֲבֹדַת הַמִּשְׁכָּן: וְיִטְרוּן יָת מַטַרְתֵּהּ וְיָת מַטְרַת כָּל כְּנִשְׁתָּא קֳדָם מַשְׁכַּן זִמְנָא לְמִפְלַח יָת פָּלְחַן מַשְׁכְּנָא: (ח) וְשָׁמְרוּ אֶת כָּל כְּלֵי אֹהֶל מוֹעֵד וְאֶת מִשְׁמֶרֶת בְּנֵי יִשְׂרָאֵל לַעֲבֹד אֶת עֲבֹדַת הַמִּשְׁכָּן: וְיִטְרוּן יָת כָּל מָאנֵי מַשְׁכַּן זִמְנָא וְיָת מַטְּרַת בְּנֵי יִשְׂרָאֵל לְמִפְלַח יָת פָּלְחַן מַשְׁכְּנָא: (ט) וְנָתַתָּה אֶת הַלְוִיִּם לְאַהֲרֹן וּלְבָנָיו נְתוּנִם נְתוּנִם הֵמָּה לוֹ מֵאֵת בְּנֵי יִשְׂרָאֵל: וְתִתֵּן יָת לֵוָאֵי לְאַהֲרֹן וְלִבְנוֹהִי מְסִירִין יְהִיבִין אִנוּן לֵהּ מִן בְּנֵי יִשְׂרָאֵל: (י) וְאֶת אַהֲרֹן וְאֶת בָּנָיו תִּפְקֹד וְשָׁמְרוּ אֶת כְּהֻנָּתָם וְהַזָּר הַקָּרֵב יוּמָת: וְיָת אַהֲרֹן וְיָת בְּנוֹהִי תְּמַנֵי וְיִטְרוּן יָת כְּהֻנַתְהוֹן וְחִלוֹנַי דְיִקְרַב יִתְקְטֵל: (יא) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר: וּמַלִיל יְיָ עִם משֶׁה לְמֵימָר: (יב) וַאֲנִי הִנֵּה לָקַחְתִּי אֶת הַלְוִיִּם מִתּוֹךְ בְּנֵי יִשְׂרָאֵל תַּחַת כָּל בְּכוֹר פֶּטֶר רֶחֶם מִבְּנֵי יִשְׂרָאֵל וְהָיוּ לִי הַלְוִיִּם: וַאֲנָא הָא קְרֵבִית יָת לֵוָאֵי מִגוֹ בְּנֵי יִשְׂרָאֵל חֲלַף כָּל בּוּכְרָא פְּתַח וַלְדָא מִבְּנֵי יִשְׂרָאֵל וִיהוֹן מְשַׁמְשִׁין קֳדָמַי לֵוָאֵי: (יג) כִּי לִי כָּל בְּכוֹר בְּיוֹם הַכֹּתִי כָל בְּכוֹר בְּאֶרֶץ מִצְרַיִם הִקְדַּשְׁתִּי לִי כָל בְּכוֹר בְּיִשְׂרָאֵל מֵאָדָם עַד בְּהֵמָה לִי יִהְיוּ אֲנִי יְהוָה: אֲרֵי דִילִי כָּל בּוּכְרָא בְּיוֹמָא דִקְטָלִית כָּל בּוּכְרָא בְּאַרְעָא דְמִצְרַיִם אַקְדֵשִׁית קֳדָמַי כָּל בּוּכְרָא בְּיִשְׂרָאֵל מֵאֱנָשָׁא עַד בְּעִירָא דִילִי יְהוֹן אֲנָא יְיָ: חמישי (יד) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה בְּמִדְבַּר סִינַי לֵאמֹר: וּמַלִיל יְיָ עִם משֶׁה בְּמַדְבְּרָא דְסִינַי לְמֵימָר: (טו) פְּקֹד אֶת בְּנֵי לֵוִי לְבֵית אֲבֹתָם לְמִשְׁפְּחֹתָם כָּל זָכָר מִבֶּן חֹדֶשׁ וָמַעְלָה תִּפְקְדֵם: מְנֵי יָת בְּנֵי לֵוִי לְבֵית אֲבָהַתְהוֹן לְזַרְעֲיַתְהוֹן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא תִּמְנִנוּן: (טז) וַיִּפְקֹד אֹתָם מֹשֶׁה עַל פִּי יְהוָה כַּאֲשֶׁר צֻוָּה: וּמְנָא יָתְהוֹן משֶׁה עַל מֵימְרָא דַיְיָ כְּמָא דְאִתְפַּקָד: (יז) וַיִּהְיוּ אֵלֶּה בְנֵי לֵוִי בִּשְׁמֹתָם גֵּרְשׁוֹן וּקְהָת וּמְרָרִי: וַהֲווֹ אִלֵין בְּנֵי לֵוִי בִּשְׁמָהַתְהוֹן גֵרְשׁוֹן וּקְהָת וּמְרָרִי: (יח) וְאֵלֶּה שְׁמוֹת בְּנֵי גֵרְשׁוֹן לְמִשְׁפְּחֹתָם לִבְנִי וְשִׁמְעִי: וְאִלֵין שְׁמָהַת בְּנֵי גֵרְשׁוֹן לְזַרְעֲיַתְהוֹן לִבְנִי וְשִׁמְעִי: (יט) וּבְנֵי קְהָת לְמִשְׁפְּחֹתָם עַמְרָם וְיִצְהָר חֶבְרוֹן וְעֻזִּיאֵל: וּבְנֵי קְהָת לְזַרְעֲיַתְהוֹן עַמְרָם וְיִצְהָר חֶבְרוֹן וְעֻזִיאֵל: (כ) וּבְנֵי מְרָרִי לְמִשְׁפְּחֹתָם מַחְלִי וּמוּשִׁי אֵלֶּה הֵם מִשְׁפְּחֹת הַלֵּוִי לְבֵית אֲבֹתָם: וּבְנֵי מְרָרִי לְזַרְעֲיַתְהוֹן מַחְלִי וּמוּשִׁי אִלֵין אִנוּן זַרְעֲיַת לֵוָאֵי לְבֵית אֲבָהַתְהוֹן: (כא) לְגֵרְשׁוֹן מִשְׁפַּחַת הַלִּבְנִי וּמִשְׁפַּחַת הַשִּׁמְעִי אֵלֶּה הֵם מִשְׁפְּחֹת הַגֵּרְשֻׁנִּי: לְגֵרְשׁוֹן זַרְעִית לִבְנִי וְזַרְעִית שִׁמְעִי אִלֵין אִנוּן זַרְעֲיַת גֵרְשׁוֹן: (כב) פְּקֻדֵיהֶם בְּמִסְפַּר כָּל זָכָר מִבֶּן חֹדֶשׁ וָמָעְלָה פְּקֻדֵיהֶם שִׁבְעַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת: מִנְיָנֵיהוֹן בְּמִנְיַן כָּל דְכוּרָא מִבַּר יַרְחָא וּלְעֵלָא מִנְיָנֵיהוֹן שַׁבְעָא אַלְפִין וַחֲמֵשׁ מְאָה: (כג) מִשְׁפְּחֹת הַגֵּרְשֻׁנִּי אַחֲרֵי הַמִּשְׁכָּן יַחֲנוּ יָמָּה: זַרְעֲיַת גֵרְשׁוֹן אֲחוֹרֵי מַשְׁכְּנָא יִשְּׁרוּן מַעַרְבָא: (כד) וּנְשִׂיא בֵית אָב לַגֵּרְשֻׁנִּי אֶלְיָסָף בֶּן לָאֵל: וְרַב בֵּית אַבָּא לְבֵית גֵרְשׁוֹן אֶלְיָסָף בַּר לָאֵל: (כה) וּמִשְׁמֶרֶת בְּנֵי גֵרְשׁוֹן בְּאֹהֶל מוֹעֵד הַמִּשְׁכָּן וְהָאֹהֶל מִכְסֵהוּ וּמָסַךְ פֶּתַח אֹהֶל מוֹעֵד: וּמַטְרַת בְּנֵי גֵרְשׁוֹן בְּמַשְׁכַּן זִמְנָא מַשְׁכְּנָא וּפְרָסָא חוֹפָאֵהּ וּפְרָסָא דִתְרַע מַשְׁכַן זִמְנָא: (כו) וְקַלְעֵי הֶחָצֵר וְאֶת צִפּוּנָא: (לו) וּפְקֻדַּת מִשְׁמֶרֶת בְּנֵי מְרָרִי קַרְשֵׁי הַמִּשְׁכָּן וּבְרִיחָיו וְעַמֻּדָיו וַאֲדָנָיו וְכָל כֵּלָיו וְכֹל עֲבֹדָתוֹ: וְדִי מְסִיר לְמַטְרַת בְּנֵי מְרָרִי דַפֵּי מַשְׁכְּנָא בִּבְנֵי יִשְׂרָאֵל: וּמְנָא משֶׁה כְּמָא דִי פַקֵיד יְיָ יָתֵהּ יָת כָּל בּוּכְרָא בִּבְנֵי יִשְׂרָאֵל: (מג) וַיְהִי כָל בְּכוֹר זָכָר בְּמִסְפַּר שֵׁמוֹת מִבֶּן חֹדֶשׁ וָמַעְלָה לִפְקֻדֵיהֶם סִלְעִין בְּסִלְעֵי קוּדְשָׁא: (נא) וַיִּתֵּן מֹשֶׁה אֶת כֶּסֶף הַפְּדֻיִם לְאַהֲרֹן וּלְבָנָיו עַל פִּי יְהוָה כַּאֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה: וִיהַב משֶׁה יָת כְּסַף פְּרִיקַיָא לְאַהֲרֹן וְלִבְנוֹהִי עַל מֵימְרָא דַיְיָ כְּמָא דִי פַקֵיד יְיָ יָת משֶׁה:  שביעי ד (א) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה וְאֶל אַהֲרֹן לֵאמֹר: וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר: (ב) נָשֹׂא אֶת רֹאשׁ בְּנֵי קְהָת מִתּוֹךְ בְּנֵי לֵוִי לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם: קַבִּילוּ יָת חֻשְׁבַּן בְּנֵי קְהָת מִגוֹ בְּנֵי לֵוִי לְזַרְעֲיַתְהוֹן לְבֵית אֲבָהַתְהוֹן: (ג) מִבֶּן שְׁלֹשִׁים שָׁנָה וָמַעְלָה וְעַד בֶּן חֲמִשִּׁים שָׁנָה כָּל בָּא לַצָּבָא לַעֲשׂוֹת מְלָאכָה בְּאֹהֶל מוֹעֵד: מִבַּר תְּלָתִין שְׁנִין וּלְעֵלָא וְעַד בַּר חַמְשִׁין שְׁנִין כָּל דְאָתֵי לְחֵילָא לְמֶעְבַּד עִבִדְתָּא בְּמַשְׁכַּן זִ מְ נָא: (ד) זֹאת עֲבֹדַת בְּנֵי קְהָת בְּאֹהֶל מוֹעֵד קֹדֶשׁ הַקֳּדָשִׁים: דֵין פּוּלְחַן בְּנֵי קְהָת בְּמַשְׁכַּן זִמְנָא קוֹדֶשׁ קוּדְשַׁיָא: (ה) וּבָא אַהֲרֹן וּבָנָיו בִּנְסֹעַ הַמַּחֲנֶה וְהוֹרִדוּ אֵת פָּרֹכֶת הַמָּסָךְ וְכִסּוּ בָהּ אֵת אֲרֹן הָעֵדֻת: וְיֵיעוֹל אַהֲרֹן וּבְנוֹהִי בְּמֵטַל מַשְׁרִיתָא וִיפָרְקוּן יָת פָּרֻכְתָּא דִפְרָסָא וִיכַסוּן בַּהּ יָת אֲרוֹנָא דְסַהֲדוּתָא: (ו) וְנָתְנוּ עָלָיו כְּסוּי עוֹר תַּחַשׁ וּפָרְשׂוּ בֶגֶד כְּלִיל תְּכֵלֶת מִלְמָעְלָה וְשָׂמוּ בַּדָּיו: וְיִתְּנוּן עֲלוֹהִי חוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִפְרְסוּן לְבוּשׁ גְמִיר תִּכְלָא מִלְעֵלָא וִישַׁוּוּן אֲרִיחוֹהִי: (ז) וְעַל שֻׁלְחַן הַפָּנִים יִפְרְשׂוּ בֶּגֶד תְּכֵלֶת וְנָתְנוּ עָלָיו אֶת הַקְּעָרֹת וְאֶת הַכַּפֹּת וְאֶת הַמְּנַקִּיֹּת וְאֵת קְשׂוֹת הַנָּסֶךְ וְלֶחֶם הַתָּמִיד עָלָיו יִהְיֶה: וְעַל פָּתוֹרָא דִלְחֵם אַפַּיָא יִפְרְסוּן לְבוּשׁ תִּכְלָא וְיִתְּנוּן עֲלוֹהִי יָת מָגִּיסַיָא וְיָת בָּזִיכַיָא וְיָת מְכִילָתָא וְיָת קַסְווֹת נִסוּכָא וְלַחְמָא תְּדִירָא עֲלוֹהִי יְהֵי:  (ח) וּפָרְשׂוּ עֲלֵיהֶם בֶּגֶד תּוֹלַעַת שָׁנִי וְכִסּוּ אֹתוֹ בְּמִכְסֵה עוֹר תָּחַשׁ וְשָׂמוּ אֶת בַּדָּיו: וְיִפְרְסוּן עֲלֵיהוֹן לְבוּשׁ צְבַע זְהוֹרִי וִיכַסוּן יָתֵהּ בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁוּוּן יָת אֲרִיחוֹהִי: (ט) וְלָקְחוּ בֶּגֶד תְּכֵלֶת וְכִסּוּ אֶת מְנֹרַת הַמָּאוֹר וְאֶת נֵרֹתֶיהָ וְאֶת מַלְקָחֶיהָ וְאֶת מַחְתֹּתֶיהָ וְאֵת כָּל כְּלֵי שַׁמְנָהּ אֲשֶׁר יְשָׁרְתוּ לָהּ בָּהֶם: וְיִסְבוּן לְבוּשׁ תִּכְלָא וִיכַסוּן יָת מְנַרְתָּא דְאַנְהוֹרֵי וְיָת בּוֹצִינָהָא וְיָת צִבְתָהָא וְיָת מַחְתְּיָתָהָא וְיָת כָּל מָנֵי מִשְׁחָא דִי יְשַׁמְשׁוּן לָהּ בְּהוֹן: (י) וְנָתְנוּ אֹתָהּ וְאֶת כָּל כֵּלֶיהָ אֶל מִכְסֵה עוֹר תָּחַשׁ וְנָתְנוּ עַל הַמּוֹט: וְיִתְּנוּן יָתָהּ וְיָת כָּל מָנָהָא לְחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִתְּנוּן עַל אֲרִיחָא: (יא) וְעַל מִזְבַּח הַזָּהָב יִפְרְשׂוּ בֶּגֶד תְּכֵלֶת וְכִסּוּ אֹתוֹ בְּמִכְסֵה עוֹר תָּחַשׁ וְשָׂמוּ אֶת בַּדָּיו: וְעַל מַדְבְּחָא דְדַהֲבָא יִפְרְסוּן לְבוּשׁ תִּכְלָא וִיכַסוּן יָתֵהּ בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁוּוּן יָת אֲרִיחוֹהִי: (יב) וְלָקְחוּ אֶת כָּל כְּלֵי הַשָּׁרֵת אֲשֶׁר יְשָׁרְתוּ בָם בַּקֹּדֶשׁ וְנָתְנוּ אֶל בֶּגֶד תְּכֵלֶת וְכִסּוּ אוֹתָם בְּמִכְסֵה עוֹר תָּחַשׁ וְנָתְנוּ עַל הַמּוֹט: וְיִסְבוּן יָת כָּל מָנֵי שִׁמוּשָׁא דִי יְשַׁמְשׁוּן בְּהוֹן בְּקוּדְשָׁא וְיִתְּנוּן לִלְבוּשׁ תִּכְלָא וִיכַסוּן יָתְהוֹן בְּחוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וְיִתְּנוּן עַל אֲרִיחָא: (יג) וְדִשְּׁנוּ אֶת הַמִּזְבֵּחַ וּפָרְשׂוּ עָלָיו בֶּגֶד אַרְגָּמָן: וְיִסְפוּן יָת קִטְמָא דְמַדְבְּחָא וְיִפְרְסוּן עֲלוֹהִי לְבוּשׁ אַרְגְוָן: (יד) וְנָתְנוּ עָלָיו אֶת כָּל כֵּלָיו אֲשֶׁר יְשָׁרְתוּ עָלָיו בָּהֶם אֶת הַמַּחְתֹּת אֶת הַמִּזְלָגֹת וְאֶת הַיָּעִים וְאֶת הַמִּזְרָקֹת כֹּל כְּלֵי הַמִּזְבֵּחַ וּפָרְשׂוּ עָלָיו כְּסוּי עוֹר תַּחַשׁ וְשָׂמוּ בַדָּיו: וְיִתְּנוּן עֲלוֹהִי יָת כָּל מָנוֹהִי דִי יְשַׁמְשׁוּן עֲלוֹהִי בְהוֹן יָת מַחְתְּיָתָא וְיָת צִנוֹרְיָתָא וְיָת מַגְרוֹפְיָתָא וְיָת מִזְרְקַיָא כֹּל מָנֵי מַדְבְּחָא וְיִפְרְסוּן עֲלוֹהִי חוֹפָאָה דִמְשַׁךְ סַסְגוֹנָא וִישַׁווּן אֲרִיחוֹהִי: (טו) וְכִלָּה אַהֲרֹן וּבָנָיו לְכַסֹּת אֶת הַקֹּדֶשׁ וְאֶת כָּל כְּלֵי הַקֹּדֶשׁ בִּנְסֹעַ הַמַּחֲנֶה וְאַחֲרֵי כֵן יָבֹאוּ בְנֵי קְהָת לָשֵׂאת וְלֹא יִגְּעוּ אֶל הַקֹּדֶשׁ וָמֵתוּ אֵלֶּה מַשָּׂא בְנֵי קְהָת בְּאֹהֶל מוֹעֵד: וִישֵׁיצֵי אַהֲרֹן וּבְנוֹהִי לְכַסָאָה יָת קוּדְשָׁא וְיָת כָּל מָנֵי קוּדְשָׁא בְּמֵטַל מַשְׁרִיתָא וּבָתַר כֵּן יַעֲלוּן בְּנֵי קְהָת לְמִטַּל וְלָא יִקְרְבוּן לְקוּדְשָׁא וְלָא יְמוּתוּן אִלֵין מַטוּל בְּנֵי קְהָת בְּמַשְׁכַּן זִמְנָא: (טז) וּפְקֻדַּת אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן שֶׁמֶן הַמָּאוֹר וּקְטֹרֶת הַסַּמִּים וּמִנְחַת הַתָּמִיד וְשֶׁמֶן הַמִּשְׁחָה פְּקֻדַּת כָּל הַמִּשְׁכָּן וְכָל אֲשֶׁר בּוֹ בְּקֹדֶשׁ וּבְכֵלָיו: וְדִי מְסִיר לְאֶלְעָזָר בַּר אַהֲרֹן כַּהֲנָא מִשְׁחָא דְאַנְהָרוּתָא וּקְטוֹרֶת בּוּסְמַיָא וּמִנְחָתָּא תְדִירָא וּמִשְׁחָא דִרְבוּתָא מַטְרַת כָּל מַשְׁכְּנָא וְכָל דִי בֵהּ בְּקוּדְשָׁא וּבְמָנוֹהִי: מפטיר (יז) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה וְאֶל אַהֲרֹן לֵאמֹר: וּמַלִיל יְיָ עִם משֶׁה וְעִם אַהֲרֹן לְמֵימָר: (יח) אַל תַּכְרִיתוּ אֶת שֵׁבֶט מִשְׁפְּחֹת הַקְּהָתִי מִתּוֹךְ הַלְוִיִּם: לָא תְשֵׁיצוּן יָת שִׁבְטָא זַרְעֲיַת קְהָת מִגוֹ לֵוָאֵי: (יט) וְזֹאת עֲשׂוּ לָהֶם וְחָיוּ וְלֹא יָמֻתוּ בְּגִשְׁתָּם אֶת קֹדֶשׁ הַקֳּדָשִׁים אַהֲרֹן וּבָנָיו יָבֹאוּ וְשָׂמוּ אוֹתָם אִישׁ אִישׁ עַל עֲבֹדָתוֹ וְאֶל מַשָּׂאוֹ: וְדָא עִבִידוּ לְהוֹן     כולא יהיה קצף. אם תעשה כמצותי לא יהיה קצף, ואם לאו שיכנסו זרים בעבודתם זו יהיה קצף, כמו שמצינו במעשה קרח (במדבר יז, יא) כי יצא הקצף וגו': באתת. כל דגל יהיה לו אות. מפה צבועה תלויה בו. צבעו של זה לא כצבעו של זה, צבע כל אחד כגון אבנו הקבועה בחשן, ומתוך כך יכיר כל אחד את דגלו. דבר אחר באותות לבית אבתם באות שמסר להם יעקב אביהם כשנשאוהו ממצרים, שנאמר (בראשית נ, יב) ויעשו בניו לו כן כאשר צום, יהודה ויששכר וזבולן ישאוהו מן המזרח, וראובן ושמעון וגד מן הדרום וכו', כדאיתא בתנחומא (במדבר יב) בפרשה זו: מנגד. מרחוק מיל, כמו שנאמר ביהושע (יהושע ג, ד) אך רחוק יהיה ביניכם וביניו כאלפים אמה, שיוכלו לבא בשבת, משה ואהרן ובניו והלוים חונים בסמוך לו: קדמה. לפנים הקרויה קדם, ואיזו, זו רוח מזרחית. והמערב קרוי אחור: ראשנה יסעו. כשרואין הענן מסתלק, תוקעין הכהנים בחצוצרות ונוסע מחנה יהודה תחלה, וכשהולכין, הולכין כדרך חניתן, הלוים והעגלות באמצע, דגל יהודה במזרח, ושל ראובן בדרום, ושל אפרים במערב, ושל דן בצפון: ונסע אהל מועד. לאחר שני דגלים הללו: כאשר יחנו כן יסעו. כמו שפרשתי, הליכתן כחניתן, כל דגל מהלך לרוח הקבועה לו: על ידו. על מקומו. ואין לשון יד זז ממשמעו, רוח שלצדו, קרוי על ידו, הסמוכה לו לכל הושטת ידו. אי''ן שו''ן איש''א בלע''ז [על ידו]: ועליו. כתרגומו ודסמיכין עלוהי: ואלה תולדת אהרן ומשה. ואינו מזכיר אלא בני אהרן. ונקראו תולדות משה, לפי שלמדן תורה. מלמד שכל המלמד את בן חברו תורה, מעלה עליו הכתוב כאלו ילדו: ביום דבר ה' את משה. נעשו אלו התולדות שלו, שלמדן מה שלמד מפי הגבורה: על פני אהרן. בחייו: ושרתו אתו. ומהו השרות, ושמרו את משמרתו, לפי ששמירת המקדש עליו שלא יקרב זר, כמו שנאמר (במדבר יח, א) אתה ובניך ובית אביך אתך תשאו את עון המקדש, והלוים הללו מסיעין אותם, זה הוא השרות: ושמרו את משמרתו. כל מנוי שהאדם ממנה עליו ומטל עליו לעשותו, קרוי משמרת בכל המקרא ובלשון משנה, כמו שאמרו בבגתן ותרש (מגילה יג, ב) והלא אין משמרתי ומשמרתך שוה, וכן משמרות כהנה ולויה: ואת משמרת בני ישראל. שכלן היו זקוקין לצרכי המקדש, אלא שהלוים באים תחתיהם בשליחותם, לפיכך לוקחים מהם המעשרות בשכרן, שנאמר (במדבר יח, לא) כי שכר הוא לכם חלף עבודתכם: נתונים המה לו. לעזרה: מאת בני ישראל. כמו מתוך בני ישראל, כלומר משאר כל העדה נבדלו לכך בגזרת המקום, והוא נתנם לו, שנאמר (במדבר ח, יט) ואתנה את הלוים נתונים וגו': ואת אהרן ואת בניו תפקד. לשון פקידות ואינו לשון מנין: ושמרו את כהנתם. קבלת דמים וזריקה והקטרה ועבודות המסורות לכהנים: ואני הנה לקחתי. ואני מהיכן זכיתי בהן: מתוך בני ישראל. שיהיו ישראל שוכרין אותן לשרות שלי. על ידי הבכורות זכיתי בהם ולקחתים תמורתם, לפי שהיתה העבודה בבכורות, וכשחטאו בעגל נפסלו, והלוים שלא עבדו עבודה זרה נבחרו תחתיהם: מבן חדש ומעלה. משיצא מכלל נפלים הוא נמנה לקרא שומר משמרת הקדש. אמר ר' יהודה ברבי שלום למוד הוא אותו השבט להיות נמנה מן הבטן, שנאמר (במד' כו, כט) אשר ילדה אתה ללוי במצרים, עם כניסתה בפתח מצרים ילדה אתה. ונמנית בשבעים נפש, שכשאתה מונה חשבונם לא תמצאם אלא שבעים חסר אחת, והיא השלימה את המנין: על פי ה'. אמר משה לפני הקדוש ברוך הוא היאך אני נכנס לבתי כלם ולתוך אהליהם לדעת מנין יונקיהם. אמר לו הקדוש ברוך הוא עשה אתה את שלך ואני אעשה את שלי. הלך משה ועמד על פתח האהל והשכינה מקדמת לפניו, ובת קול יוצאת מן האהל ואומרת כך וכך תינוקות יש באהל זה, לכך נאמר על פי ה': לגרשון משפחת הלבני. כלומר לגרשון היו הפקודים משפחת הלבני ומשפחת השמעי פקודיהם כך וכך: המשכן. יריעות התחתונות: והאהל. יריעות עזים העשויות לגג: מכסהו. עורות אילים ותחשים: ומסך פתח. הוא הוילון: ואת מיתריו. של משכן והאהל ולא של חצר: משפחת בני קהת יחנו וגו' תימנה. וסמוכין להם דגל ראובן החונים תימנה, אוי לרשע ואוי לשכנו, לכך לקו מהם דתן ואבירם ומאתים וחמשים איש עם קרח ועדתו, שנמשכו עמהם במחלקתם: והמסך. היא הפרכת, שאף היא קרויה פרכת המסך: ונשיא נשיאי הלוי. ממנה על כלם. ועל מה היא נשיאותו, פקדת שומרי משמרת הקדש, על ידו היה פקדת כלם: משה ואהרן ובניו. וסמוכין להם דגל מחנה יהודה, והחונים עליו יששכר וזבולן, טוב לצדיק טוב לשכנו, לפי שהיו שכניו של משה שהיה עוסק בתורה, נעשו גדולים בתורה, שנאמר (תהלים ס, ט) יהודה מחקקי, ומבני יששכר יודעי בינה וגו' (דה''י א' יב לג) מאתים ראשי סנהדראות, ומזבולן משכים בשבט סופר (שופטים ה יד): אשר פקד משה ואהרן. נקוד על ואהרן, לומר שלא היה במנין הלוים: שנים ועשרים אלף. ובפרטן אתה מוצא שלש מאות יתרים, בני גרשון שבעת אלפים וחמש מאות, בני קהת שמונת אלפים ושש מאות, בני מררי ששת אלפים ומאתים, [הרי שנים ועשרים אלף ושלש מאות]. ולמה לא כללן עם השאר ויפדו את הבכורות, ולא יהיו זקוקים השלשה ושבעים ומאתים בכורות העודפים על המנין לפדיון, אמרו רבותינו במסכת בכורות (בכורות ה א) אותן שלש מאות לוים בכורות היו ודים שיפקיעו עצמם מן הפדיון: פקד כל בכור זכר וגו' מבן חדש ומעלה. משיצא מכלל ספק נפלים: ואת בהמת הלוים וגו'. לא פדו בהמות הלוים את בכורי בהמה טהורה שלישראל, אלא את פטרי חמוריהם, ושה אחד של בן לוי פטר כמה פטרי חמורים שלישראל, תדע שהרי מנה העודפים באדם ולא מנה העודפים בבהמה: ואת פדויי השלשה וגו'. ואת הבכורות הצריכין להפדות בהם, אלו השלשה ושבעים ומאתים העודפים בהם, יתרים על הלוים, מהם תקח חמשת שקלים לגלגלת, כך היתה מכירתו של יוסף עשרים כסף, שהיה בכורה של רחל: העדפים על פדויי הלוים. על אותן שפדו הלוים בגופן: חמשה וששים ושלש מאות ואלף. כך סכום החשבון חמשת שקלים לגלגלת. למאתים בכורות אלף שקל, לשבעים בכורות שלש מאות וחמשים שקל, לשלש בכורות חמשה עשר שקל. אמר כיצד אעשה, בכור שאומר לו תן חמשת שקלים, יאמר לי אני מפדויי הלוים. מה עשה, הביא שנים ועשרים אלף פתקין וכתב עליהן בן לוי, ומאתים ושבעים ושלשה פתקין כתב עליהן חמשה שקלים, בללן ונתנן בקלפי, אמר להם בואו וטלו פתקיכם לפי הגורל: נשא את ראש וגו'. מנה מהם את הראויין לעבודת משא, והם מבן שלשים ועד בן חמשים שנה, והפחות משלשים לא נתמלא כחו. מכאן אמרו בן שלשים לכח, והיותר על בן חמשים כחו מכחיש מעתה: קדש הקדשים. המקדש שבכלן הארון והשלחן והמנורה והמזבחות והפרוכת וכלי שרת: ובא אהרן ובניו וגו'. יכניסו כל כלי וכלי לנרתקו המפורש לו בפרשה זו, ולא יצטרכו הלוים בני קהת אלא לשאת: בנסוע המחנה. כשהענן מסתלק הם יודעין שיסעו: קערת וכפות קשות ומנקיות. כבר פרשתי במלאכת המשכן: הנסך. הכסוי לשון מסך, כדכתיב (שמות כה, כט) אשר יסך בהן: מלקחיה. כמין צבת שמושך בה את הפתילה לכל צד שירצה: מחתתיה. כמין כף קטנה ושוליה פשוטין ולא סגלגלים, ואין לה מחצה לפניה אלא מצדיה, וחותה בה את דשן הנרות כשמטיבן: נרתיה. לוצי''ש בלע''ז [מנורות] שנותנים בהן השמן והפתילות: אל מכסה עור תחש. כמין מרצוף: את כל כלי השרת אשר ישרתו בם בקדש. בתוך המשכן שהוא קדש, והן כלי הקטרת שמשרתין בהם במזבח הפנימי: ודשנו את המזבח. מזבח הנחשת: ודשנו. יטלו את הדשן מעליו: ופרשו עליו בגד ארגמן. ואש שירדה מן השמים רבוצה תחת הבגד כארי בשעת המסעות, ואינה שורפתו, שהיו כופין עליה פסכתר של נחשת: מחתת. שבהן חותים גחלים לתרומת הדשן, שעשויה כמין מחבת שאין לה אלא שלש מחצות, ומלפניה שואבת את הגחלים: מזלגות. צנוריות של נחשת שבהן מכין באברים שעל המזבח להפכן כדי שיתעכלו יפה ומהר: יעים. הם מגרפות ובלע''ז ודי''ל [מגרפה] והן של נחשת ובהן מכבדין את הדשן מעל המזבח: לכסת את הקדש. הארון והמזבח: ואת כל כלי הקדש. המנורה וכלי שרת: ומתו. שאם יגעו חייבין מיתה בידי שמים: ופקדת אלעזר. שהוא ממנה עליהם לשאת אותם, שמן וקטרת ושמן המשחה ומנחת התמיד, עליו מטל לצוות ולזרז ולהקריב בעת חנייתן: פקדת כל המשכן. ועוד היה ממנה על משא בני קהת לצוות איש איש על עבודתו ועל משאו, והוא המשכן וכל אשר בו, כל הסדורים למעלה בפרשה זו, אבל משא בני גרשון ומררי שאינן מקדש הקדשים, על פי איתמר היה, כמו שכתוב בפרשת נשא (פסוק כח): אל תכריתו. אל תגרמו להם שימותו: ולא יבאו לראות כבלע את הקדש. לתוך נרתק שלו, כמו שפרשתי למעלה (פסוק ה) בפרשה זו, ופרשו עליו בגד פלוני וכסו אותו     53. וְלֹא יִהְיֶה קֶצֶף / PARA QUE NO HAYA CÓLERA. Ello implica que si hacen conforme a este mandamiento, no habrá cólera divina; pero si no, a saber, que los legos entran para participar en la labor propia de los leviim, entonces sí habrá cólera. Esto es similar a lo que hallamos con respecto al incidente provocado por Kóraj, donde vemos que Di-s causó una plaga por razones similares: “Pues ha surgido la cólera [הַקֶּצֶף] de parte del Eterno.”  2. בְאֹתֹת / CONFORME A LAS INSIGNIAS. Cada estandarte tendrá una insignia distintiva: un pedazo de tela teñida de un color particular, que estará colgada a la vista de todos. El color de cada una de las insignias será diferente del color de las demás. El color de la insignia de cada uno de los estandartes será el mismo que el color de la piedra específica que le ha sido fijada en el Pectoral [joshén]. De este modo cada uno podrá reconocer su estandarte. Según otra explicación, esta frase también puede ser leída como “según los signos [בְּאֹתֹת], conforme a sus casas paternas”, queriendo decir: según el signo que su ancestro Yaacob les transmitió cuando, después de fallecer, lo transportaron de Mitzráim para sepultarlo en la Tierra de Israel, como se declara: “Sus hijos [de Yaacob] hicieron por él conforme él les había encomendado.” La forma en que lo transportaron a Israel es la siguiente: Yehudá, Isajar y Zebulún portaron su ataúd del lado este; Reubén, Shimón y Gad, del lado sur, etc., como se declara en el comentario del midrash Tanjumá a esta sección.  מִנֶּגֶד / A CIERTA DISTANCIA. A una distancia de un mil, como se declara en el libro de Yehoshúa: “Pero habrá una distancia entre ustedes [los israelitas] y ella [el Arca] como de dos mil codos.” La razón de esta distancia era para que pudieran venir a las inmediaciones del Tabernáculo en Shabat, ya que Moshé, Aharón y sus hijos, y los leviim acampaban junto al Tabernáculo. 3. קֵדְמָה / AL FRENTE. La parte frontal de algo es llamada קֶדֶם. ¿Y qué lado es? Es el lado oriental. De igual manera, el oeste es llamado אָחוֹר, “parte posterior”.  9. רִאשֹׁנָה יִסָּעוּ / ELLOS VIAJARÁN EN PRIMER LUGAR. Cuando veían que la Nube que los guiaba en el desierto emprendía la marcha, los kohanim [sacerdotes] daban la señal de partir tocando trompetas. El campamento de Yehudá era el primero en iniciar la marcha. Y cuando las tribus de Israel marchaban por el desierto, marchaban siguiendo el orden que tenían cuando acampaban: los leviim y los carruajes en medio; el estandarte de Yehudá al este; el de Reubén al sur; el de Efraim al oeste, y el de Dan al norte. 17. וְנָסַע אֹהֶל מוֹעֵד / LA TIENDA DE LA CITA... VIAJARÁ. Detrás de estos dos estandartes.  כַּאֲשֶׁר יַחֲנוּ כֵּן יִסָּעוּ / TAL COMO ACAMPEN, ASÍ VIAJARÁN. Tal como ya expliqué, que el orden en que viajaban era el mismo que cuando acampaban; cada estandarte marchaba en el lado que le había sido asignado. עַל יָדוֹ / EN SU SITIO. Aquí esta expresión significa: en su sitio. A pesar de ello, el vocablo יָד no se aparta de su sentido usual. Pues el lugar situado junto a la mano de uno es llamado עַל יָדוֹ, literalmente, “lo que está a la mano”, ya que es lo que está próximo a uno siempre que se extienda la mano. En francés antiguo, en son aise.  20. וְעָלָיו / Y PRÓXIMO A ÉL. Aquí este término debe ser entendido tal como lo traduce el Targum: וְדִסְמִיכִין עֲלוֹהִי, y los que están cerca de él eran los de la tribu de Menashé.  1. וְאֵלֶּה תּוֹלְדוֹת אַהֲרֹן וּמֹשֶׁה / ESTAS SON LAS GENERACIONES DE AHARÓN Y DE MOSHÉ. Pero aunque habla de “las generaciones de Aharón y de Moshé, aquí la Torá sólo menciona a los hijos de Aharón. En realidad, los hijos de Aharón fueron llamados “generaciones [hijos] de Moshé” porque él les enseñó Torá. Esto enseña que cualquiera que enseñe Torá al hijo de su prójimo, la Escritura se lo considera como si él mismo lo hubiera engendrado.  בְּיוֹם דִּבֶּר ה' אֶת מֹשֶׁה / EL DÍA QUE EL ETERNO HABLÓ A MOSHÉ. Esta descendencia de Aharón se hizo espiritualmente suya [de Moshé] porque en ese día él por primera vez les enseñó lo que había aprendido de la boca del Omnipotente.  4. עַל פְּנֵי אַהֲרֹן / EN PRESENCIA DE... AHARÓN. Aquí la frase עַל פְּנֵי implica: en vida de él.  6. וְשֵׁרְתוּ אֹתוֹ / PARA QUE LE SIRVAN. ¿Y cuál es la naturaleza de este servicio? La respuesta la da el versículo siguiente: “Ellos custodiarán el cargo de él.” Pues la custodia del Santuario, a saber, que en él no deberá entrar ningún lego, le incumbía a él [Aharón], como se declara: “Tú y tus hijos, y la casa de tu padre contigo, portarán la iniquidad del Santuario.” Y los leviim les asistían en esta tarea; a esto se refiere el “servicio” mencionado en este versículo.  7. וְשָׁמְרוּ אֶת מִשְׁמַרְתּוֹ / ELLOS CUSTODIARÁN EL CARGO DE ÉL. Cualquier puesto o cargo sobre el cual un individuo haya sido nombrado, y que es responsabilidad suya cumplirlo, es denominado מִשְׁמֶרֶת en todo lugar de la Escritura y en el lenguaje de la Mishná. De modo similar se declara respecto a Bigtán y Teresh: “Pero mi cargo oficial [מִשְׁמֶרֶת] y tu cargo oficial no son iguales.” Y lo mismo se aplica a los cargos de la kehuná [sacerdocio] y de la leviyá [el levirato]. 8. וְאֶת מִשְׁמֶרֶת בְּנֵי יִשְׂרָאֵל / Y EL CARGO DE LOS HIJOS DE ISRAEL. Se le denomina así porque todos, incluidos los israelitas, estaban obligados a cuidar de las necesidades del Santuario, sólo que los leviim ocupaban su lugar, representándolos. Por ello es que los leviim tomaban el diezmo [maaser] de los israelitas a modo de recompensa por su representación, como se declara respecto al diezmo: “Pues es el pago para ustedes a cambio de su servicio en la Tienda de la Cita.”  9. נְתוּנִם הֵמָּה לוֹ / DONADOS SON ELLOS PARA ÉL. A modo de ayuda.  מֵאֵת בְּנֵי יִשְׂרָאֵל / DE PARTE DE LOS HIJOS DE ISRAEL. Esta frase es similar en significado a decir: “de entre [מִתּוֹךְ] los Hijos de Israel.” Es decir, del resto de la asamblea ellos han sido separados para esta tarea por decreto del Omnipotente, y Él es quien se los ha dado a él, como se declara: “Y Yo entregué a los leviim [como] donados a Aharón y a sus hijos de entre los Hijos de Israel, etc.”  10. וְאֶת אַהֲרֹן וְאֶת בָּנָיו תִּפְקֹד / Y COMISIONARÁS A AHARÓN Y A SUS HIJOS. La palabra תִּפְקֹד en este versículo denota comisionar y no computar.  וְשָׁמְרוּ אֶת כְּהֻנָּתָם / Y ELLOS CUSTODIARÁN SU CARGO SACERDOTAL. Éste consiste en la recepción de la sangre de las ofrendas en recipientes especiales, rociarla en el Altar, quemar las partes designadas del animal, así como otras tareas entregadas a cargo de los kohanim.  12. וַאֲנִי הִנֵּה לָקַחְתִּי / Y EN CUANTO A MÍ, HE AQUÍ QUE HE TOMADO. Quiere decir: Y en cuanto a Mí, ¿de qué manera tomé derechos sobre ellos, los leviim, “de entre los Hijos de Israel” para que los israelitas los hayan, por decirlo así, alquilado para desempeñar Mi servicio? Por medio de los primogénitos israelitas que no dejé que perecieran cuando murieron los de Mitzráim es que Yo poseo ahora derechos sobre ellos, ya que tomé a los leviim a Mi servicio a cambio de ellos. E inicialmente el servicio a Di-s incumbía a los primogénitos, pero cuando éstos pecaron con el Becerro de Oro se descalificaron para dicho servicio. Pero los leviim, quienes no rindieron culto a la idolatría, fueron escogidos para el servicio divino en lugar de ellos.  15. מִבֶּן חֹדֶשׁ וָמַעְלָה / DE UN MES DE EDAD EN ADELANTE. A partir de que el recién nacido salga de la categoría de nefalim será contado para llamársele “guardián de la custodia del Santuario”. Rabí Yehudá hijo de Rabí Shalom, dijo: Es usual para los miembros de esta tribu de Leví ser contados desde que nacen, como se declara respecto a Yojébed: “Y el nombre de la esposa de Amram era Yojébed, hija de Leví, a quien [su madre] dio a luz para Leví en Mitzráim”, lo cual es explicado en el sentido de que su madre la dio a luz justo al entrar por la puerta de la ciudad fronteriza de Mitzráim, y aun así es incluida como parte de las setenta personas que fueron a Mitzráim junto con Yaacob. Pues cuando haces la cuenta de la cantidad de personas que fueron con Yaacob, hallarás que sólo son sesenta y nueve, y ella fue la que completó el número de setenta.  16. עַל פִּי ה' / POR LA PALABRA DEL ETERNO. Moshé dijo ante el Santo –bendito es–: “¿Cómo he de entrar al interior de sus tiendas para averiguar el número de niños pequeños?” El Santo –bendito es– le respondió: “Tú haz lo tuyo y Yo haré lo Mío.” Moshé iba y se paraba a la entrada de cada una de las tiendas, habiendo entrado primero la Presencia Divina a la tienda, y un eco divino [bat kol] salía de la tienda y le decía: “Tantos y tantos niños pequeños hay en esta tienda.” Por esta razón es que respecto al censo de los leviim se declara: “Por la palabra del Eterno.”  21. לְגֵרְשׁוֹן מִשְׁפַּחַת הַלִּבְנִי / DE GUERSHÓN [ERA LA] FAMILIA DE LIBNÍ. Es decir, de Guershón, los que fueron contados fueron la familia de Libní y la familia de Shimí, y de ellos los contados fueron tantos y tantos, como se menciona en el v. 22.  25. הַמִשְׁכָּן / EL TABERNÁCULO. Aquí este término designa a las cortinas inferiores.  וְהָאֹהֶל / LA TIENDA. Esto se refiere a las cortinas de vello de cabra, las cuales fueron hechas a modo de techo para las cortinas inferiores. מִכְסֵהוּ / SU CUBIERTA. Esto se refiere a las pieles de carnero y de tájash. וּמָסַךְ פֶּתַח / Y LA PANTALLA DE LA ENTRADA. Esto se refiere a la cortina vertical [vilón].  26. וְאֶת מֵיתָרָיו / ASÍ COMO SUS CUERDAS. Del “Tabernáculo” [las cortinas inferiores] y de la “Tienda” [las cortinas superiores], pero no las cuerdas del Atrio. 29. מִשְׁפְּחֹת בְּנֵי קְהַת יַחֲנוּ... תֵּימָנָה  / LAS FAMILIAS DE LOS HIJOS DE KEHAT ACAMPABAN... AL SUR. Y junto a ellas, las tres tribus reunidas en torno al estandarte de Reubén, que también acampaban al sur. ¡Ay del malvado y ay de su vecino! Pues debido a esto algunos de los de la tribu de Reubén –Datán y Abiram y doscientos cincuenta hombres más, que en su mayoría eran de la tribu de Reubén– fueron castigados junto con Kóraj y su grupo, que eran de la tribu de Leví, por haberse dejado llevar en la disensión que ellos provocaron.  31. וְהמָּסָךְ / Y LA PANTALLA. Esto se refiere al Velo [parójet], que también es llamado “pantalla”.  32. וּנְשִׂיא נְשִׂיאֵי הַלְּוִי / Y... EL DIRIGENTE DE LOS LÍDERES DE LOS LEVIIM. Esto quiere decir que Elazar había sido nombrado líder sobre todos. ¿Y en qué consistía su liderazgo? La supervisión de los guardianes del cargo del Tabernáculo estaba en sus manos; en eso consistía su nombramiento sobre todos.  38. מֹשֶׁה וְאַהֲרֹן וּבָנָיו / MOSHÉ Y AHARÓN Y SUS HIJOS. Y junto a ellos estaba el estandarte del campamento de la tribu de Yehudá; y junto a él estaban las tribus de Isajar y Zebulún. ¡Bueno es para el justo, y bueno es para su vecino! Puesto que los miembros de estas tribus eran vecinos de Moshé –quien permanentemente se dedicaba a la Torá– se volvieron grandes en Torá, como se declara respecto a la tribu de Yehudá: “Yehudá es mi legislador.” Y respecto a la tribu de Isajar: “Y de los hijos de Isajar, que eran hombres con entendimiento de los tiempos para saber lo que Israel debía hacer, sus jefes eran doscientos”, frase que alude a los doscientos presidentes del Sanhedrín que surgieron de esta tribu. Y respecto a Zebulún: “Y de Zebulún surgieron los que manejan la vara del escriba.”  39. אֲשֶׁר פָּקַד מֹשֶׁה וְאַהֲרֹן / QUE MOSHÉ Y AHARÓN HABÍAN CONTADO. En el texto hebreo, encima del nombre אַהֲרֹן, “Aharón”, hay puntos, que sirven para indicar que él no fue incluido en el número total de leviim.  שְׁנַיִם וְעֶשְׂרִים אָלֶף / VEINTIDÓS MIL. Sin embargo, al contarlos en detalle encontrarás que hay trescientos de más: los descendientes de Guershón son siete mil quinientos; los descendientes de Kehat, ocho mil seiscientos; y los descendientes de Merarí, seis mil doscientos, lo que da un total de veintidós mil trescientos. Siendo así, ¿por qué la Torá no incluyó a los trescientos adicionales junto con los demás, de tal modo que el número total de leviim pudiera redimir al número total de los primogénitos israelitas? Si los hubiera incluido, los doscientos setenta y tres primogénitos israelitas que se excedían de la cuenta total de leviim no hubieran tenido necesidad de rescate. Al respecto, nuestros Maestros dijeron en el Tratado de Bejorot que esos mismos trescientos leviim eran primogénitos, y les era suficiente con liberarse ellos mismos de la necesidad del rescate, sin además tener que redimir a los primogénitos israelitas.  40. פְּקֹד כָּל בְּכֹר זָכָר וְגוֹ' מִבֶּן חֹדֶשׁ וָמָעְלָה / CUENTA A TODOS LOS PRIMOGÉNITOS VARONES... A PARTIR DE UN MES EN ADELANTE. Es decir, a partir de que salga de la categoría de néfel, un recién nacido considerado no–viable.  45. וְאֶת בֶּהֱמַת הַלְוִיִּם וְגוֹ' / Y LAS BESTIAS DE LOS LEVIIM, ETC. En realidad, los animales de los leviim no redimieron a los animales primerizos puros de los israelitas, sino sólo a los asnos que primero abrieron la matriz de sus madres. Un solo cordero de un leví liberaba a varios asnos de los israelitas que primero hubieron abierto la matriz de sus madres. Sabrás que así es puesto que la Torá enumeró a los seres humanos excedentes de los primogénitos israelitas, mientras que no enumeró a los animales excedentes de los israelitas.  46. וְאֶת פְּדוּיֵי הַשְּׁלֹשָׁה וְגוֹ' / Y EN CUANTO A LOS [QUE DEBEN SER] REDIMIDOS DE LOS... TRES... ETC. Este versículo y el siguiente deben ser entendidos del siguiente modo: Y en cuanto a los primogénitos israelitas que necesitan ser redimidos, siendo éstos los doscientos setenta y tres que hay en exceso al número de leviim, de ellos tomarás cinco siclos de plata [shekalim] por cabeza. Ése había sido el precio de venta de Yosef –veinte piezas de plata–, ya que él era el primogénito de Rajel.  49. הָעֹדְפִים עַל פְּדוּיֵי הַלְוִיִּם / QUE EXCEDÍAN A LOS REDIMIDOS DE LOS LEVIIM. Esta frase significa: Que excedían en número a aquellos que los leviim habían redimido con sus propias personas.  50. חֲמִשָּׁה וְשִׁשִּׁים וּשְׁלֹשׁ מֵאוֹת וָאֶלֶף / MIL TRESCIENTOS SESENTA Y CINCO. Esta es la cuenta total de “cinco siclos por cabeza”: cinco siclos multiplicados por doscientos primogénitos equivalen a mil siclos; más cinco siclos multiplicados por setenta primogénitos, trescientos cincuenta siclos; más cinco siclos multiplicados por tres primogénitos, quince siclos. Pero Moshé todavía tenía un problema. Pensó para sí: ¿Cómo he de hacer? Si le digo a un cierto primogénito israelita: “Entrégame los cinco siclos de tu rescate”, él me podrá responder: “No tengo por qué dártelos; pídeselos a otro, pues yo soy de los que son redimidos por los leviim.” ¿Qué fue lo que Moshé hizo para resolver el problema? Trajo veintidós mil papeletas y escribió en cada una de ellas “leví”, y en otras doscientas setenta y tres papeletas escribió “cinco siclos”. Las revolvió y las puso en urnas. Luego dijo a todos los israelitas: “Van y tomen sus papeletas por sorteo”, y de este modo evitó que cada israelita pudiera librarse de pagar los cinco siclos. 2. נָשֹׂא אֶת רֹאשׁ וְגוֹ' / HAGAN EL CENSO, ETC. Es decir, de entre ellos haz la cuenta de los que son aptos para el servicio de portar [מַשָּׂא] los utensilios del Tabernáculo. Y ellos eran los que estaban comprendidos entre las edades de treinta y cincuenta años, pues el que tenía menos de treinta años todavía no había alcanzado la plenitud de su fuerza. Sobre la base de este versículo nuestros Maestros dijeron: “Un hombre de treinta años ha alcanzado la plenitud de su fuerza.” Por otro lado, el que era mayor de cincuenta años, su fuerza comenzaba a disminuir.  4. קֹדֶשׁ הַקֳּדָשִׁים / LO MÁS SANTO. Aquí esta expresión se refiere a los utensilios del Tabernáculo más sagrados de todos: el Arca [arón]; la Mesa [shulján]; el Candelabro [menorá]; los dos Altares [mizbéaj]; el Velo [parójet]; y los demás utensilios del servicio sacrificatorio.  5. וּבָא אַהֲרֹן וּבָנָיו וְגוֹ' / AHARÓN Y SUS HIJOS VENDRÁN, ETC. Deberán meter cada uno de los utensilios en su envoltura protectora que es mencionada explícitamente en esta sección, de tal modo que los leviim descendientes de Kehat sólo tengan que portarlos.  בִּנְסֹעַ הַמַּחֲנֶה / CUANDO EL CAMPAMENTO EMPRENDA EL VIAJE. Cuando la Nube enviada por Di-s se aleje, ellos sabrán que deberán emprender la marcha.  7. קְעָרֹת וְכַּפֹּת וּמְנַקִּיֹּת וּקְשׂוֹת / PLATOS, CUCHARONES, SOPORTES Y TUBOS. Ya he explicado estos utensilios en la sección que trata acerca de la labor del Tabernáculo.  הַנָּסֶךְ / QUE RECUBREN. Este vocablo literalmente significa “la cubierta”; es de la misma raíz que מָסָךְ, “pantalla”, “cubierta”, como está escrito respecto a estos utensilios: “Con los cuales [la Mesa] será cubierta.”  9. מַלְקָחֶיהָ / SUS TENAZAS. Estos utensilios eran como una especie de tenazas con las cuales jalaban los pabilos hacia cualquier lado que se desease.  מַחְתֹּתֶיהָ / SUS PALETAS. Este utensilio era como una especie de cucharón pequeño, cuyo fondo era plano y no cóncavo que no tenía salientes hacia el frente sino únicamente a los lados. Con ella se rastrillaban las cenizas de las candelas cuando se las limpiaba.  נֵרֹתֶיהָ / SUS CANDELAS. Esta palabra significa lo mismo que luces en francés antiguo.  Eran los utensilios en los cuales ponían el aceite y los pabilos.  10. אֶל מִכְסֵה עוֹר תָּחַשׁ / DENTRO DE UNA CUBIERTA DE PIEL DE TÁJASH. Era como una especie de alforja o saco grande.  12. אֶת כָּל כְּלֵי הַשָּׁרֵת אֲשֶׁר יְשָׁרְתוּ בָם בַּקֹּדֶשׁ / TODOS LOS UTENSILIOS DE SERVICIO CON LOS CUALES MINISTRABAN EN EL SANTUARIO. Es decir, dentro del Tabernáculo mismo [mishkán], que era santo. Y éstos eran los utensilios para la quema del sahumerio con los que se oficiaba en el Altar interior de oro.  13. וְדִשְּׁנוּ אֶת הַמִּזְבֵּחַ / DESHOLLINARÁN EL ALTAR. Esto se refiere al Altar exterior de cobre, situado en el Atrio.  וְדִשְּׁנוּ / DESHOLLINARÁN. Esta palabra significa que le quitarán las cenizas de encima.  וּפָרְשׂוּ עָלָיו בֶּגֶד אַרְגָּמָן / Y EXTENDERÁN SOBRE ÉL UNA PRENDA DE LANA PÚRPURA. Pero el fuego que había caído del cielo sobre el Altar yacía debajo de la vestimenta, semejante a un león echado, durante las jornadas de viaje en el desierto. Sin embargo, no quemaba la vestimenta que lo cubría porque encima de él ponían un enorme recipiente de cobre.  14. מַחְתֹּת / BRASEROS. Con los cuales rastrillaban las brasas para la separación de las cenizas. Estaban hechos como una especie de cazuela que únicamente tenía tres salientes a los lados y en la parte posterior, y por la parte frontal absorbía las brasas.  מִזְלָגֹת / GARFIOS. Eran ganchillos de cobre con los que golpeaban los miembros de los animales sacrificados que estaban sobre el Altar de cobre para voltearlos, a fin de que se consumieran bien y rápidamente. יָעִים / BADILES. Eran paletillas, y en francés antiguo son llamadas vedil. Eran de cobre, y con ellas se limpiaban las cenizas que estaban sobre el Altar.  15. לְכַסֹּת אֶת הַקֹּדֶשׁ / DE RECUBRIR LO SANTO. Es decir, el Arca y el Altar.  וְאֶת כָּל כְּלֵי הַקֹּדֶשׁ / Y TODOS LOS UTENSILIOS DEL SANTUARIO. El Candelabro y los utensilios de servicio. וָמֵתוּ / Y MUERAN. Pues si tocan los utensilios sagrados, se harán acreedores a la pena de muerte a manos del Cielo.  16. וּפְקֻדַּת אֶלְעָזָר / Y EL CARGO DE ELAZAR. Que era el que había sido puesto a cargo de ellos para portarlos, consistía en: el aceite, el sahumerio, y el aceite de la unción y la oblación de harina continua [minjat hatamid]. A él le correspondía el cargo de ordenar y urgir que las tareas necesarias fueran realizadas a la hora en que acampaban, así como que la ofrenda correspondiente a cada circunstancia fuera ofrecida.  פְּקֻדַּת כָּל הַמִּשְׁכָּן / LA COMISIÓN DE TODO EL TABERNÁCULO. Y Elazar también había sido nombrado sobre el servicio de carga que realizaban los descendientes de Kehat, para dar órdenes y urgir a cada hombre respecto a su servicio y a su carga, la cual consistía en el Tabernáculo mismo y todo lo que contenía, a saber, todos los artefactos enumerados más arriba en esta misma sección, cuyo estado era Santísimo [kódesh hakodashim]. Pero la carga que debían portar los descendientes de Guershón y de Merarí –cuyo estado no era Santísimo– quedaba bajo las órdenes de Itamar, como está escrito en la parashat Nasó.  18. אַל תַּכְרִיתוּ / NO PROVOQUEN LA EXTERMINACIÓN. Es decir, no sean los causantes de que los leviim mueran.  20. וְלֹא יָבֹאוּ לִרְאוֹת כְּבַלַּע אֶת הַקֹּדֶשׁ / PERO NO DEBEN VENIR A VER CUANDO LO SANTO ES ENVUELTO.     Mishná

    Tratado Ohalot, capítulo 8

    (1) Hay cosas que transmiten la impureza, pero impiden que se propague, otras que la transmiten y no impiden que se propague, otras que impiden que se propague pero no la transmiten, y otras que ni la transmiten ni impiden que se propague. Estas son las cosas que la transmiten pero impiden que se propague: Los baúles, las arcas, los armarios, las colmenas de paja, las colmenas de caña y las bodegas de los barcos alejandrinos, por cuanto se apoyan en bases y pueden contener cuarenta Seá de líquidos, o dos Kor de sólidos. Las pieles, los mandiles de cuero, las mantas, las sábanas, las alfombras y las esteras, si se fabrican con ellos tiendas. Asimismo, los rebaños de animales impuros o puros, las jaulas de los animales salvajes y de las aves, las aves reposando, y el lugar que vacía una madre para colocar a su hijo entre las espigas. Además, el iris, la hiedra, el cardo borriquero, la calabaza griega, y los alimentos puros. Rabí Yojanán Ben-Nurí no está de acuerdo en cuanto a los alimentos puros, a excepción de las tortas de higo.  (2) (También) Los salientes, los balcones, los palomares, los picos de las rocas, las rocas, las ventanas de las rocas, los dientes de las rocas, las ramas cubridoras y las ramas salientes, que pueden resistir un revoque suave, según Rabí Meir. Los sabios dicen: Un revoque mediano. ¿Qué son las “Ramas cubridoras”? Aquellas que proceden de un árbol que se extiende por encima de la tierra. ¿Y las “Ramas salientes”? Aquellas que sobresalen del cerco.  (3) Las siguientes cosas transmiten (la impureza) pero no impiden que se propague: Los baúles, las arcas, los armarios, las colmenas de paja, las colmenas de caña y las bodegas de los barcos alejandrinos, que no se apoyan en bases y no pueden contener cuarenta Seá de líquidos, o dos Kor de sólidos. Las pieles, los mandiles de cuero, las mantas, las sábanas, las alfombras, las esteras, si no se fabrican con ellos tiendas. También, los animales domésticos y salvajes muertos, y los alimentos impuros. A estos hay que agregarle los molinos humanos.  (4) Las siguientes cosas impiden que se propague (la impureza) y no la transmiten: El estambre extendido, las cuerdas de la cama, los recipientes de la basura, y los enrejados de las ventanas.  (5) Las siguientes cosas ni la transmiten ni impiden que se propague: Las plantas y las verduras metidas en tierra, a excepción de las especies ya mencionadas. Las placas de granizo, la nieve, la escarcha, el hielo, la sal, la persona que salta de un lugar a otro, la que brinca, las aves volando, un manto agitado por el viento, y el barco que navega por encima del agua. Si se ata el barco a un lugar que lo mantiene parado, o se coloca una piedra encima del manto, sí transmite la impureza. Rabí Yosé dice: la cabina que se encuentra en un barco no transmite la impureza.  (6) Dos barriles, que contienen dos medios Kazáyit (de impureza), si están sellados y se encuentran en el interior de una casa, se consideran puros, pero la casa se impurifica. Si se abre uno de ellos, tanto él como la casa se impurifican, pero el otro permanece puro. Igualmente ocurre si se trata de dos cuartos que se hallan abiertos hacia el interior de la casa.

    Rabí Ovadiá de Bartenura

    1. Hay COSAS… Estas cosas, si se colocan por encima de una impureza, y a la vez, cubren personas u otros objetos, transmiten dicha impureza a dichas personas y objetos, aunque no los toquen.  IMPIDEN QUE SE PROPAGUE. Si la impureza se encuentra debajo de ellas, impiden que las personas o cosas que estén por encima se contagien, ya que ellas las aíslan ante la impureza. Igualmente, si la impureza se encuentra por encima de ellas por el exterior, aíslan a las personas o cosas que se hallen por debajo.  BAÚLES. Cajas grandes de madera similares a un armario.  ARCAS. Cajas más pequeñas, del tamaño de una mesa.  COFRES. De madera que se usa para guardar las cosas de valor.  COLMENAS DE PAJA. Cajas grandes fabricadas con paja, hojas de sauce o ramas de palmera.  COLMENAS DE CAÑA. Especie de cesta fabricada con cañas.  LAS BODEGAS… Los grandes barcos que navegan por los grandes mares contienen un gran recipiente de madera, como una especie de aljibe, para llevar en él agua potable para beber.  CUARENTA SEÁ. Siendo que alcanzan una gran medida no son susceptibles de impurificarse, y además, impiden que se propague la impureza.  MANDILES. Delantal de piel que visten los curtidores de pieles mientras trabajan.  MANTAS. Pieles que se extienden sobre las camas para acostarse sobre ellas. En árabe se llaman “Natá”.  SI SE FABRICAN TIENDAS. Si se extienden en forma de tienda. A pesar de que tanto las pieles, como los mandiles de cuero, las mantas, las sábanas, las alfombras y las esteras son objetos impurificables, cuando se extienden para formar una tienda impiden que la impureza se propague. Las tiendas, aunque se pueden impurificar, sirven de aislante ante la impureza.  REBAÑOS. Cuando están parados y no caminan, y la cabeza de uno se encuentra entre las patas del otro.  LAS JAULAS. Los habitáculos de los animales salvajes y de las aves. Esta palabra, en hebreo, concuerda con la expresión: “La morada donde habitas” (Éxodo 15:17).  REPOSANDO. Atadas a una pared o a un árbol.  ENTRE LAS ESPIGAS. Cubriéndolo con las espigas como si fuera una casa. EL IRIS… Todas estas plantas poseen hojas muy anchas.  TORTAS DE HIGO. Porque solían sentarse debajo de ellas por su sombra, más que debajo de cualesquiera otros alimentos, tal como se aprende de la Toseftá. La Halajá no coincide con Rabí Yojanán Ben Nurí.  2. SALIENTES. La madera o piedra solitarias que sobresalen de la pared se denomina “Saliente”.  BALCONES. Como miradores. Construcciones de piedra o madera que sobresale del muro.  PALOMARES. Lugar donde viven las palomas.  LOS PICOS. Así se traducen al arameo las palabras: “Salientes de las rocas” (Yeshayahu 57:5), “Picos”.  ROCAS. Se refiere a los huecos que hay en las rocas. Los picos y las rocas se refieren a cosas similares, sólo que una se forma por la erosión de la lluvia y otro se forma por encima.  LAS VENTANAS. Especie de agujero por donde penetra la luz. Hay quien interpreta que se trata de cuevas y galerías.  DIENTES. Los salientes puntiagudos de la roca.  REVOQUE SUAVE. Como cuando se fabrican con sauce o cañas y se recubren con poco barro.  MEDIANO. Pero si sólo resisten un revoque ligero, ni transmiten la impureza ni la aíslan. La Ley concuerda con los sabios.  ÁRBOL QUE SE EXTIENDE. Se utiliza el mismo vocablo hebreo que en: “Cubren con sus alas” (Éxodo 25:20).  SOBRESALEN DEL CERCO. Se acostumbra colocar pinchos y púas donde empieza el cerco y que sobresalen hacia el exterior. La palabra “Saliente”, procede de la expresión: “Sobresale su pelo” (Números 6:5).  3. NO PUEDEN CONTENER. Siendo que son susceptibles de impurificarse, no sirven como barrera para que no se propague la impureza.  NO SE FABRICAN… No están dispuestos en forma de cabaña, sino que están extendidos sobre una claraboya, por ejemplo.  MOLINOS HUMANOS. Es decir, los molinos manuales, que al ser transportables son susceptibles de impurificarse.  4. ESTAMBRE EXTENDIDO. Se refiere a los hilos de la urdimbre que se tensan en el telar antes de tejer las prendas. Si estuviesen extendidos sobre una claraboya que se encuentra entre la planta baja y la alta de una casa, encontrándose un foco de impureza abajo, impide que se impurifiquen los objetos de arriba. No obstante, si estuviera cubriendo un cadáver o un objeto impuro, no transmite la impureza pues no se llama “Techo” hasta que no esté el tejido hecho.  LAS CUERDAS. Se refiere a las cuerdas que están atadas en la parte baja de la cama formando un enrejado.  RECIPIENTES. Las cajas en las cuales se saca la basura de las casas.  ENREJADOS. Dichas rejas detienen el paso de la impureza, impidiendo que pase al otro lado, siempre que no tengan un hueco del tamaño de un puño al cuadrado, pues si lo tuviera, sí pasaría a través de él la impureza.  5. YA MENCIONADAS. Aquellas que menciona la Mishná al principio del capítulo: El iris, la hiedra, el cardo borriquero y la calabaza griega, pero con la condición que estén adheridas a la tierra, porque si están cortadas: Si estaban en disposición de impurificarse no impiden que se propague la impureza, pues las cosas que son susceptibles de impurificarse no sirven como barrera. Si no fueron dispuestas para ello, se consideran como los alimentos puros.  PLACA DE GRANIZO. O sea, un bloque de granizo, como se dice en los Salmos (147:17): “Arroja la nieve en trozos”. También se puede explicar que el granizo forma una especie de manto. En francés antiguo “Alcawalet”.  ESCARCHA. Se refiere al agua que fluye y que al caer se congela.  HIELO. Las aguas de los ríos congeladas.  SAL. Todas estas cosas ni se impurifican, ni impiden que se transmita la impureza, ya que se deshacen y no duran.  QUE SALTA. Se refiere a los animales que saltan. En los hombres se dice “Brincar”. Hay quien opina que tanto “Saltar”, como “Brincar” se refieren al hombre, sólo que saltar se refiere cuando adelanta un pie en el salto mientras que el otro aún está en el suelo, y brincar, es hacerlo con ambos pies a la vez.  UN MANTO. Al soplar el viento lo levanta como si fuera un palio.  COLOCA UNA PIEDRA. Se refiere cuando coloca una piedra en un extremo e la prenda, y el viento levanta la otra punta como si fuera una cabaña.  TRANSMITE. Si se colocara como techo por encima de un cadáver o un objeto impurificado, transmitiría la impureza a otros objetos.  LA CABINA. Por ejemplo, cuando se encuentra en lo alto del barco tapando los objetos que se encuentran en él, o sobre un cadáver que se halla en el mar mientras navega. Otra explicación: Si la cabina que está en el barco no aguanta un fuerte viento, incluso si cubriera un cadáver o un objeto impurificado no transmitiría la impureza, ya que no se conceptúa como una tienda. La Halajá no coincide con Rabí Yosé. 6. DOS MEDIOS KAZÁYIT. La mitad de un Kazáyit en uno, y la mitad en el otro.  SE IMPURIFICA. A pesar que están sellados no impiden que la impureza se propague y la casa se impurifica. Aunque se encuentren en dos barriles diferentes, siendo que se hallan en la misma casa se suman para completar la cantidad requerida para que una casa se impurifique. No obstante, los barriles siguen siendo puros, ya que la impureza contenida en el medio Kazáyit del un barril no penetra en el otro, siendo que están sellados.  DOS CUARTOS. Aunque las dos puertas estén cerradas y se halle un medio Kazáyit en uno, y otro medio en el otro. La casa se impurifica porque la impureza, finalmente, saldrá de ella. En cambio, los cuartos siguen estando puros, pues la impureza no suele salir de allí.

    (Comentario sobre el Zóhar)

    En los sábados, novilunios, en las conmemoraciones y en los días festivos el fuego se calma en aquel lugar, y no se castiga. No obstante, no salen de allí, como el resto de los malvados que tienen descanso. Todos aquellos que profanan los sábados y las festividades, y que no se preocupan en absoluto por el Honor de su Señor, respetándolos, sino que los profanan en público, al igual que ellos no guardan los sábados y las festividades en este mundo, tampoco los cuidan en el otro mundo y no tienen descanso. Dijo Rabí Yosé: ¡No es así!, pues cuidan los sábados y las festividades en el Guehinam a la fuerza. Dijo Ribí Yehudá: Eso se refiere a los gentiles, que aunque no están obligados a cuidar el Shabat en este mundo, los obligan a cuidarlo allí. Cuando se india el Shabat, cuando el día se consagra, van pasando anuncios en todas las dependencias del Guehinam que dicen: ¡Dejen de castigar a los malvados, pues el Rey Santo se acerca y el día se ha consagrado, y Él protege a todos. Enseguida, los castigos se detienen y los malvados descansan (, y también todos aquellos que solían profanar el Shabat y los Días festivos en público dejan de ser castigados) (Otra versión: e incluso los gentiles que no fueron ordenados de cuidar el Shabat y las festividades, allí los guardan a la fuerza). Sin embargo, aquellos que no cuidan el Shabat, nunca el fuego del Guehinam se aparta de ellos. Entonces, todos los malvados que se encuentran en el Guehinam preguntan: ¿Por qué a ellos se les trata de forma diferente al resto de los malvados? ¿Por qué no tienen descanso? Los castigadores les responden: Aquellos son los impíos que se rebelaron contra el Todopoderoso, y transgredieron toda la Torá, puesto que no respetaron el Shabat allí. Por esto no tendrán descanso eternamente.

    1. Quien tritura en una medida de un higo seco, se condena. Asimismo, quien tritura condimentos o especias con un mortero, transgrede esta prohibición y se condena. Quien mezcla tierra con agua, se condena por ser un derivado de amasar. ¿Cuál es la medida que condena a quien lo realiza? Lo necesario para hacer la apertura del crisol que usan los orfebres. La mezcla no se considera si se realiza con ceniza, arena, salvado, o cosas similares.     Recopilación de los libros de ética

    “Bendiga mi alma al Eterno, y mis entrañas, Su Santo Nombre” (Salmos 103:1). Esto nos enseña que incluso las entrañas y los intestinos del ser humano deben bendecir al Todopoderoso. Por esta razón, se deben revisar muy bien antes de rezar. No se puede entrar con el vientre lleno de excrementos cuando se pretende bendecir al Santo, Bendito Él.
