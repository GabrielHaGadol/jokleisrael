    רש"י

    ב(ב) וידבר אלהים אל משה. דבר אתו משפט על שהקשה לדבר ולומר (לעיל ה כב) למה הרעותה לעם הזה: ויאמר אליו אני ה'. נאמן לשלם שכר טוב למתהלכים לפני. ולא לחנם שלחתיך כי אם לקים דברי שדברתי לאבות הראשונים. ובלשון הזה מצינו שהוא נדרש בכמה מקומות אני ה' נאמן לפרע, כשהוא אמור אצל עונש, כגון (ויקרא יט יב) וחללת את שם אלהיך אני ה', וכשהוא אמור אצל קיום מצות, כגון (ויקרא כב לא) ושמרתם מצותי ועשיתם אותם אני ה', נאמן לתן שכר: (ג) וארא. אל האבות: באל שדי. הבטחתים הבטחות ובכלן אמרתי להם אני אל שדי: ושמי ה' לא נודעתי להם. לא הודעתי אין כתיב כאן אלא לא נודעתי לא נכרתי להם במדת אמתות שלי שעליה נקרא שמי ה' נאמן לאמת דברי, שהרי הבטחתים ולא קימתי: וגם הקמתי את בריתי וגו' . (ד) וגם כשנראיתי להם באל שדי הצבתי והעמדתי בריתי ביני וביניהם: לתת להם את ארץ כנען. לאברהם בפרשת מילה נאמר (בראשית יז א) אני אל שדי וגו' , (בראשית יז ח) ונתתי לך ולזרעך אחריך את ארץ מגוריך, ליצחק (בראשית כו ג) כי לך ולזרעך אתן את כל הארצות האל והקימותי את השבועה אשר נשבעתי לאברהם, ואותה שבועה שנשבעתי לאברהם באל שדי, אמרתי ליעקב (בראשית לה יא. יב) אני אל שדי פרה ורבה וגו' ואת הארץ אשר וגו', הרי שנדרתי להם ולא קימתי: (ה) וגם אני. כמו שהצבתי והעמדתי הברית יש עלי לקים, לפיכך שמעתי את נאקת בני ישראל הנואקים: אשר מצרים מעבדים אתם ואזכר. אותו הברית. כי בברית בין הבתרים אמרתי לו (בראשית טו יד) וגם את הגוי אשר יעבדו דן אנכי: (ו) לכן. על פי אותה השבועה: אמר לבני ישראל אני ה'. הנאמן בהבטחתי: והוצאתי אתכם. כי כן הבטחתיו (שם שם) ואחרי כן יצאו ברכוש גדול: סבלת מצרים. טורח משא מצרים:                                                                                                                                                                                                                                                                                                           ן

    כ מִרְמָה בְּלֶב חֹרְשֵׁי רָע וּלְיֹעֲצֵי שָׁלוֹם שִׂמְחָה: רַמָּיוּתָא בְּלִבְּהוֹן דְּחַשְׁלִין בִּשְׁתָּא וּלְאִלֵין דְּיָעֲצִין שְׁלָמָא חֶדְוְתָא תֶהֱוֵי: כא לֹא יְאֻנֶּה לַצַּדִּיק כָּל אָוֶן וּרְשָׁעִים מָלְאוּ רָע: לָא שַׁפִּיר לְצַדִּיקָא כָּל מִדָּעַם דְּעָאתָא וְרַשִׁיעֵי מַלְיָן בִּישְׁתָּא: כב תּוֹעֲבַת יְהוָה שִׂפְתֵי שָׁקֶר וְעֹשֵׂי אֱמוּנָה רְצוֹנוֹ: מְרַחֲקָתֵיהּ דֶּאֱלָהָא סִפְוָתֵיהּ דְשִׁקְרָא וּבְאִלֵין דְּעָבְדִין הֵימָנוּתָא מִתְרְעֵי: כג אָדָם עָרוּם כֹּסֶה דָּעַת וְלֵב כְּסִילִים יִקְרָא אִוֶּלֶת: בַּר נָשׁ עֲרִימָא מְכַסֶּה יְדִיעֲתָא וְלִבְּהוֹן דְּסַכְלֵי סָנֵי יְדִיעֲתָא: כד יַד חָרוּצִים תִּמְשׁוֹל וּרְמִיָּה תִּהְיֶה לָמַס: אַיְדָא דְתַקִּיפֵי תִשְׁלַט וְרַמָּאֵי יְהוֹן לְמִיסָא: כה דְּאָגָה בְלֶב אִישׁ יַשְׁחֶנָּה וְדָבָר טוֹב יְשַׂמְּחֶנָּה: מִלְּתָא דְחִילְתָּא בְּלִבֵּיהּ דְּגַבְרָא מְדַחֲלָא וּמִלַיָא טָבַיָא מְחַדְיָן לֵיהּ:                                                                                                                                                                                                                                                                                                           ן    

    רש"י

    כ(כ) מרמה בלב חרשי רע. ומתוך שה' טרודים במחשבות של תרמית אין להם שמחה וליועצי שלום שמחה: (כא) לא יאונה לצדיק. לא יארע לא תזדמן לו עבירה בלי דעת: (כג) אדם ערום כוסה דעת. אפילו בחכמתו הוא צנוע וכל שכן לדברי שטות הוא מכסה. אבל ולב כסילים יקרא אולת. מכריז אולת בקול רם: (כד) יד חרוצים. ישרים: תמשול. תעשיר: (כה) דאגה בלב איש ישחנה. ישיחנה מדעתו: ודבר טוב ישמחנה. יעסוק בתורה והיא תשמח את הדאגה שבלבו ותצילנו ממנה , ולדברי האומר ישיחנה לאחרים כך סופו של מקרא ודבר טוב שינחמנו חבירו ישמח את הדאגה:                                                                                                                                                                                                                                                                                                           ן

    א. גָּדִישׁ שֶׁלֹּא לֻקַּט תַּחְתָּיו, כָּל הַנּוֹגֵעַ בָּאָרֶץ הֲרֵי הוּא שֶׁל עֲנִיִּים. הָרוּחַ שֶׁפִּזְּרָה אֶת הָעֳמָרִים, אוֹמְדִים אוֹתָהּ כַּמָּה לֶקֶט הִיא רְאוּיָה לַעֲשׂוֹת, וְנוֹתֵן לָעֲנִיִּים. רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר, נוֹתֵן לָעֲנִיִּים בִּכְדֵי נְפִילָה: ב. שִׁבֹּלֶת שֶׁבַּקָּצִיר וְרֹאשָׁהּ מַגִּיעַ לַקָּמָה, אִם נִקְצֶרֶת עִם הַקָּמָה, הֲרֵי הִיא שֶׁל בַּעַל הַבַּיִת, וְאִם לָאו, הֲרֵי הִיא שֶׁל עֲנִיִּים. שִׁבֹּלֶת שֶׁל לֶקֶט שֶׁנִּתְעָרְבָה בַגָּדִישׁ, מְעַשֵּׂר שִׁבֹּלֶת אַחַת וְנוֹתֵן לוֹ. אָמַר רַבִּי אֱלִיעֶזֶר, וְכִי הֵיאָךְ הֶעָנִי הַזֶּה מַחֲלִיף דָּבָר שֶׁלֹּא בָא בִרְשׁוּתוֹ. אֶלָּא מְזַכֶּה אֶת הֶעָנִי בְּכָל הַגָּדִישׁ, וּמְעַשֵּׂר שִׁבֹּלֶת אַחַת וְנוֹתֵן לוֹ: ג. אֵין מְגַלְגְּלִין בְּטוֹפֵחַ, דִּבְרֵי רַבִּי מֵאִיר. וַחֲכָמִים מַתִּירִין, מִפְּנֵי שֶׁאֶפְשָׁר: ד. בַּעַל הַבַּיִת שֶׁהָיָה עוֹבֵר מִמָּקוֹם לְמָקוֹם, וְצָרִיךְ לִטֹּל לֶקֶט שִׁכְחָה וּפֵאָה וּמַעְשַׂר עָנִי, יִטֹּל, וּכְשֶׁיַּחֲזֹר לְבֵיתוֹ יְשַׁלֵּם, דִּבְרֵי רַבִּי אֱלִיעֶזֶר. וַחֲכָמִים אוֹמְרִים, עָנִי הָיָה בְּאוֹתָהּ שָׁעָה: ה. הַמַּחֲלִיף עִם הָעֲנִיִּים, בְּשֶׁלּוֹ פָּטוּר, וּבְשֶׁל עֲנִיִּים חַיָּב. שְׁנַיִם שֶׁקִּבְּלוּ אֶת הַשָּׂדֶה בַּאֲרִיסוּת, זֶה נוֹתֵן לָזֶה חֶלְקוֹ מַעְשַׂר עָנִי, וְזֶה נוֹתֵן לָזֶה חֶלְקוֹ מַעְשַׂר עָנִי. הַמְקַבֵּל שָׂדֶה לִקְצֹר, אָסוּר בְּלֶקֶט שִׁכְחָה וּפֵאָה וּמַעְשַׂר עָנִי. אָמַר רַבִּי יְהוּדָה, אֵימָתַי. בִּזְמַן שֶׁקִּבֵּל מִמֶּנּוּ לְמֶחֱצָה, לִשְׁלִישׁ וְלִרְבִיעַ. אֲבָל אִם אָמַר לוֹ שְׁלִישׁ מַה שֶּׁאַתָּה קוֹצֵר שֶׁלָּךְ, מֻתָּר בְּלֶקֶט וּבְשִׁכְחָה וּבְפֵאָה, וְאָסוּר בְּמַעְשַׂר עָנִי: ו. הַמּוֹכֵר אֶת שָׂדֵהוּ, הַמּוֹכֵר מֻתָּר וְהַלּוֹקֵחַ אָסוּר. לֹא יִשְׂכֹּר אָדָם אֶת הַפּוֹעֲלִים עַל מְנָת שֶׁיְּלַקֵּט בְּנוֹ אַחֲרָיו. מִי שֶׁאֵינוֹ מַנִּיחַ אֶת הָעֲנִיִּים לִלְקֹט, אוֹ שֶׁהוּא מַנִּיחַ אֶת אֶחָד וְאֶחָד לֹא, אוֹ שֶׁהוּא מְסַיֵּעַ אֶת אֶחָד מֵהֶן, הֲרֵי זֶה גּוֹזֵל אֶת הָעֲנִיִּים. עַל זֶה נֶאֱמַר (משלי כב) אַל תַּסֵג גְּבוּל עוֹלִים: ז. הָעֹמֶר שֶׁשְּׁכָחוּהוּ פוֹעֲלִים וְלֹא שְׁכָחוֹ בַעַל הַבַּיִת, שְׁכָחוֹ בַעַל הַבַּיִת וְלֹא שְׁכָחוּהוּ פוֹעֲלִים, עָמְדוּ עֲנִיִּים בְּפָנָיו אוֹ שֶׁחִפּוּהוּ בְקַשׁ, הֲרֵי זֶה אֵינוֹ שִׁכְחָה: ח. הַמְעַמֵּר לְכוֹבָעוֹת וּלְכֻמְסָאוֹת, לַחֲרָרָה וְלָעֳמָרִים, אֵין לוֹ שִׁכְחָה. מִמֶּנּוּ וְלַגֹּרֶן, יֶשׁ לוֹ שִׁכְחָה. הַמְעַמֵּר לַגָּדִישׁ, יֶשׁ לוֹ שִׁכְחָה. מִמֶּנּוּ וְלַגֹּרֶן, אֵין לוֹ שִׁכְחָה. זֶה הַכְּלָל, כָּל הַמְעַמֵּר לְמָקוֹם שֶׁהוּא גְמַר מְלָאכָה, יֶשׁ לוֹ שִׁכְחָה. מִמֶּנּוּ וְלַגֹּרֶן, אֵין לוֹ שִׁכְחָה. לְמָקוֹם שֶׁאֵינוֹ גְמַר מְלָאכָה, אֵין לוֹ שִׁכְחָה. מִמֶּנּוּ וְלַגֹּרֶן, יֶשׁ לוֹ     ר' עובדיה מברטנורה

    א(א) גדיש שלא לוקט תחתיו. מי שאסף והגדיש את העומרים למקום אחד, בתוך השדה שעדיין לא לקטו שם העניים את הלקט: כל הנוגעות בארץ לעניים. דקנסוהו רבנן לפי שהגדיש על גב הלקט, אפילו הגדיש חטים על גבי לקט של שעורים, הוי כל הנוגע מן החטים בארץ לעניים: הרוח שפזרה את העמרים. על גבי הלקט: כדי נפילה. כדי שיעור שרגיל ליפול בשעת קצירה שהם ארבעה קבין לכור זרע והכור הוא שלשים סאין ומשום דאמר תנא קמא אומדים כמה לקט ראויה לעשות, קא משמע לן רבן שמעון בן גמליאל אין הדבר צריך אומד שכבר קצוב ועומד כדי נפילה, בשלשים סאין של זרע רגיל ליפול ממנו בשעת קצירה ארבעה קבין וכך יתן לעניים:  (ב) שבולת שבקציר. שייר בקצירתו שבולת אחת שלא קצר וראש אותה שבולת מגיע לקמה, אם אותה שבולת נקצרת עם הקמה הרי היא של בעל הבית. שהקמה מצילתה דלא קרינן בה לא תשוב לקחתו (דברים כד): מעשר שבולת אחת ונותן לו. לפי שחייב ליתן לו דבר מתוקן כמו שהיתה שבולת של לקט דפטורה מן המעשרות, ואף על פי שעדיין לא נתמרח, חל שם מעשר עליו כמו מעשר ראשון שהקדימו בשבלין שחל עליו שם מעשר אף על פי שעדיין לא נתמרח. וכיצד הוא עושה, מביא שתי שבלים מגדיש זה שנתערבה בו שבולת של לקט, ואומר על אחת מהן, אם זו לקט מוטב, ואם לאו, הרי מעשרותיה קבועים בזו השנית, ונותן לו את הראשונה. ובירושלמי פריך, וניחוש שמא זאת השניה שקבע עליה המעשרות של לקט היא, וכיון שהיא פטורה מן המעשר, אין המעשרות קובעים בה, שאין מפרישים מן הפטור על החיוב, ומשני שלוקח שבולת שלישית ואומר, אם השניה שקבעתי בה המעשרות היא לקט ואינה יכולה להיות מעשר, הרי זו השלישית מעשר על הראשונה: אמר רבי אליעזר וכי היאך העני הזה מחליף דבר שלא בא ברשותו. רבי אליעזר לדבריהם דרבנן קאמר להו, לדידכו דפלגיתו עלי בעשיר שלקט פאה לעני שאני אומר זכה לו ואתון אמריתו דלא זכה לו, היאך העני הזה מחליף דבר שלא בא ברשותו, דהא אין בעל השדה יכול לזכות לו את הלקט לפי דבריכם, ונמצא דלקט זה לא בא ברשותו של עני, אלא לדבריכם מזכה את העני בכל הגדיש על מנת להחזיר, דמתנה על מנת להחזיר שמה מתנה, ונמצאת אותה שבולת שהיא לקט באה ליד העני ויכול להחליפה באחרת. ורבנן אמרי הכא עשו את שאינו זוכה כזוכה, וחשבינן להאי שבולת של לקט כאלו באה לידו של עני, אף על גב דלא זכה בה, לענין זה שיכול להחליפה באחרת. והלכה כחכמים:  (ג) אין מגלגלין בטופח. מין קטנית דק שקורין לו בערבי גילבא''ן. ורמב''ם פירש שהוא מין זרע שקורין לו קורטמ''ן ודומה לשעורים, ואומר רבי מאיר שאין מגלגלין זה המין עם שאר מיני זרעים לקצור אותם ביחד, כי הלקט שהיה ראוי ליפול משאר מיני זרעים, יפול מזה המין הגרוע, ונמצא מפסיד לעניים: וחכמים מתירים מפני שאפשר. שיפול לקט משאר זרעים כמו מן הטופח. והלכה כחכמים. פירוש אחר, אין מעלין מים בגלגל מן הבור להשקות את השדה או להרביצה עד שיהא משקה טופח עליה, בשעה שהעניים באים בשביל הלקט, מפני שהוא מפסיד לעניים, וחכמים מתירין מפני שאפשר שישומו ההפסד של עניים בזה ויתן בעל השדה כפי השומא שישומו עליו. ולרבי מאיר דאמר אין מגלגלין, שמין לבעל הבית הפסדו ונוטל מן העניים:  (ד) עני היה באותה שעה. ופטור מלשלם. והלכה כחכמים: (ה) המחליף עם העניים. נותן תבואה או פירות לעני בחלוף מה שלקט: שלו פטור. מן המעשר מה שנתן לו העני שהוא Al declarar a Moshé Su Nombre Inefable [‘ה], Di-s le quiso decir: Yo soy Fiel para otorgar buena recompensa a los que marchan delante de Mí. Y no de balde te he enviado, sino para hacer que se cumplan Mis palabras que dije a los primeros patriarcas.  Y hallamos que el Nombre Inefable es interpretado en este mismo sentido en varios lugares de la Torá, cuando dice “Yo soy el Eterno [‘ה]”, lo que implica: Yo soy Fiel para retribuir. Cuando está enunciado en el contexto de un castigo, como por ejemplo en el versículo: “Y habrás profanado el Nombre de tu Di-s; Yo soy el Eterno” (Levítico 19:12), ahí implica que es Fiel para castigar.  Y cuando está enunciado en el contexto del cumplimiento de los mandamientos, como por ejemplo en el versículo: “Guardarán Mis mandamientos y los harán; Yo soy el Eterno” (ibíd. 22:31), en tal caso significa: Yo soy Fiel para otorgar recompensa.  3. וָאֵרָא / YO ME REVELÉ.     Tratado Peá, capítulo 5

    (1) Si se forma una montaña de gavillas, en un lugar donde aún no se recolectó el Leket, todo lo que toque la tierra, será para los pobres. No obstante, si el viento esparce las gavillas sobre una parcela de la que no fue aún extraído el Leket, entonces, se calcula qué cantidad de Leket pertenecería ser retirado de dicha parcela, y se la entrega a los pobres. Rabán Shimón Ben-Gamliel dice: Dará solamente la cantidad de espigas que suele caerse normalmente en el momento de la cosecha. (2) Una espiga que no fue cortada cuando segaba, y que al inclinarla, su extremo superior llega a tocar el resto de las espigas de la mies: Si puede ser cortada junto al resto de la mies que aún no segó, entonces, le pertenece al dueño. En caso contrario, hay que dejarla para los pobres. Si una espiga que pertenece al Leket se mezcla con las espigas de una parva, toma una espiga que ya fue diezmada y se la entrega al pobre. Dijo Ribí Eliézer: ¿Cómo puede el pobre intercambiar una espiga que aún no le pertenece? Lo que debe hacer es, ceder al pobre la parva completa, y después, diezmar una espiga para entregársela. (3) No se debe cosechar a la vez trigo y Tófaj, según Ribí Meir. Los Sabios sí lo permiten, porque no hay perjuicio para el pobre. (4) Si una persona que no era pobre viajaba de un lugar a otro, y tuvo necesidad de agarrar parte del Leket, Shijjá, Peá o del diezmo de los pobres, lo puede tomar y cuando regrese a su casa que lo pague. Esta es la opinión de Rabí Eliézer. El resto de los sabios dijo: En ese momento, era realmente pobre. (5) Quien intercambia con los pobres de lo suyo, queda exento, de los de los pobres, queda obligado. Dos, recibieron un campo en calidad de aparceros: Uno puede dar al otro el diezmo de los pobres, y el otro puede dar al primero el diezmo de los pobres. Quien recibe un campo para segarlo, tiene prohibido tomar para sí el Leket, la Shijjá, la Peá y el Maaser Aní. Dijo Rabí Yehudá: ¿En qué casos? Sólo cuando recibe la mitad, un tercio o una cuarta parte. No obstante, si le dijo: De lo que recolectes, la tercera parte es tuya, entonces, tiene permiso para agarrar Leket, Shijjá y Peá, pero no lo tiene para el Maaser Aní. (6) Quien vende su campo, se le permite tomar parte de este, pero no, a quien lo adquirió. No se debe contratar obreros con la condición de que sus hijos recolecten las donaciones tras ellos. Quien no permite que los pobres recolecten las donaciones, o si deja a unos y a otros no, o ayuda a uno de ellos, es considerado como si robara a los pobres. Sobre esto se dijo: “No retrocedas el linde de siempre” (Proverbios 22:28). (7) Una gavilla olvidada por los obreros, pero que no fue olvidada por el dueño, o que fue olvidada por el dueño, pero no por los obreros, o delante de la cual se colocaron los pobres o la cubrieron con heno, no se considera olvidada o “Shijjá”. (8) Si se amontonan espigas en forma de sombrero (montón de gran altura), o en forma de Jumsá (Montón de menor altura), o en forma de círculo, o en forma de pequeñas gavillas, durante la formación de todos ellos no existe Shijjá. Desde allí hasta el granero, hay Shijjá. Mientras las recopila para formar un gran montículo, hay Shijjá. Desde allí hasta el granero, no hay Shijjá. Esta es la norma: Mientras las amontona para llevarlas hacia donde se termina su manipulación, existe la Shijjá. Desde allí hasta el granero, no se da la Shijjá. Desde allí hasta donde el trabajo no se acaba, tampoco existe Shijjá, pero desde allí hasta el granero, sí se aplica la Shijjá.

    1. SI SE FORMA UNA MONTAÑA. Si alguien recopila y amontona gavillas en un mismo lugar, sobre un campo donde aún los pobres no recolectaron el Leket. LO QUE TOQUE LA TIERRA. Los sabios quisieron castigar así a quien amontonaba las gavillas encima del Leket. Incluso cuando amontonó gavillas de trigo sobre el Leket de cebada, todo el trigo que tocaba tierra se les daba a los pobres. SI EL VIENTO ESPARCE LAS GAVILLAS. Sobre el Leket. QUE SUELE CAERSE. Esto es, la cantidad de espigas que caen en el momento de la siega, que se calcula, en cuatro Kabín por cada Kur, cuando un Kur equivale a cuarenta Seín. El primer exponente de la Mishná habló de “la cantidad de Leket que pertenecería ser retirado de dicha parcela”, precisamente, mientras que, Rabán Shimón Ben-Gamliel, difiere de él en que la cantidad está fijada de antemano para el conjunto de todos los campos, y no depende de aquel en concreto. Por tanto, no cabe hacer ningún cálculo. UNA ESPIGA QUE NO FUE CORTADA. Cuando segó, dejó una espiga cuya cabeza alcanza a la mies; si dicha espiga puede ser cortada con el resto, entonces le pertenece al dueño. En este caso, no cabe el concepto de “No volverás para recogerla” (Deuteronomio 24:19). TOMA UNA ESPIGA... DIEZMADA. Debe entregarle algo exento del gravamen del diezmo, tal como lo está la espiga que forma parte del Leket. Aunque el grano aún no fue aplastado, recae sobre él el concepto de “Diezmo”, al igual que ocurre con el “Maaser Rishón”, o “Primer Diezmo”, que, aunque aún no haya sido aplastado, recae sobre él el concepto de “Diezmo”. ¿Cómo lo realiza? Trae dos espigas del montón donde se mezcló la espiga perteneciente el Leket. Entonces dice sobre una de ellas: “Si esta pertenece al Leket, bien, y si no, el diezmo queda hecho de esta segunda, y le da la primera. En el Talmud Yerushalmi se pregunta: ¿Qué ocurre si, precisamente, esta segunda espiga era la que pertenecía al Leket, donde, de ser así estaría exenta de separar el diezmo? Se contesta: Toma una tercera espiga y repite la misma fórmula sobre ella. DIJO RIBÍ ELIÉZER... Ribí Eliézer replica a los Sabios: Ustedes, que me contradijeron con respecto al rico que recolectó la “Peá” para darla a un pobre, donde yo dije que la cesión era válida y ustedes afirmaron que no, ¿cómo puede ser que en este caso, un pobre puede intercambiar algo, que aún no llegó a sus manos? Según ustedes, el dueño del campo no puede otorgarle la propiedad del Leket, o sea, no está en sus manos. Por lo tanto: según ustedes, el pobre debe recibir todo el campo con la intención de devolverlo, ya que lo que se regala con esta condición, legalmente se conceptúa como regalo. Entonces, una vez que está en su posesión, la puede cambiar por otra. Los sabios contestaron: En este caso especial, se le otorgó al pobre, lo que no le correspondería en otra situación. Consideramos esta espiga del Leket como si la tuviera en sus manos el pobre, aunque, en realidad no le pertenece, sólo para que la pueda intercambiar con otra. La Ley, se rige según la opinión de los Sabios. 3. TRIGO Y “TÓFAJ”. Este es una legumbre delgada conocida en árabe como “Guilbán”. Para Maimónides, es un cereal llamado “Cortemán”, y que se parece a la cebada. Según Ribí Meir, no se debe mezclar esa especie con otras a la hora de la recolección, ya que es posible que el Leket que debería caer de las demás especies, caiga de esta otra, donde los pobres se verían perjudicados. LOS SABIOS SÍ LO PERMITEN. Porque puede quedar Leket, tanto de las demás especies como del Tófaj. La Halajá se rige de acuerdo con el dictamen de los Sabios. Según otra opinión, no se debe extraer agua del pozo para regar (Tofeaj) el campo copiosamente, a la hora que los pobres vienen a retirar el Leket, porque, de esta forma, perjudica a los pobres. Los Sabios lo permiten, porque creen que el perjuicio puede ser cuantificado y, consecuentemente, puede ser compensado con posterioridad, a los recolectores del Leket. Para Ribí Meir que dijo que no se deben mezclar las diferentes especies, se calcula la pérdida sufrida por el dueño, y se les descuenta a los pobres. 4. ERA REALMENTE POBRE. O sea, lo que recibió del pobre, el Leket, la Shijjá o la Peá, queda exento del diezmo. DE LOS POBRES. Lo que da el amo del campo al pobre de su cosecha, debe ser previamente diezmado. DOS, RECIBIERON UN CAMPO. Siendo ambos pobres. EN CALIDAD DE APARCEROS. Pudiendo obtener cada uno, la mitad, o un tercio, o una cuarta parte de la cosecha, según lo convenido. En este momento, son conceptuados como propietarios, donde, un propietario pobre, no puede tomar parte del Leket de su propio campo, tal como versa: “No recolectarás, para el pobre...” (Levítico 19:10). Esta es una advertencia para el pobre, para que no recolecte de su propio campo, sino que debe apartar el Diezmo del Pobre para dárselo a otro. Sin embargo, esta regla solo afecta al campo propio, por lo que, ambos pueden intercambiarse el diezmo del campo del compañero. CUANDO RECIBE LA MITAD... De lo que aún sigue adherido a la tierra, o sea, antes de la siega. Pero, si le asigna la mitad de lo que ya se recolectó, entonces, es el propietario del campo quien queda obligado a donar Leket, Shijjá y Peá, en cuyo caso, el obrero puede apoderarse de estas donaciones. Aunque el asunto de la Shijjá, sólo queda indicado en el supuesto de olvidarse la gavilla al amontonarlas, evidentemente, después de segadas, aún así, aquí está exento el obrero, pues no cabe el concepto de “cuando siegues tu campo”, siendo que se apropia de su salario después que salió el grano de la tierra. Una persona que se convirtió al judaísmo después de haber segado su campo queda obligado al precepto de no tocar lo que olvidó, Shijjá, según la opinión expuesta por Ribí Yehudá, simplemente, porque el párrafo dijo “de tu campo”. No obstante, el nuevo converso, precisamente por esta razón, está exento del Leket, porque, cuando se convirtió, o sea, cuando quedó obligado a cumplir los preceptos, el campo ya estaba segado. 6. QUIEN VENDE SU CAMPO. Quiere decir, que vendió el campo junto con lo plantado. Sin embargo, si sólo vendió la mies y se quedó con el campo, ambos tienen prohibido adueñarse del Leket, Shijjá y Peá, pues, frente a uno se prohibió porque dice el párrafo: no de “tu campo”, y frente al otro, no de “tu cosecha”. SE LE PERMITE. Tomar dichas tres donaciones, en caso de ser pobre. QUE SUS HIJOS RECOLECTEN. Porque, con esto, el obrero estaría cobrando parte de su salario, y el propietario, se estaría beneficiando a costa de los pobres. NO RETROCEDAS EL LINDE DE SIEMPRE. No leas “de siempre” (En hebreo, Olam), sino, “de los que emigraron” (En hebreo, Olim, literalmente, los que suben a la Tierra de Israel). Hay quien explica que se refiere a los emigrantes salidos de Egipto, para que no cambiaran las advertencias que se les encomendaron en Egipto. Otros explican: Se refiere a los que bajaron su nivel económico. No obstante, el término “los que suben”, es sólo un eufemismo. Así también, se dice en hebreo “Hombre de gran claridad” (Saguí Nehor), a los hombres ciegos. 7. NO FUE OLVIDADA POR EL DUEÑO. Como por ejemplo, si él se encontraba en el campo junto con los obreros. Sin embargo, si el dueño se encontraba en la ciudad, aunque no se olvide de dicha gavilla, se conceptúa como Shijjá. 8. EN FORMA DE SOMBRERO. Acostumbraban a unir espigas en forma de sombrero, al igual que anudaban espigas en forma de corona para colocarlas sobre la cabeza. JUMSÁ. Pilas de gavilla de menor altura que las de forma de sombrero. Eran inclinadas hacia debajo de forma que no sobresalieran demasiado, tal como aparece en el versículo: “¿No está guardado junto a Mí?” (Deuteronomio 32:34). EN FORMA DE CÍRCULO. Pila de gavilla, aún menor que la anterior, dispuesta circularmente como para poder hornear un pequeño pastel. Según otra interpretación, solían amontonar las espigas segadas en montículos hasta su posterior recolección. Entonces, las llevaban a otro lugar, el granero, para trillarlas. Según las formas que les daban a dichos montículos, recibían los diferentes nombres citados. PEQUEÑAS GAVILLAS. Juntado dos o tres de estas, formaban una gavilla normal. NO EXISTE SHIJJÁ. O sea, si olvidó en el campo alguno de estos montículos mientras los trasladaban a algunos de los lugares aquí citados, no eran denominados como Shijjá, por cuanto está escrito: “Cuado siegues la mies de tu campo, y olvides una gavilla” (ibíd.. 24:19). Al igual que se habla de una siega a la cual no puede seguir otra siega, la gavilla debe ser aquella que no le siga ninguna otra. Con esto, se excluye a aquellas gavillas que después se convierten en gavillas mayores. DESDE ALLÍ HASTA EL GRANERO. Quien lleva las gavillas de los montículos hacia el granero, olvidando una gavilla en alguno de ellos, se llama Shijjá. MIENTRAS LAS RECOPILA. Queriendo que queden allí para trillarlas, se llama Shijjá. DESDE ALLÍ HASTA EL GRANERO. Pero si se arrepiente más tarde y las lleva a otro lugar diferente para trillarlas, no se llama Shijjá.

    Dijo Rab Huná: Quien reza detrás de una Bet-Hakeneset, se denomina Rashá (Malvado), tal como versa: “Los malvados caminan en derredor” (Salmos 12:9). Dijo Abayé: Esto se dijo sólo cuando no miran hacia la Bet-Hakeneset, pero si miran hacia la Bet-Hakeneset, no importa. En cierta ocasión, un hombre se encontraba rezando detrás de una Bet-Hakeneset, mientras miraba para otro lado. Pasó por allí Eliyahu, y lo vio. Se le apareció con aspecto de comerciante árabe, y le dijo: “¡Das la impresión de que existen dos poderes en el mundo! Sacó un puñal y lo mató. Le preguntó un Sabio a Rab Bibí, hijo de Abayé (Otros dicen que la pregunta se la hizo Rab Bibí a Rab Najmán, hijo de Yitzjak): “¿Qué significa: «tal como el antílope acecha hambriento a los hombres»?” (ibíd..). Le contestó: Se refiere a las cosas más importantes de este mundo y que la gente menosprecia. Ribí Yojanán y Ribí Eliézer dijeron: Cuando una persona necesita (dinero) de otras, la cara le cambia (por la vergüenza) como si fuera un “krum”. Así lo expresa el versículo: “Como un «krum», les supone a las personas el menosprecio” (ibíd..). Pero ¿qué es un “krum”? Cuando vino Rab Dimí (desde Israel) respondió: Existe un ave en las ciudades costeras llamado “krum”, que cuando el sol brilla cambia a otros tantos colores. Ribí Amé y Ribí Asé dijeron: (Cuando una persona necesita de su prójimo) se siente como si lo juzgaran y lo condenaran a dos cosas contrapuestas, fuego y agua, tal como versa: “Haz hecho montar al hombre sobre nuestras cabezas. Pasamos por el fuego y por el agua” (ibíd. 66:12). Y dijo Ribí Jelbó en nombre de Rab Huná: Deberá toda persona ser cuidadoso con la Tefilá de Minjá, ya que Eliyahu fue escuchado en la Tefilá de Minjá, según el versículo: “Y aconteció al atardecer (Minjá) que se acercó el Profeta Eliyahu y dijo... Respóndeme, Di-s, respóndeme” (Reyes I 18:36-7). “Respóndeme”, haciendo que descienda fuego de los cielos, y “respóndeme”, para que no crean que todo esto es producto de la hechicería. Rabí Yojanán dijo: También, (será cuidadoso) con la Oración de Arbit, tal como versa: “Considera mi plegaria como el incienso que colocan ante Ti, cuando alzo mis manos durante la ofrenda vespertina” (Salmos 141:2). Rab Najmán, hijo de Yitzjak dijo: También con la Oración de Shajrit, pues así está escrito: “Eterno, por las mañanas atiende mi voz. Por las mañanas, las ordeno ante Ti, y confío” (ibíd. 5:4).

    Zóhar

    Vaerá 26a

    Comenzó Rabí Shimón diciendo: “Temed por la espada pues, irritar (al Eterno) es uno de los pecados que provocan que la espada aparezca, y sabed que hay justicia” (Job 19:29). Está escrito, “que hay juicio”. “Temed por la espada”, ¿qué espada? “La espada vengadora, para vengar el (incumplimiento del) Pacto” (Levítico 26:25). Esta espada se yergue para ver quién miente sobre el Pacto. Esta misma espada se venga de él. Por esto se dijo: “Irritar (al Eterno) es uno de los pecados que provoca que la espada aparezca”. ¿Por qué razón? Porque todo el que miente sobre el Pacto, aleja de sí el deseo, y no toma el que toma, ni lo coloca en su lugar, pues no sabe cuál es su lugar. Quien cuida el Pacto hace que dicho Pacto vuelva a su lugar, y, de esta forma, se bendicen los mundos superiores y los inferiores. ¿Quiénes son los que devuelven el Pacto a su lugar? Los Tzadikim mientras que viven en este mundo. ¿De dónde lo aprendemos? De aquí, donde vemos escrito: “y, además establecí Mi Pacto con ellos para darles la Tierra de Kenaán, la tierra de sus peregrinajes en la cual habitaron” (Éxodo 6:4). ¿Qué significa la tierra de sus peregrinajes (en hebreo, Me-gure-hem)? Lo mismo que cuando dices: “Temed (en hebreo, guru) por la espada”. Por cuanto es el lugar que infunde temor al mundo. Por ese se dijo: “Temed por la espada”. “En la cual habitaron (en hebreo, garu)”, o sea, que desde el momento que se acercaron al Eterno temieron en aquella tierra con gran temor, con el fin de cumplir Sus preceptos. Si en ella, el temor no se posa sobre sus cabezas, nunca tendrán temor a Di-s con respecto al resto de los preceptos. Ven y observa. Cuando hay un despertar en los mundos inferiores, cuando el Pueblo de Israel despertó y alzó su voz hacia el Eterno, ¿qué está escrito? “Y recordé Mi Pacto” (ibíd. 5), o sea, Él se acordó del Pacto. Entonces, existe un despertar que estimula una asociación única, pues, siendo que el Pacto se reavivó, entonces, toda la unión se reaviva. De aquí que: “Y recordé Mi Pacto”. Esto es, unirlo a su lugar. Por eso está dicho: “Por tanto, di a los Hijos de Israel: “Yo soy el Eterno”.

    Rambam, Leyes Sobre el Arrepentimiento, capítulo 3

    (1) Quienes infunden temor a la congregación, sin la intención de servir así al Eterno, o sea, el que doblega al Pueblo por la fuerza, logrando que este le tema, y lo hace para su propio beneficio, sin ninguna mira más trascendente, tal como sucede con los reyes gentiles; cada una de estas veinticuatro personas nombradas, aunque sean judíos, no recibirán su parte en el Mundo Venidero. (2) Existen otras transgresiones menos graves y que, sin embargo, los Sabios dijeron sobre ellas, que quien las transgrede asiduamente, no recibirá su porción en el Mundo Venidero. Por tanto, lo más oportuno es alejarse de ellas. Estas son: Quien inventa un apodo para nombrar a su prójimo, quien nombra al prójimo según un apodo, quien avergüenza al prójimo en público, quien pretende ganar honores a costa de la humillación del prójimo, quien desprecia a los Sabios, quien desprecia a sus Rabinos, quien desprecia las festividades y quien profana lo que es sagrado. (3) ¿Cuándo decimos que todos estos no tomarán su parte en el Mundo Venidero? Cuando fallece sin haberse arrepentido, pero si se arrepiente de su maldad, y muere habiéndose arrepentido, sí recibirá su parte en el Mundo Venidero, puesto que no hay nada que se anteponga ante el arrepentimiento. Incluso, si una persona renegó hasta de las cosas más esenciales durante todos los días de su vida, y, finalmente, se arrepiente, recibirá su heredad en el Mundo Venidero, tal como versa: “Paz, paz, para el que se encuentra alejado y para el que se acerca, dijo el Eterno. Yo lo perdonaré” (Isaías 57:19). (4) Todos los malvados, herejes y otros semejantes, que se arrepienten, tanto de forma abierta como de forma oculta, deben ser aceptados, tal como versa: “Retornad, hijos rebeldes” (Jeremías 3:22). Aunque en este momento es rebelde, pues se arrepintió de forma oculta, no abierta, se le acepta como arrepentido.

    Moral

    Zóhar, Parte II 2a

    Tres personas hacen que la Presencia Divina se aleje del mundo, y provocan que el Eterno desaparezca, - si es que esto fuera posible, Di-s no lo permita, - y, como consecuencia, las súplicas de los seres humanos no sean oídas. Estas son: Quien mantiene relaciones íntimas con una mujer Nidá (Mujer que menstrúa, o que, después del período de menstruación, no se sumergió en el Baño Ritual, o Mikvé para purificarse). No existe otra impureza peor que esta. Él se impurifica y, además, toda otra persona que mantiene contacto físico con él. Allá donde se encuentre, la Presencia Divina se aleja de él. Además, provoca la aparición de malas enfermedades, para él y para los suyos, aquellos que nazcan de la impureza. (Cuando un hombre se acerca a una mujer Nidá, se contagia de su impureza y permanece en todos sus miembros. La simiente que de él sale) Absorbe el espíritu impuro, y, así, vivirá todos sus días, impuro. Su construcción y su fundamento son la impureza. La peor y más grave de todas las impurezas de este mundo. Apenas se acerca a la mujer Nidá, la impureza se apodera de él, tal como versa: “Y llevará la impureza sobre él” (Levítico 15:24). La segunda, se refiere a aquel hombre que mantiene relaciones íntimas con una mujer gentil. Hace penetrar el sello del Pacto Sagrado en el dominio del Lado Oculto, tal como versa: “Y copuló con la hija de un dios extraño” (Malaquías 2:11). No hay celo como el que el Eterno siente por el sello del Pacto Sagrado, fundamento del Nombre Sagrado y fundamento de la fe. Quien introduce el sello del Pacto Sagrado en el dominio del Lado oculto, es como si mintiera en Nombre del Santo, Bendito Él. Quien miente con el Sello del Rey, es como si mintiera al Rey. No tendrá parte junto al Di-s de Israel, sino, a través de un grande y constante arrepentimiento. La tercera, quien asesina a sus hijos, esto es, el feto que su esposa porta en su vientre. Aquel que provoca su muerte en las entrañas de la madre, destruye lo que el Eterno construyó. El Eterno nos salve y nos haga arrepentirnos completamente. Así sea Su Voluntad.
