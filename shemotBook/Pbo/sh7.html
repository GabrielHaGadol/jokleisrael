מֹשֶׁה וַיִּשְׁאֲלוּ מִמִּצְרַיִם כְּלֵי כֶסֶף וּכְלֵי זָהָב וּשְׂמָלֹת: וּבְנֵי יִשְׂרָאֵל עֲבָדוּ כְּפִתְגָמָא דְמשֶׁה וּשְׁאִילוּ מִמִצְרַיִם מָנִין דִכְסַף וּמָנִין דִדְהַב וּלְבוּשִׁין: (לו) וַיהוָה נָתַן אֶת חֵן הָעָם בְּעֵינֵי מִצְרַיִם וַיַּשְׁאִלוּם וַיְנַצְּלוּ אֶת מִצְרָיִם: וַייָ יְהַב יָת עַמָא לְרַחֲמִין בְּעֵינֵי מִצְרַיִם וּשְׁאֵילִנוּן וְרוֹקִינוּ יָת מִצְרַיִם: (לז) וַיִּסְעוּ בְנֵי יִשְׂרָאֵל מֵרַעְמְסֵס סֻכֹּתָה כְּשֵׁשׁ מֵאוֹת אֶלֶף רַגְלִי הַגְּבָרִים לְבַד מִטָּף: וּנְטָלוּ בְנֵי יִשְׂרָאֵל מֵרַעְמְסֵס לְסֻכּוֹת כְּשִׁתְּ מֵאָה אַלְפִין רִגְלָאִין גֻבְרַיָא בַּר מִטַפְלָא: (לח) וְגַם עֵרֶב רַב עָלָה אִתָּם וְצֹאן וּבָקָר מִקְנֶה כָּבֵד מְאֹד: וְאַף נוּכְרָאִין סַגִיאִין סְלִיקוּ עִמְהוֹן וְעָנָא וְתוֹרִין בְּעִירָא סַגִי לַחֲדָא: (לט) וַיֹּאפוּ אֶת הַבָּצֵק אֲשֶׁר הוֹצִיאוּ מִמִּצְרַיִם עֻגֹת מַצּוֹת כִּי לֹא חָמֵץ כִּי גֹרְשׁוּ מִמִּצְרַיִם וְלֹא יָכְלוּ לְהִתְמַהְמֵהַּ וְגַם צֵדָה לֹא עָשׂוּ לָהֶם: וַאֲפוֹ יָת לַיְשָׁא דִי אַפִּיקוּ מִמִצְרַיִם גְרִיצָן פַּטִירָן אֲרֵי לָא חֲמָע אֲרֵי אִתָּרָכוּ מִמִצְרַיִם וְלָא יָכִילוּ לְאִתְעַכָּבָא וְאַף זְוָדִין לָא עֲבָדוּ לְהוֹן: (מ) וּמוֹשַׁב בְּנֵי יִשְׂרָאֵל אֲשֶׁר יָשְׁבוּ בְּמִצְרָיִם שְׁלֹשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה: וּמוֹתַב בְּנֵי יִשְׂרָאֵל דִי יְתִיבוּ בְּמִצְרָיִם אַרְבַּע מְאָה וּתְלָתִין שְׁנִין: (מא) וַיְהִי מִקֵּץ שְׁלֹשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה וַיְהִי בְּעֶצֶם הַיּוֹם הַזֶּה יָצְאוּ כָּל צִבְאוֹת יְהוָה מֵאֶרֶץ מִצְרָיִם: וַהֲוָה מִסוֹף אַרְבַּע מְאָה וּתְלָתִין שְׁנִין וַהֲוָה בִּכְרַן יוֹמָא הָדֵין נְפָקוּ כָּל חֵילַיָא דַיְיָ מֵאַרְעָא דְמִצְרָיִם: (מב) לֵיל שִׁמֻּרִים הוּא לַיהוָה לְהוֹצִיאָם מֵאֶרֶץ מִצְרָיִם הוּא הַלַּיְלָה הַזֶּה לַיהוָה שִׁמֻּרִים לְכָל בְּנֵי יִשְׂרָאֵל לְדֹרֹתָם: לֵילֵי נְטִּיר הוּא קֳדָם יְיָ לְאַפָּקוּתְהוֹן מֵאַרְעָא דְמִצְרַיִם הוּא לֵילְיָא הָדֵין קֳדָם יְיָ נְטִירִין לְכָל בְּנֵי יִשְׂרָאֵל לְדָרֵיהוֹן: (מג) וַיֹּאמֶר יְהוָה אֶל מֹשֶׁה וְאַהֲרֹן זֹאת חֻקַּת הַפָּסַח כָּל בֶּן נֵכָר לֹא יֹאכַל בּוֹ: וַאֲמַר יְיָ לְמשֶׁה וּלְאַהֲרֹן דָא גְזֵירַת פִּסְחָא כָּל בַּר יִשְׂרָאֵל דְיִשְׁתַּמַד לָא יֵכוּל בֵּהּ: (מד) וְכָל עֶבֶד אִישׁ מִקְנַת כָּסֶף וּמַלְתָּה אֹתוֹ אָז יֹאכַל בּוֹ: וְכָל עָבֵד גְבַר זְבִינֵי כַסְפָּא וְתִגְזַר יָתֵהּ בְּכֵן יֵכוּל בֵּהּ: (מה) תּוֹשָׁב וְשָׂכִיר לֹא יֹאכַל בּוֹ: תּוֹתָבָא וַאֲגִירָא לָא יֵכוּל בֵּהּ: (מו) בְּבַיִת אֶחָד יֵאָכֵל לֹא תוֹצִיא מִן הַבַּיִת מִן הַבָּשָׂר חוּצָה וְעֶצֶם לֹא תִשְׁבְּרוּ בוֹ: בַּחֲבוּרָא חֲדָא יִתְאָכֵל לָא תַפֵּיק מִן בֵּיתָא מִן בִּסְרָא לְבָרָא וְגַרְמָא לָא תִתְבְּרוּן בֵּהּ: (מז) כָּל עֲדַת יִשְׂרָאֵל יַעֲשׂוּ אֹתוֹ: כָּל כְּנִשְׁתָּא דְיִשְׂרָאֵל יַעְבְדוּן יָתֵהּ: (מח) וְכִי יָגוּר אִתְּךָ גֵּר וְעָשָׂה פֶסַח לַיהוָה הִמּוֹל לוֹ כָל זָכָר וְאָז יִקְרַב לַעֲשֹׂתוֹ וְהָיָה כְּאֶזְרַח הָאָרֶץ וְכָל עָרֵל לֹא יֹאכַל בּוֹ: וַאֲרֵי יִתְגַיַר עִמָךְ גִיוֹרָא וְיַעְבֵּד פִּסְחָא קֳדָם יְיָ מִגְזַר לֵהּ כָּל דְכוּרָא וּבְכֵן יִקְרַב לְמֶעְבְּדֵהּ וִיהֵי כְּיַצִיבָא דְאַרְעָא וְכָל עַרְלָא לָא יֵכוּל בֵּהּ: (מט) תּוֹרָה אַחַת יִהְיֶה לָאֶזְרָח וְלַגֵּר הַגָּר בְּתוֹכְכֶם: אוֹרַיְתָא חֲדָא יְהֵא לְיַצִיבָא וּלְגִיוֹרָא דְיִתְגַיְרוּן בֵּינֵיכוֹן: (נ) וַיַּעֲשׂוּ כָּל בְּנֵי יִשְׂרָאֵל כַּאֲשֶׁר צִוָּה יְהוָה אֶת מֹשֶׁה וְאֶת אַהֲרֹן כֵּן עָשׂוּ: וַעֲבָדוּ כָּל בְּנֵי יִשְׂרָאֵל כְּמָא דִי פַקִיד יְיָ יָת משֶׁה וְיָת אַהֲרֹן כֵּן עֲבָדוּ: (נא) וַיְהִי בְּעֶצֶם הַיּוֹם הַזֶּה הוֹצִיא יְהוָה אֶת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם עַל צִבְאֹתָם: וַהֲוָה בִּכְרַן יוֹמָא הָדֵין אַפֵּיק יְיָ יָת בְּנֵי יִשְׂרָאֵל מֵאַרְעָא דְמִצְרַיִם עַל חֵילֵיהוֹן: שביעי יג (א) וַיְדַבֵּר יְהוָה אֶל מֹשֶׁה לֵּאמֹר: וּמַלֵיל יְיָ עִם משֶׁה לְמֵימָר: (ב) קַדֶּשׁ לִי כָל בְּכוֹר פֶּטֶר כָּל רֶחֶם בִּבְנֵי יִשְׂרָאֵל בָּאָדָם וּבַבְּהֵמָה לִי הוּא: אַקְדֵשׁ קֳדָמַי כָּל בּוּכְרָא פָּתַח כָּל וַלְדָא בִּבְנֵי יִשְׂרָאֵל בֶּאֱנָשָׁא וּבִבְעִירָא דִילִי הוּא: (ג) וַיֹּאמֶר מֹשֶׁה אֶל הָעָם זָכוֹר אֶת הַיּוֹם הַזֶּה אֲשֶׁר יְצָאתֶם מִמִּצְרַיִם מִבֵּית עֲבָדִים כִּי בְּחֹזֶק יָד הוֹצִיא יְהוָֹה אֶתְכֶם מִזֶּה וְלֹא יֵאָכֵל חָמֵץ: וַאֲמַר משֶׁה לְעַמָא הֲווֹ דְכִירִין יָת יוֹמָא הָדֵין דִי נְפַקְתּוּן מִמִצְרַיִם מִבֵּית עַבְדוּתָא אֲרֵי בִּתְקוֹף יְדָא אַפֵּיק יְיָ יָתְכוֹן מִכָּא וְלָא יִתְאֲכֵל חֲמִיעַ: (ד) הַיּוֹם אַתֶּם יֹצְאִים בְּחֹדֶשׁ הָאָבִיב: יוֹמָא דֵין אַתּוּן נָפְקִין בְּיַרְחָא דְאַבִּיבָא: (ה) וְהָיָה כִי יְבִיאֲךָ יְהוָה אֶל אֶרֶץ הַכְּנַעֲנִי וְהַחִתִּי וְהָאֱמֹרִי וְהַחִוִּי וְהַיְבוּסִי אֲשֶׁר נִשְׁבַּע לַאֲבֹתֶיךָ לָתֶת לָךְ אֶרֶץ זָבַת חָלָב וּדְבָשׁ וְעָבַדְתָּ אֶת הָעֲבֹדָה הַזֹּאת בַּחֹדֶשׁ הַזֶּה: וִיהֵי אֲרֵי יָעֵלִנָךְ יְיָ לְאַרְעָא דִכְנַעֲנָאֵי וְחִתָּאֵי וֶאֱמוֹרָאֵי וְחִוָאֵי וִיבוּסָאֵי דִי קַיִים לַאֲבָהָתָךְ לְמִתַּן לָךְ אַרְעָא עַבְדָא חֲלָב וּדְבָשׁ וְתִפְלַח יָת פָּלְחָנָא הָדָא בְּיַרְחָא הָדֵין: (ו) שִׁבְעַת יָמִים תֹּאכַל מַצֹּת וּבַיּוֹם הַשְּׁבִיעִי חַג לַיהוָה: שִׁבְעָה יוֹמִין תֵּיכוּל פַּטִירָא וּבְיוֹמָא שְׁבִיעָאָה חַגָא קֳדָם יְיָ: (ז) מַצּוֹת יֵאָכֵל אֵת שִׁבְעַת הַיָּמִים וְלֹא יֵרָאֶה לְךָ חָמֵץ וְלֹא יֵרָאֶה לְךָ שְׂאֹר בְּכָל גְּבֻלֶךָ: פַּטִיר יִתְאֲכֵל יָת שִׁבְעַת יוֹמִין וְלָא יִתְחֲזֵי לָךְ חֲמִיעַ וְלָא יִתְחֲזֵי לָךְ חֲמִיר בְּכָל תְּחוּמָךְ: (ח) וְהִגַּדְתָּ לְבִנְךָ בַּיּוֹם הַהוּא לֵאמֹר בַּעֲבוּר זֶה עָשָׂה יְהוָה לִי בְּצֵאתִי מִמִּצְרָיִם: וּתְחַוֵי לִבְנָךְ בְּיוֹמָא הַהוּא לְמֵימָר בְּדִיל דֵין עֲבַד יְיָ לִי בְּמִפְּקִי מִמִצְרָיִם: (ט) וְהָיָה לְךָ לְאוֹת עַל יָדְךָ וּלְזִכָּרוֹן בֵּין      י(יד) לזכרון. לדורות: וחגתם אתו. יום שהוא לך לזכרון אתה חוגגו. ועדין לא שמענו אי זהו יום הזכרון, תלמוד לומר (שמות יג ג) זכור את היום הזה אשר יצאתם, למדנו שיום היציאה הוא יום של זכרון. ואיזה יום יצאו, תלמוד לומר (במדבר לג ג) ממחרת הפסח יצאו, הוי אומר יום חמשה עשר בניסן הוא של יום טוב, שהרי ליל חמשה עשר אכלו את הפסח ולבקר יצאו: לדרתיכם. שומע אני מעוט דורות שנים, תלמוד לומר חקת עולם תחגהו: (טו) שבעת ימים. שטיינ''א של ימים: שבעת ימים מצות תאכלו. ובמקום אחר הוא אומר (דברים טז ח) ששת ימים תאכל מצות, למד על שביעי של פסח שאינו חובה לאכול מצה, ובלבד שלא יאכל חמץ. מנין אף ששה רשות תלמוד לומר ששת ימים. זו מדה בתורה, דבר שהיה בכלל ויצא מן הכלל ללמד, לא ללמד על עצמו בלבד יצא, אלא ללמד על הכלל כלו יצא, מה שביעי רשות אף ששה רשות, יכול אף לילה הראשון רשות, תלמוד לומר (להלן פסוק יח) בערב תאכלו מצות, הכתוב קבעו חובה: אך ביום הראשון תשביתו שאר. מערב יום טוב, וקרוי ראשון לפי שהוא לפני השבעה, ומצינו מקדם קרוי ראשון, כמו (איוב טו ז) הראישון אדם תולד, הלפני אדם נלדת, או אינו אלא ראשון של שבעה, תלמוד לומר (שמות לד כה) לא תשחט על חמץ, לא תשחט הפסח ועדין חמץ קים: הנפש ההוא. כשהיא בנפשה ובדעתה, פרט לאנוס: מישראל. שומע אני תכרת מישראל ותלך לה לעם אחר, תלמוד לומר במקום אחר (ויקרא כב ג) מלפני, בכל מקום שהוא רשותי: (טז) מקרא קדש. מקרא שם דבר, קרא אותו קדש לאכילה ושתיה וכסות: לא יעשה בהם. אפלו על ידי אחרים: הוא לבדו. הוא ולא מכשיריו שאפשר לעשותן מערב יום טוב: לכל נפש. אפלו לבהמה יכול אף לנכרים, תלמוד לומר לכם: (יז) ושמרתם את המצות. שלא יבאו לידי חמוץ מכאן אמרו תפח, תלטוש בצונן. רבי יאשיה אומר אל תהי קורא את המצות אלא את המצוות, כדרך שאין מחמיצין את המצה כך אין מחמיצין את המצוה אלא אם באה לידך עשה אותה מיד: ושמרתם את היום הזה. ממלאכה: לדרתיכם חקת עולם. לפי שלא נאמר דורות וחקת עולם על המלאכה אלא על החגיגה, לכך חזר ושנאו כאן, שלא תאמר אזהרת (לעיל טז) כל מלאכה לא יעשה, לא לדורות נאמרה, אלא לאותו הדור: (יח) עד יום האחד ועשרים. למה נאמר, והלא כבר נאמר (לעיל טו) שבעת ימים, לפי שנאמר ימים, לילות מנין, תלמוד לומר עד יום האחד ועשרים וגו': (יט) לא ימצא בבתיכם. מנין לגבולין, תלמוד לומר (להלן יג ז)     Rashí

    Si ocurría que había un mitzrí en casa de un israelita, se hubiera podido pensar que se salvaría junto con el israelita.  Para indicar que no es así, este versículo enfatiza: “Y no habrá plaga entre ustedes”, lo que implica que sí la habrá entre los mitzrim que estén en las casas de ustedes.  Por otra parte, si había un israelita en casa de un mitzrí, yo hubiera podido entender que también el israelita sería abatido como lo sería el mitzrí.  Para indicar que no es así, este mismo versículo enfatiza: “Y no habrá plaga entre ustedes”.  14. לְזִכָּרוֹן / REMEMBRANZA. Para las generaciones futuras.  וְחַגֹּתֶם אֹתוֹ / Y LO CELEBRARÁN. Esto quiere decir que debes celebrar el día que es para ti una remembranza.  Pero todavía no hemos escuchado cuál de todos los días mencionados hasta ahora deberá ser remembranza para ustedes. Es por esta razón que la Torá declara: “Recuerda este día en que ustedes salieron” (Infra 13:3).  Eso nos enseña que el día en que ocurrió la salida de Mitzráim es el día de la remembranza.  ¿Y en qué día salieron?  Para informarnos de ello otro versículo dice: “El día siguiente a la ofrenda de Pésaj salieron” (Números 33:3).  Por lo tanto, se puede concluir que el 15 de Nisán fue el día de la celebración del festival, ya que en la noche del 15 de Nisán comieron la ofrenda de Pésaj y a la mañana siguiente salieron de Mitzráim.  לְדֹרֹתֵיכֶם וְגוֹ' / PARA SUS GENERACIONES, ETC. De esta expresión yo hubiera podido entender que el número mínimo de la palabra en plural “generaciones” es dos. Para indicar que no es así, este versículo declara: “Lo celebrarán por estatuto perpetuo”.  שִׁבְעַת יָמִים / UN PERIODO DE SIETE DÍAS. Esta expresión se refiere a un seteine de días.  שִׁבְעַת יָמִים מַצּוֹת תֹּאכֵלוּ / DURANTE UN PERIODO DE SIETE DÍAS COMERÁN PAN ÁZIMO. Sin embargo, en otro lugar la Torá dice: “Durante un periodo de seis días comerán pan ázimo” (Deuteronomio 16:8).  Esta aparente contradicción de versículos enseña que en el séptimo día de Pésaj no constituye una obligación comer matzá [pan ázimo], sino que es algo opcional, siempre y cuando la persona no coma jametz.  Ahora bien, ¿de dónde se infiere que incluso en los restantes seis días comer matzot es algo opcional y no obligatorio? Para ello el versículo declara: “Durante un periodo de siete días comerás pan ázimo”.  Éste es un principio de interpretación en la Torá: cualquier tópico específico que haya sido incluido en una categoría general, y luego haya sido extraído de la categoría general para enseñar una idea nueva acerca del tópico específico, no sólo fue extraído para enseñar acerca de sí mismo, sino para enseñar algo nuevo con respecto a la categoría general misma. Aplicando este principio a este caso, inferimos que así como comer matzá en el séptimo día de Pésaj es opcional, así también lo es durante los otros seis días del festival. En base a este razonamiento, se hubiera podido pensar que comer matzá es opcional incluso en la primera noche de Pésaj. Para indicar lo contrario, la Torá declara con respecto a la primera noche: “Por la noche comerán pan ázimo” (Infra 18).  Al decir esto, la Escritura lo instituyó como obligación.  אַךְ בַּיּוֹם הָרִאשׁוֹן תַּשְׁבִּיתוּ שְׂאֹר / PERO EN EL DÍA PREVIO ELIMINARÁN LA LEVADURA. En este versículo, la expresión “el primer día” se refiere a la víspera del día del festival. Pero aquí la Torá lo llama “el primer día” porque precede a los siete días del festival mismo. Y vemos que también en otros contextos lo que precede a algo es denominado “primero”.  Por ejemplo, en la frase: “¿Acaso naciste primero que Adam?”, la cual en realidad quiere decir: ¿Acaso naciste antes que Adam? ¿O quizás no es así, sino que la expresión “el primer día” realmente se refiere al primer día de los siete días del festival?  Para descartar esta suposición, el versículo declara: “No degollarán [la ofrenda de Pésaj] junto con sustancias leudadas, etc.”, frase que quiere decir: no degüellen la ofrenda de Pésaj mientras todavía haya jametz en su posesión. הַנֶּפֶשׁ הַהִוא / ESA ALMA. Mientras esté en posesión de su voluntad y conciencia, lo cual excluye a alguien que haya sido forzado a hacerlo.  מִיִּשְׂרָאֵל / DE ISRAEL. De esta frase yo entendería que dicha alma sería “cortada” de Israel en el sentido de expulsión física, pero que podría irse a otra nación.  Para indicar que no es así, la Torá declara en otro lugar: “Será cortada de Mi presencia” (Levítico 22:3), lo cual implica necesariamente que será “cortada” de cualquier lugar que sea Mi dominio.  16. מִקְרָא קֹדֶשׁ / CONVOCACIÓN DE SANTIDAD. La palabra מִקְרָא es un sustantivo. El significado de esta expresión es: Llama al primer día del festival “consagrado” con respecto a comer, beber y vestirse de un modo acorde con su carácter. לֹא יֵעָשֶׂה בָהֶם / NINGUNA [LABOR] DEBERÁ SER REALIZADA EN ELLOS. Incluso por medio de otras personas. הוּא לְבַדּוֹ / SÓLO ESO. Quiere decir: sólo eso –la labor necesaria para comer– podráser hecho, pero no sus preparativos que haya sido posible hacerlos desde la víspera del festival.  לְכָל נֶפֶשׁ / POR CUALQUIER ALMA. Incluso para los animales domésticos. Ahora bien, se podría haber pensado que la expresión “cualquier alma” también incluye a los gentiles.  Para indicar que no es así, este mismo versículo enfatiza “para ustedes”, lo cual implica para “ustedes” pero no para gentiles.  17. וּשְׁמַרְתֶּם אֶת הַמַּצּוֹת / GUARDARÁN EL PAN ÁZIMO. Para que no fermente.  De este versículo los Sabios dijeron: “Si la masa destinada a las matzot está a punto de inflamarse, que la persona que la esté amasando le eche agua fría para que el proceso de fermentación se retarde”.  Por otro lado, Rabí Yoshiyá dijo: “No leas la palabra únicamente como מַּצּוֹת, matzot [pan ázimo], sino también como מִּצְוֹת mitzvot [preceptos].  Así como no hay que dejar que las matzot fermenten, así tampoco hay que dejar ‘fermentar’ los preceptos, sino que si se te presenta una oportunidad para cumplir un precepto, hazlo de inmediato. וּשְׁמַרְתֶּם אֶת הַיּוֹם הַזֶּה / GUARDARÁN ESTE DÍA. Es decir, lo preservarán de la labor. לְדֹרֹתֵיכֶם חֻקַּת עוֹלָם / PARA SUS GENERACIONES, COMO ESTATUTO PERPETUO. Puesto que los términos “generaciones” y “estatuto perpetuo” no habían sido enunciados con respecto a la prohibición de realizar labores, sino en referencia a la celebración misma del festival, la Torá los repite de nuevo en este contexto con el propósito de evitar que digas: La prohibición de “no será realizada ninguna labor” no fue enunciada con respecto a las generaciones futuras, sino únicamente con respecto a esa generación que salió de Mitzráim. 18. עַד יוֹם הָאֶחָד וְעֶשְׂרִים / HASTA EL VIGÉSIMOPRIMER DÍA. ¿Por qué razón fue enunciada esta frase?  ¿Acaso no había sido dicho ya que no deberá ser hallada ninguna levadura durante “un periodo de siete días [יָמִים]”?  En realidad, este concepto es repetido aquí porque en aquel versículo sólo había sido enunciado יָמִים, “días”.  Ahora bien, ¿de dónde se deriva que ello también incluye a las noches?  Para indicar eso, este versículo declara: “Hasta el vigesimoprimer día del mes”, frase que necesariamente incluye a los días junto con sus noches. 19.לֹא יִמָּצֵא בְּבָתֵּיכֶם/ NO DEBERÁ HALLARSE [LEVADURA] EN SUS CASAS. ¿De dónde se deriva que esta prohibición también se aplica a la levadura que se halla dentro de las fronteras?  Para indicar eso, el versículo declara: “En todas tus fronteras” (Infra 13:7).  Siendo así, ¿por qué este versículo dice “en sus casas”?  Con el propósito de hacer la siguiente analogía: así como tu casa se halla bajo tu dominio y allí se aplica la prohibición, así también la prohibición sólo se aplica a tus fronteras siempre y cuando estén bajo tu dominio.  Ello excluye el jametz de un gentil que haya sido depositado con un israelita para que lo guarde y sobre el cual éste no haya aceptado ninguna responsabilidad en caso de pérdida o robo.  כִּי כָּל אֹכֵל מַחְמֶצֶת / PUES CUALQUIERA QUE INGIERA SUSTANCIAS LEUDANTES. Este versículo enseña que el castigo de karet se aplica a la persona que ingiera levadura. Pero ¿por qué la Torá necesita decirnos esto?  ¿Acaso no había decretado ya el castigo para la ingestión de jametz, alimento leudado?  En realidad, esta prohibición fue enunciada aquí para que no digas: La Torá decreta un castigo por el jametz porque éste es comestible, pero no decreta un castigo por la levadura, ya que ésta en sí misma no es comestible.  Por otro lado, si únicamente hubiera decretado un castigo por la levadura, pero no por el jametz, yo hubiera podido decir que la Torá decretó un castigo por la levadura porque ésta hace que otros alimentos fermenten, pero no decretó un castigo sobre el jametz porque éste no hace que otros alimentos fermenten.  Por esta razón –para incluir tanto a la levadura como a los alimentos leudados– ambos castigos fueron enunciados explícitamente.  בַּגֵּר וּבְאֶזְרַח הָאָרֶץ / YA SEA PROSÉLITO O NATIVO DE LA TIERRA. Puesto que el milagro de la salida de Mitzráim, conmemorado por este precepto, fue hecho directamente por Israel, era necesario incluir explícitamente a los prosélitos.  20. מַחְמֶצֶת לֹא תֹאכֵלוּ / NINGUNA SUSTANCIA LEUDANTE COMERÁN. En este versículo se enuncia la prohibición de comer levadura.  כָּל מַחְמֶצֶת / NINGUNA SUSTANCIA LEUDANTE. La palabra כָּל, “ninguna”, es enunciada con el propósito de incluir sus mezclas en la prohibición.  בְּכֹל מוֹשְׁבֹתֵיכֶם תֹּאכְלוּ מַצּוֹת / EN TODAS SUS MORADAS COMERÁN PAN ÁZIMO. Esta frase fue enunciada para enseñar que la matzá con la cual se cumple el precepto deberá ser halájicamente apta para ser ingerida “en todas sus moradas”.  Ello excluye matzot hechas del segundo diezmo [maaser shení] y las hogazas de la ofrenda de agradecimiento [todá], {ya que éstas no pueden ser ingeridas “en todas sus moradas”, sino sólo en Yerushalaim}.  21. מִשְׁכוּ / SAQUEN. La persona que ya posea un rebaño de ovejas o cabras deberá “sacar” el animal para la ofrenda de su propio rebaño.  וּקְחוּ / O COMPREN. La persona que no posea rebaño propio deberá comprar del mercado su animal para la ofrenda.  לְמִשְׁפְּחֹתֵיכֶם / PARA SUS FAMILIAS. Esta expresión es idéntica en significado a: “Un cordero [o cabrito] por casa paterna.”  22. אֵזוֹב / HISOPO. Este término designa a una especie de planta que posee tallos.  אֲגֻדַּת אֵזוֹב / UN MANOJO DE HISOPO. Tres raíces juntas reciben en hebreo el nombre de אֲגֻדָּה, “manojo”. אֲשֶׁר בַּסָּף / QUE ESTÁ EN LA JOFAINA. El término סָּף denota un utensilio. Es similar en significado al de la frase: “Instrumentos [סִפּוֹת] de plata” (Reyes II 12:14).  מִן הַדָּם אֲשֶׁר בַּסָּף/ CON LA SANGRE QUE ESTÁ EN LA JOFAINA. ¿Por qué razón la Torá repite de nuevo esta frase?  Para que no digas que es suficiente con un remojo del hisopo en la sangre para las tres aplicaciones de sangre, una en cada jamba de la puerta y la tercera en el dintel.  Por esta razón se menciona de nuevo la frase “que está en la jofaina”, con ello implicando que cada aplicación deberá ser realizada con la sangre que está en la jofaina.  Para cada contacto de la sangre en la puerta deberá haber un remojo correspondiente.  וְאַתֶּם לֹא תֵצְאוּ וְגוֹ' / Y EN CUANTO A USTEDES, QUE [NADIE] SALGA, ETC. Este versículo informa que después de que desde el Cielo se le ha dado permiso a la fuerza de destrucción para dañar, ya no distingue entre el justo y el malvado.  Y la noche constituye el dominio propio de las fuerzas dañinas, como se declara: “En ella [la noche] se arrastran todas las bestias del bosque” (Salmos 104:20). 23. וּפָסַח / [EL ETERNO] PASARÁ. Esta expresión implica aquí que se apiadará de ellos.  Pero también se puede decir que significa “pasará por”.  וְלֹא יִתֵּן הַמַּשְׁחִת / Y NO DEJARÁ AL DESTRUCTOR. La expresión לֹא יִתֵּן significa aquí que no le dará al agente destructor poder para entrar en las casas de los israelitas.  Este significado de la frase es similar al que tiene en el versículo: “Y Di-s no le dio [el poder] para hacerme mal” (Génesis 31:7).  25.וְהָיָה כִּי תָבֹאוּ אֶל הָאָרֶץ / Y CUANDO ENTREN A LA TIERRA. Este versículo hace que la realización de este precepto dependa la entrada en la Tierra de Israel.  Y en el desierto los israelitas sólo estuvieron obligados a ofrecer una sola ofrenda de Pésaj, la cual realizaron en el segundo año después de la salida de Mitzráim, por mandato expreso de Di-s.  כַּאֲשֶׁר דִּבֵּר / TAL COMO ÉL HABLÓ. ¿Y dónde habló esta promesa?  En el versículo: “Y llevaré a ustedes a la tierra, etc.” (Supra 6:8).  27. וַיִּקֹּד הָעָם / ENTONCES EL PUEBLO SE INCLINÓ. Por las buenas nuevas de la liberación y la entrada en la Tierra de Israel, y por la buena nueva de los hijos que tendrían.  28. … וַיֵּלְכוּוַיַּעֲשׂוּ בְּנֵי יִשְׂרָאֵל/ Y LOS HIJOS DE ISRAEL FUERON E HICIERON. Pero ¿acaso habían ya llevado a cabo estos preceptos?  ¿Acaso no se les había comunicado esto desde Rosh Jódesh, el primer día del mes, dos semanas antes de ofrecer la ofrenda de Pésaj? En realidad, puesto que en ese día aceptaron cumplir estos preceptos, la Escritura se los considera como si ya lo hubieran hecho.  וַיֵּלְכוּ וַיַּעֲשׂוּ / FUERON E HICIERON. La Escritura hace mención específica de su ida y de su puesta en práctica con el propósito de otorgarles recompensa tanto por la ida como por la puesta en práctica. כַּאֲשֶׁר צִוָּה ה' אֶת מֹשֶׁה וְאַהֲרֹן / TAL COMO EL ETERNO HABÍA ORDENADO A MOSHÉ Y A AHARÓN. Este versículo expresa el elogio de Israel, pues informa que no omitieron ninguna cosa de todos los preceptos que Moshé y Aharón les habían transimitido.  ¿Y qué significa la frase “así hicieron”, en apariencia redundante? Hay que informar que también Moshé y Aharón lo hicieron así, al igual que los demás.  29. וַה' / EL ETERNO. En todo lugar de la Torá en que se enuncia la expresión וַה', esa expresión se refiere a Di-s y a Su tribunal celeste, ya que el prefijo וַ expresa la idea de añadir, al igual que se utiliza la expresión “fulano y zutano”. הִכָּה כָל בְּכוֹר / MATÓ A TODOS LOS PRIMOGÉNITOS. Incluso al primogénito de otra nación que estuviese en Mitzráim.  מִבְּכֹר פַּרְעֹה / DESDE EL PRIMOGÉNITO DEL FARAÓN. El Faraón también era un primogénito, pero fue dejado con vida entre los primogénitos.  Y con respecto a él se dice: “A causa de esto te he mantenido, para mostrarte Mi poder” (Supra 9:16), manifestación que tuvo lugar en el Mar de Juncos.  עַד בְּכוֹר הַשְּׁבִי / HASTA EL PRIMOGÉNITO DEL CAUTIVO. Porque se habían alegrado del infortunio de Israel. Y además, para que no dijesen: “Nuestra deidad ha traido la retribución sobre los mitzrim”.  Y el primogénito de la esclava estaba incluido dentro de los que murieron en la plaga, ya que la Torá menciona a todos, desde el más distinguido –el primogénito del Faraón– hasta el menos distinguido –el primogénito del cautivo–, y el primogénito de una esclava es más distinguido que el primogénito de un cautivo. 30. וַיָּקָם פַּרְעֹה / EL FARAÓN SE LEVANTÓ. De su lecho.  לַיְלָה / DE NOCHE. Por la gravedad del caso, el Faraón no actuó conforme a la usanza de los reyes, los cuales se levantan de su lecho a la tercera hora del día.  הוּא / ÉL. Él se levantó primero, y después sus sirvientes.  Esto enseña que el Faraón fue de casa en casa de sus sirvientes y los levantó.  כִּי אֵין בַּיִת אֲשֶׁר אֵין שָׁם מֵת / PUES NO HABÍA CASA ALGUNA EN LA QUE NO HUBIERA UN MUERTO. Si allí había un primogénito, moría; si no había un primogénito, entonces el hijo más importante de la casa era considerado “primogénito” y moría, como está escrito: “Yo también a él lo haré [como] primogénito” (Salmos 89:28).  Según otra interpretación, en todas las casas de Mitzráim había muertos porque las mujeres mitzriot eran infieles a sus maridos y daban a luz de hombres solteros, y por ello tenían muchos hijos primogénitos, a veces hasta cinco por mujer, siendo cada uno de los hijos nacidos ilícitamente el primogénito para el padre que lo engendró.  31. וַיִּקְרָא לְמֹשֶׁה וּלְאַהֲרֹן לַיְלָה / LLAMÓ A MOSHÉ Y A AHARÓN DE NOCHE. Este versículo informa que el Faraón estuvo yendo de puerta en puerta en la ciudad, gritando: “¿Dónde pernocta Moshé?  ¿Dónde pernocta Aharón?”  גַּם אַתֶּם / TANTO USTEDES. Los varones.  גַּם בְּנֵי יִשְׂרָאֵל / COMO LOS HIJOS DE ISRAEL. Esta frase se refiere a los niños.  וּלְכוּ עִבְדוּ וְגוֹ' כְּדַבֶּרְכֶם / Y MARCHEN, SIRVAN... TAL COMO HABLARON. Quería decirles: Que todo se haga tal como ustedes mismos dijeron, y no como yo dije. Con esto me retracto de mis actitudes anteriores: “No enviaré fuera a Israel” (Supra 5:2) queda cancelada; “¿quiénes son los que irán?” (Supra 10:8) queda cancelada; “sólo sus rebaños y sus reses se quedarán fijos” (Ibíd. 24) queda cancelada.  32.גַּם צֹאנְכֶם גַּם בְּקַרְכֶם קְחוּ / TAMBIÉN TOMEN SUS REBAÑOS Y SUS RESES. ¿Qué significa aquí la frase “tal como hablaron”? Alude a la declaración de Moshé: “También tú nos entregarás animales para los sacrificios y las ofrendas de ascensión”.  En este versículo, el Faraón accedió a la exigencia de Moshé, pues declaró: “Tomen animales míos tal como ustedes hablaron.”  וּבֵרַכְתֶּם גַּם אֹתִי / Y BENDÍGANME TAMBIÉN A MÍ. Lo que quiso decir fue: “Recen por mí para que no muera, puesto que yo también soy primogénito”.  33. כֻּלָּנוּ מֵתִים / TODOS ESTAMOS MURIENDO. Ellos dijeron: Esta plaga no es como el decreto inicial de Moshé, pues él había dicho que “morirá todo primogénito” (Supra 11:5), pero en esta plaga incluso los hijos normales murieron, incluso cinco o diez en una sóla casa”.  34.טֶרֶם יֶחְמָץ / ANTES DE QUE FERMENTASE. Los mitzrim no les permitieron esperar el tiempo suficiente para que la harina fermentase.  מִשְׁאֲרֹתָם / SUS SOBRAS. Es decir, las sobras de la matzá y de las hierbas amargas.  עַל שִׁכְמָם / SOBRE SUS HOMBROS. A pesar de que los israelitas se llevaron consigo muchos animales que hubieran podido cargar en lugar de ellos, trataron amorosamente el objeto con el cual habían cumplido el precepto de la ofrenda de Pésaj.  35.כִּדְבַר מֹשֶׁה / CONFORME A LA PALABRA DE MOSHÉ. Que les había dicho en Mitzráim: “Que cada hombre pida a su vecino [mitzrí] utensilios de plata y oro” (Ibíd. 2). וּשְׂמָלֹת / Y VESTIMENTAS. También éstas eran más importantes para ellos que la plata y el oro. Lo que haya sido mencionado posteriormente en el versículo es más importante que lo mencionado primero.  36.וַיַּשְׁאִלוּם / Y LES PRESTARON. Incluso aquello que los israelitas no les habían pedido, los mitzrim se los dieron.  Un mitzrí le decía a un israelita: “Tú dices que yo te dé uno.  Toma dos y márchate”.  וַיְנַצְּלוּ / ASÍ VACIARON. Este término debe ser entendido como lo traduce el Targum: וְרוֹקֵינוּ, y vaciaron.  37. מֵרַעְמְסֵס סֻכֹּתָה / DE RAMESÉS A SUCOT. Entre estos dos lugares había una distancia de ciento veinte mils.  A pesar de eso, los israelitas llegaron allí en un instante, como se declara: “Y los porté sobre alas de águila” (Infra 19:4).  הַגְּבָרִים / VARONES. A partir de veinte años para arriba.  38. עֵרֶב רַב / UNA MUCHEDUMBRE ABIGARRADA. Una mezcla abigarrada de conversos de diversas naciones.  39.עֻגֹת מַצּוֹת / TORTAS ÁZIMAS. Esta expresión designa una torta hecha de matzá.  Una masa que no ha fermentado es llamada מַצָה, matzá. וְגַם צֵדָה לֹא עַשׂוּ לָהֶם/ Y TAMPOCO SE HABÍAN PREPARADO PROVISIONES. Para el camino. Este versículo expresa un elogio de los israelitas, que no dijeron: “¿Cómo hemos de salir al desierto sin provisiones?”  Al contrario, creyeron en la palabra de Di-s y partieron.  A esa actitud se refiere lo que se describe en los Profetas: “Recordé por ti la devoción de tu juventud, el amor de tus nupcias, tu marcha en el desierto tras de Mí, en una tierra no sembrada” (Jeremías 2:2).  ¿Y cuál es la recompensa por ello descrita después?  El versículo siguiente afirma: “Israel es santo para el Eterno, etc.” (Ibíd. 3).  40.אֲשֶׁר יָשְׁבוּ בְּמִצְרָיִם / DURANTE LA QUE HABITARON EN MITZRÁIM. Después de sus demás residencias durante las cuales también habitaron como forasteros en tierra ajena. שְׁלֹשִׁים שָׁנָה וְאַרְבַּע מֵאוֹת שָׁנָה / CUATROCIENTOS TREINTA AÑOS. La suma total de años a partir del nacimiento de Itzjak hasta este momento fue de cuatrocientos años.  Desde el momento en que Abraham tuvo descendencia, ya se cumplía la profecía descrita en el versículo: “Pues tu descendencia seráextranjera” (Génesis 15:13), más treinta años que transcurrieron desde que fue promulgado el decreto divino en el Pacto entre las Partes hasta el nacimiento de Itzjak.  Y no es posible decir que los israelitas habitaron durante cuatrocientos años sólo en Mitzráim, ya que Kehat, que era el abuelo de Moshé, era uno de los que llegaron a Mitzráim junto con Yaacob. Ahora bien, vé y haz la cuenta de todos sus años y de todos los años de su hijo Amram, padre de Moshé, más los ochenta años que tenía Moshé al salir de Mitzráim, y no hallarás que son tantos como se describe aquí.  Además, Kehat necesariamente vivió muchos años antes de descender a Mitzráim, y también muchos de los años de Amram están comprendidos dentro de los años de Kehat, y muchos de los años de Moshé están comprendidos dentro de los años de Amram. Así, pues, en total no hallarás cuatrocientos años desde la llegada de los israelitas a Mitzráim.  Por lo tanto, estás obligado a decir que el tiempo transcurrido en las demás residencias de los israelitas antes de llegar a Mitzráim también son definidas como “extranjería”, en cumplimiento de la profecía de “tu descendencia será extranjera”.  E incluso el tiempo que los patriarcas vivieron en Jebrón es definida como extranjería, como se declara: “...Kiriat Arbá es Jebrón, donde moraron Abraham e Itzjak” (Ibíd. 35:27).  Y también dice el versículo con respecto a la estadía de los patriarcas en Kenaan: “La tierra de sus moradas como forasteros, en la cual habitaron.” (Supra 6:4).  Por consiguiente, estás obligado a concluir que la profecía de “tu descendencia será extranjera” comenzó a cumplirse desde que Abraham tuvo descendencia, es decir, a partir del nacimiento de Itzjak.  Y si haces la cuenta de los cuatrocientos años que transcurrieron desde el nacimiento de Itzjak, hallarás que desde la llegada de los israelitas a Mitzráim hasta su salida en realidad sólo transcurrieron doscientos diez años.  Y ésta es una de las cosas que los Sabios modificaron en su traducción de la Torá al griego para el rey Ptolomeo de Mitzráim.  41.וַיְהִי מִקֵּץ שְׁלֹשִׁים שָׁנָה וְגוֹ' וַיְהִי בְּעֶצֶם הַיּוֹם הַזֶּה / FUE AL TÉRMINO DE [CUATROCIENTOS] TREINTA AÑOS, Y FUE EN ESE MISMO DÍA. Este versículo nos informa que, puesto que ya había llegado el término del exilio de los israelitas, el Omnipresente no los retuvo más ni siquiera el tiempo que tarda un parpadeo de ojo.  El 15 de Nisán llegaron los ángeles ministeriales ante Abraham para anunciarles la buena nueva del nacimiento de Itzjak, el 15 de Nisán del año siguiente nació Itzjak, y también el 15 de Nisán se promulgó el decreto divino concerniente al exilio, que Di-s había comunicado a Abraham en el Pacto entre las Partes.  42 .{הוּא לַה'} לֵיל שִׁמֻּרִים / [ES] NOCHE DE GUARDAR {PARA EL ETERNO}. Esta noche es llamada así porque el Santo –Bendito Es– la “guardaba” en mente y la esperaba para cumplir Su promesa de sacarlos de la tierra de Mitzráim. הוּא הַלַּיְלָה הַזֶּה לַה' / ESTA ES LA NOCHE DEL ETERNO. Quiere decir: esta era la noche de la cual Di-s le había dicho a Abraham: “Esa noche Yo liberaré a tus hijos.” שִׁמֻּרִים לְכָל בְּנֵי יִשְׂרָאֵל לְדֹרֹתָם / UNA SALVAGUARDA PARA TODOS LOS HIJOS DE ISRAEL PARA SUS GENERACIONES. Quiere decir: esta noche ya ha sido preservada contra las fuerzas espirituales dañinas, en el mismo sentido que se declara: “Y [Di-s] no permitirá al destructor, etc.”.  43. זֹאת חֻקַּת הַפָּסַח / ESTE ES EL ESTATUTO DE LA OFRENDA DE PÉSAJ. En el día catorce de Nisán se les comunicó esta sección. כָּל בֶּן נֵכָר / NINGÚN EXTRAÑO. La referencia es a una persona cuyos actos se han vuelto “extraños” a su Padre que está en los Cielos.  Esta categoría implica tanto a un gentil como a un israelita apóstata.  44.וּמַלְתָּה אֹתוֹ אָז יֹאכַל בּוֹ / LO CIRCUNCIDARÁS; [SÓLO] ENTONCES PODRÁ COMER DE ELLA. El sujeto de la útima cláusula se refiere al amo.  Ello nos informa que el hecho de no circuncidar a su siervo le impedirá comer de la ofrenda de Pésaj; ésta es la opinión de Rabí Yehoshúa.  Sin embargo, Rabí Eliézer dijo: el no circuncidar a sus siervos no le impedirá comer de la ofrenda de Pésaj.  Siendo así, ¿qué es lo que el versículo quiere decir con “entonces podrá comerlo”?  Se refiere al siervo.  45.תּוֹשָׁב / RESIDENTE. Esto se refiere a un residente extranjero. וְשָׂכִיר / ASALARIADO. Esto se refiere a un gentil. ¿Y para qué es necesario este versículo?  ¿Acaso no se trata de un incircunciso, y ya se ha dicho que “ningún extraño podrá comer” (Infra 48) la ofrenda de Pésaj? No obstante, era necesario enunciar este versículo para excluir, por ejemplo, a un árabe y a un guibeoní circuncidados que sean residentes o asalariados.  46. בְּבַיִת אֶחָד יֵאָכֵל / EN UNA SOLA CASA DEBERÁ SER COMIDA. Es decir, en un solo grupo de gente.  Este versículo indica que las personas registradas para la ofrenda de Pésaj no deberán formar dos grupos y dividírsela para comerla.  ¿Se puede afirmar sin dudas que esta frase se refiere a un solo grupo, o quizás –como se entiende literalmente– se refiere a una sola casa –es decir, un solo lugar–, en cuyo caso enseñaría que si las personas registradas ya comenzaron a comerlo en el patio y en ese momento cayeron lluvias, no podrán entrar a la casa para terminar de comerlo?  Para indicar que no es así, el versículo declara: “En las casas en las cuales se lo comerá” (Supra 7).  De aquí se deduce que la persona que lo coma podrá hacerlo en dos lugares. לֹא תוֹצִיא מִן הַבַּית / NO SACARÁS [LA CARNE] FUERA DE LA CASA. Es decir, del grupo.  וְעֶצֶם לֹא תִשְׁבְּרוּ בוֹ {יֵאָכֵל} / {DEBERÁ SER COMIDA...} Y NO LE QUEBRARÁN NINGÚN HUESO. Se refiere a un hueso apto para ser ingerido. Por ejemplo si en el hueso hay carne en la medida de un kezait, entonces se le aplicará la prohibición de romper un hueso.  Pero si no hay en él la medida de un kezait de carne, entonces no se le aplicará la prohibición de romper un hueso.  47.כָּל עֲדַת יִשְׁרָאֵל יַעֲשׂוּ אֹתוֹ / TODA LA ASAMBLEA DE ISRAEL LO HARÁ ¿Por qué razón se enuncia este pasaje?  Puesto que acerca de la ofrenda de Pésaj en Mitzráim el versículo dice: “Un cordero por casa paterna” (Supra 3), lo cual indica que se registraron para él por familias, se podría haber pensado que esto también se aplica a la ofrenda de Pésaj para las generaciones futuras. Pero indicar que no es así, este versículo declara: “Toda la asamblea de Israel lo hará”.  48. וְעָשָׂה פֶסַח / Y HAGA LA OFRENDA DE PÉSAJ. De este versículo se podría pensar que todo individuo que se convierta al judaísmo deberá hacer la ofrenda de Pésaj inmediatamente después de su conversión.  Para indicar que no es así, el versículo declara que “será como el nativo de la tierra”, es decir, como cualquier otro judío. Así como el nativo sólo ofrece la ofrenda de Pésaj en el catorce de Nisán, así también el converso sólo lo podrá hacer el catorce de Nisán.  וְכָל עָרֵל לֹא יֹאכַל בּוֹ / PERO NINGÚN INCIRCUNCISO COMERÁ DE ELLA. Esta frase fue enunciada para incluir a quien se le hayan muerto sus hermanos a resultas de la circuncisión, que no se convierte en apóstata [meshumad] a pesar de que quede incircunciso.  Puesto que ese individuo queda exento del precepto de la cirncuncisión, esta ley no se aprende del versículo: “Ningún extraño comerá de ella”.  49.תּוֹרָה אַחַת וְגוֹ' / UNA MISMA LEY, ETC. Este versículo se enuncia con el propósito de igualar un converso a un israelita nativo también para los demás mandamientos de la Torá.  2. פֶּטֶר כָּל רֶחֶם / EL QUE ABRA MATRIZ. Esta expresión designa al recién nacido que fue el primero en abrir la matriz. Es similar en significado a la palabra פּוֹטֵר en el versículo: “Como quien hace irrumpir [פּוֹטֵר] aguas haciendo un agujero en una presa, es quien provoque el inicio de una querella” (Proverbios 17:14).  Y de igual modo, la frase יַפְטִירוּ בְּשָׂפָה significa: “abrirán los labios” (Salmos 22:8). לִי הוּא  / MÍO ES. Para Mí mismo los he adquirido por medio de la plaga que causé a los primogénitos de Mitzráim. 3.זָכוֹר אֶת הַיּוֹם הַזֶּה / RECUERDEN ESTE DÍA. Este versículo nos enseña que cada día hay que recordar la salida de Mitzráim. 4. בְּחֹדֶשׁ הָאָבִיב / EN EL MES DE LA PRIMAVERA. Pero ¿acaso no sabían en que mes salieron? ¿Qué es entonces lo que este versículo enseña? La respuesta es que Moshé no les estaba informando de la época, sino que les dijo lo siguiente: “Vean la bondad que Di-s les ha concedido al sacarlos en un mes idóneo para salir, ni con sol ardiente ni con frío, y tampoco con lluvias”.  Y de modo similar se declara: “[Di-s] saca a los prisioneros cuando es propicio” (Salmos 68:7), es decir, en un mes adecuado para salir. 5.אֶל אֶרֶץ הַכְּנַעֲנִי וְגוֹ' / A LA TIERRA DEL KENAANÍ. A pesar de que en este versículo sólo se enumera a cinco de los siete pueblos de Kenaan, implica a todos los siete pueblos, ya que todos ellos están incluidos dentro de la categoría de “kenaanim”. Ahora bien, el pueblo “kenaaní” es mencionado por separado porque el nombre de una de las tribus de Kenaan era solamente “kenaaní”. נִשְׁבַּע לַאֲבֹתֶיךָ / JURÓ A TUS PADRES. Acerca de Abraham se declara: “En ese día el Eterno hizo un pacto con Abram, etc.” (Génesis 15:18).  Con respecto a Itzjak se declara: “Habita en esta tierra, etc.” (Ibíd. 26:3).  Y con respecto a Yaacob se declara: “La tierra sobre la que yaces, etc.” (Ibíd. 28:3).  זָבַת חָלָב וּדְבָשׁ / QUE MANA LECHE Y MIEL. La leche mana de las cabras y la miel mana de los dátiles y de los higos.  אֶת הָעֲבוֹדָה הַזֹּאת / ESTE SERVICIO. De la ofrenda de Pésaj. Sin embargo, ¿acaso no había sido dicho más arriba: “Y sucederá que cuando entren a la tierra, etc.”? (Supra 12:25).  Entonces, ¿por qué razón la Escritura lo repite de nuevo? Debido a un tópico que es introducido por primera vez en este versículo. En un versículo anterior se dice: “Y sucederá que cuando sus hijos les digan: ¿Qué es este servicio para ustedes?”  En ese lugar, el versículo habla acerca del hijo malvado, el cual se sustrajo a sí mismo de la categoría de los que cumplen este servicio.  Pero aquí se dice: “Y relatarás [וְהִגַּדְתָּ] a tu hijo”, refiriéndose al hijo que no sabe lo suficiente para preguntar.  La Escritura te enseña en ese versículo que tú deberás iniciar el tema de la salida de Mitzráim con relatos agádicos que atraen el corazón de quienes los escuchan.  8.בַּעֲבוּר זֶה / FUE POR ÉSTO. Es decir, para que yo cumpla Sus mandamientos; por ejemplo la ofrenda de Pésaj, la matzá y las hierbas amargas. עָשָׂה ה' לִי / EL ETERNO ACTUÓ POR MÍ. En esta frase la Torá alude a la respuesta que se debe dar al hijo malvado, como diciendo: “El Eterno actuó por mí, pero no por ti.  Pues si tú hubieras estado allí no hubieras sido digno de ser liberado”.  9.וְהָיָה לְךָ לְאוֹת / SERÁ SEÑAL PARA TI. Es decir, la salida de Mitzráim será un signo para ti. עַל יָדְךָ וּלְזִכָּרוֹן בֵּין עֵינֶיךָ / SOBRE TU MANO, Y REMEMBRANZA ENTRE TUS OJOS. {Esto quiere decir} que escribas estos pasajes bíblicos que hablan de la salida de Mitzráim y los amarres en tu cabeza y en tu brazo. עַל יָדְךָ / SOBRE TU MANO. Sobre la mano izquierda.  Por ello la expresión “tu mano” está escrita en forma completa [יָדְכָה] en el segundo pasaje que habla acerca del mandamiento de tefilín, para interpretarlo en referencia a la mano que es débil [כֵּהָה]. 10.מִיָּמִים יָמִימָה / DE AÑO EN AÑO. Esta expresión significa: de año en año.  11. וְהָיָה כִּי יְבִאֲךָ / Y SUCEDERÁ QUE CUANDO [EL ETERNO] TE LLEVE. Entre nuestros Maestros hay quienes aprendieron de este versículo que los israelitas no consagraron a los hijos primogénitos que les nacieron en el desierto.  Y aquel Maestro que afirma que sí los consagraron, explica este “llevar” en el sentido de que si los israelitas cumplen el mandamiento de consagrar a los primogénitos en el desierto, entonces se harán dignos de entrar a la Tierra de Israel y también podrán cumplirlo allí (T. Bejorot 4b-5a). נִשְׁבַּע לְךָ / JURÓ A TI. ¿Y dónde Él te lo juró?  En el siguiente versículo: “Yo los llevaré a la tierra por la cuál he alzado Mi mano, etc.” (Supra 6:8).  וּנְתָנָהּ לָךְ / Y TE LA HAYA ENTREGADO. La Tierra de Israel deberá ser considerada en tus ojos como si Él te la hubiera entregado en ese mismo día, y no deberá ser considerada en tus ojos como una herencia de antepasados.  12. וְהַעֲבַרְתָּ / ENTONCES SEPARARÁS. El verbo וְהַעֲבַרְתָּ en este contexto sólo puede significar separación.  Y de igual modo la Torá utiliza un término similar cuando declara: “Y pondrán aparte [וְהַעֲבַרְתֶּם] su propiedad para su hija.” (Números 27:8). שֶׁגֶר בְּהֵמָה / EXPULSADO DE MATRIZ DE LAS BESTIAS. El término שֶׁגֶר se refiere a un feto cuya madre lo expulsó del vientre, enviándolo fuera antes de su tiempo.  Este versículo te enseña que la cría primeriza de un animal es consagrada en su primogenitura para fines de exentar a la cría que venga después de él de la condición de cría primeriza.  Y a pesar de que un feto malogrado [nefel] generalmente no es llamado שֶׁגֶר –como por ejemplo en el versículo: “Él bendecirá... al animal parido [שֶׁגֶר] de tu ganado” (Deuteronomio 7:13), sin embargo, en este caso este término sólo fue enunciado para enseñar acerca de un feto que ha sido expulsado del vientre antes de su tiempo.  Pues con respecto a un parto normal la Torá ya escribió al principio de este versículo: “Pondrás aparte todo el que abra matriz”. Y si dices que las crías primerizas de un animal impuro [no-kasher] también están comprendidas dentro del mandamiento de consagrar a los primerizos, para indicar que no es así la Torá declara explícitamente en otro lugar: “De todo [animal] primogénito... de tus reses o de tu rebaño” (Ibíd. 15:19), lo cual no se aplica a animales impuros. Otro modo de explicar la frase, “entonces separarás a todo el que abra matriz”, es decir que la Escritura habla acerca de primogénitos de ser humano. 13.פֶּטֶר חֲמֹר / PRIMERIZO DE ASNO. Pero no el primerizo de cualquier otro animal impuro.  Y esta ley constituye un decreto de la Escritura, ya que los primogénitos de los mitzrim fueron comparados a asnos.  Y además, los primerizos de asno tienen un status diferente al de los demás animales impuros porque ellos asistieron a Israel a su salida de Mitzráim, {pues no hubo ni uno sólo de los israelitas que no haya tomado consigo numerosos asnos} cargados con el oro y la plata de Mitzráim. תִּפְדֶּה בְשֶׂה / REDIMIRÁS POR UN CORDERO [O CABRITO]. El israelita entregaba el cordero o cabrito a un kohén, y entonces la cría primeriza que abriera matriz de asno quedaba permitida para obtener de ella beneficio personal.  Y el cordero [o cabrito] pasaba a la condición de profano –es decir, no consagrado– en manos del kohén.  וַעֲרַפְתּוֹ / LE DARÁS UN TAJO EN LA CERVIZ. Le daba un tajo con un cuchillo largo en la parte posterior del cuello, matándolo.  El individuo que no había redimido su asno había causado una pérdida monetaria al kohén al no redimirlo, y por ello se le hacía sufrir una pérdida monetaria con la muerte del animal. וְכֹל בְּכוֹר אָדָם בְּבָנֶיךָ תִּפְדֶּה / Y A TODO PRIMOGÉNITO HUMANO DE ENTRE TUS HIJOS REDIMIRÁS. En otro lugar de la Escritura se fijó el precio de su rescate en cinco selaim (Números 18:16).  14.כִּי יִשְׁאָלְךָ בִנְךָ מָחָר / CUANDO TU HIJO TE PREGUNTE EL DÍA DE MAÑANA. Hay casos en que la expresión מָחָר “mañana”, se refiere a un hecho inmediato, en cuyo caso significa literalmente “al otro día”, y hay casos en que dicha expresión se refiere a un hecho que habrá de ocurrir después de cierto tiempo; por ejemplo, en este versículo y en el siguiente: “El día de mañana [מָחָר] dirán sus hijos a nuestros hijos” (Josué 22:24), versículo dicho con respecto a los hijos de Gad y los hijos de Reubén.  מַה זֹּאת / ¿QUÉ ES ESTO? Esta es la pregunta del niño tonto que no sabe profundizar en su pregunta y vagamente interroga: “¿Qué es esto?”  Y en otro lugar la Torá dice: “¿Cuáles son los testimonios, los estatutos y las leyes, etc.?” (Deuteronomio 6:20).  Ese versículo contiene la pregunta hecha por el hijo sabio.  Así, pues, la Torá habló acerca de cuatro clases de hijos: el malvado, el que no sabe preguntar, el que pregunta en forma vaga y el que pregunta de un modo sabio.  16.וּלְטוֹטָפֹת בֵּין עֵינֶךָ / Y DIADEMA ENTRE TUS OJOS. La palabra טוֹטָפֹת, totafot, se refiere al tefilín de la cabeza.  Y debido a que está compuesto por cuatro compartimentos donde se colocan los cuatro pergaminos escritos con pasajes de la Torá, es llamado totafot.  Pues la palabra tat en el idioma kaspi significa “dos”; y la palabra pat en el idioma afriki denota “dos”, lo cual hace un total de cuatro.  Pero Menajem ben Saruk clasificó este vocablo asociándolo al verbo וְהַטֵּף en la frase: “Y predica [וְהַטֵּף] en el sur” (Ezequiel 21:2), así como al verbo תַּטִּיפוּ en la frase: “No predicarán תַּטִּיפו]]” (Miqueas 2:6).  Según él es un vocablo que expresa el concepto de “habla”.  El sentido que tiene aquí sería similar a lo dicho en un versículo anterior con respecto a los tefilín: “Y [serán] remembranza entre tus ojos”, ya que el que observe los tefilín amarrados en la parte de la frente que está entre los ojos recordará el milagro de la salida de Mitzráim y hablará de él.

    Mishná

    Tratado Pará, capítulo 5

    (1) El encargado de traer el recipiente de arcilla para el servicio de “la Purificación” (Vaca Roja), debe sumergirse, y dormir junto al horno (del alfarero). Rabí Yehudá dice: También se le permite traerlo directamente de la casa (del alfarero), y es Kasher porque todo el mundo es fiable con relación a “la Purificación”. Con respecto a la Terumá, solo tiene que abrir el horno y agarrarlo. Rabí Shimón dice: Puede agarrarlo de la segunda hilera. Rabí Yosé dice: Sólo desde la tercera. (2) Quien sumerge un recipiente para Purificación en aguas no aptas para consagrar, debe secarlo. (Si lo sumergió) En aguas que son aptas para consagrar, no necesita secarlo. Si lo sumergió para recoger agua consagrada, tanto si son aptas como si no, deberá secarlo. (3) Si se sumerge una Keruyá en aguas no aptas para consagrar, deberá volver a sumergirla en aguas aptas, y la podrá usar hasta el día en que se impurifique. Sin embargo, si se impurificó, ya no se puede utilizar para consagrar. Ribí Yehoshúa dice: Si comenzó a utilizarla para consagrar, puede seguir haciéndolo al final (una vez sumergida, en caso de que se haya impurificado). No obstante, si no tiene intención de utilizarla al final para purificar, tampoco la podrá utilizar desde el principio. Pero, sea como sea, no podrá recoger en ella agua consagrada. (4) Un tubo cortado para usarlo en la Purificación, según Rabí Eliézer, debe ser sumergido enseguida. Rabí Yehoshúa dice: Lo impurificará y después lo sumergirá. Toda persona es apta para purificar a excepción de los sordomudos, los deficientes mentales y los niños. Rabí Yehudá considera aptos a los niños, pero descalifica a las mujeres y a los hermafroditas. (5) Todo tipo de recipiente es aceptado, incluso el que fue hecho de estiércol, de piedra o de tierra. En una embarcación también se permite consagrar. No se permite consagrar sobre las paredes de los recipientes (rotos), ni sobre las asas de los cántaros, ni sobre los tapones de los barriles, ni sobre sus manos, porque no se permite ni almacenar, ni purificar, ni rociar el agua de Purificación si no está contenida en un recipiente. Solo los recipientes con tapadera previenen del contagio ante la impureza, y también, un contenedor de arcilla solo previene ante la impureza a los recipientes. (6) El huevo de los alfareros es Kasher. Rabí Yosé lo considera inválido. El huevo de una gallina, tanto Rabí Meir como Rabí Yehudá lo consideran válido, pero los Sabios, no. (7) La oquedad producida en una roca no se puede usar para almacenar, ni para purificar, ni para rociar agua de purificación. No necesita de una tapadera ajustada y no invalida un Baño Ritual o Mikvé. Si con una roca se formó un recipiente y le pegó a ella con cal posteriormente, se puede utilizar para contener agua, y para purificar y para rociar de ella. Además, necesita de una tapadera ajustada y convierte en inválido un Mikvé. Si este recipiente tiene un agujero por abajo y lo cierra con un trapo, las aguas contenidas en él no son válidas porque no están rodeadas de agua totalmente como lo haría un recipiente. Si el agujero es lateral (y alto), y lo tapa con un trapo, las aguas contenidas en él son aptas, por cuanto están totalmente rodeadas por el recipiente. Si le hizo una corona con arcilla y el agua llega a tocarla, dicha agua no es válida. No obstante, si estaba esta corona muy bien pegada al recipiente, hasta el punto de que si lo agarra por allí, agarra todo el recipiente, es válida. (8) Dos cubetas hechas en una misma piedra, si purificó con una de ellas, las aguas de la contigua no son conceptuadas como purificadoras. Si ambas estaban comunicadas por un tubo tan ancho como la boca de un odre, o que el agua rebosaba de una a otra, aunque sea en una medida como el ancho de la cáscara del ajo, y purificó en una de las dos, las aguas de la otra también se consagran. (9) Dos piedras que se juntan y forman entre ellas una cubeta, o dos artesas, o una cubeta dividida en dos, las aguas que hubiera en medio no serían consideradas puras. Si las pegó con cal o con escayola de forma que pueden mantenerse como una, las aguas de en medio se consideran purificadas.

    Rabí Ovadia de Bartenura

    1. EL ENCARGADO. Quien va al alfarero a comprar el recipiente necesario para colocar la ceniza de la Vaca Bermeja o el Agua de Purificación. DORMIR JUNTO AL HORNO. Con la intención de preservar su pureza, evitando que alguna persona impropia lo toque antes de cocerlo, que es lo que determina el final de la fabricación, y el comienzo de su posibilidad de impurificarse. Y ES KASHER. Porque si un profano dice que se cuidó de que no se impurificara, lo creemos. Así se dijo en el capítulo “Jomer Bakodesh” (Tratado de Jaguigá 22): Cualquier profano es creíble con relación a la pureza del vino para las libaciones y a la del aceite para las ofrendas Minjá, con la idea de que cada cual no construya su propio altar y queme su propia Vaca Bermeja. La Ley no se dictaminó como Rabí Yehudá. LA TERUMÁ. Si lleva un recipiente de arcilla para colocar la Terumá. ABRIR EL HORNO. Aunque haya acabado el cocido y no lo haya cuidado. Siendo que aún no se abrió y está lleno de polvo, no sospechamos que alguien profano lo haya tocado. En Terumá no fueron muy exigentes. DE LA SEGUNDA HILERA. Si un profano hubiese abierto el horno con anterioridad para tomar recipientes, no se deberá coger después de la primera fila (por ser la que más se toca), sino, de la segunda. Rabí Yosí prohíbe incluso de la segunda. La Ley no concuerda ni con Rabí Shimón ni con Rabí Yosí, y, por tanto, aunque encuentre el horno abierto todo es potencialmente puro, y podrá coger incluso de la primera hilera. 2. QUIEN SUMERGE UN RECIPIENTE. Si lo necesita para llenarlo de agua y mezclarlo con ceniza para purificar, y lo introduce en agua no apta para la purificación, como la de un baño ritual o Mikvé, con agua que no es de manantial. DEBE SECARLO. Dicho recipiente, para que no se mezclen aguas válidas con las que no lo son. No obstante, si lo sumergió en agua apta para purificar, no necesitará secarla. RECOGER AGUA CONSAGRADA. Si lo sumerge con la intención de recoger después agua mezclada con ceniza, aunque el agua donde lo sumergió fuese de manantial... DEBERÁ SECARLO. Para que no se mezclen las aguas purificadas con las que no lo son. Más adelante se estudiará en el capítulo nueve: “Un tarro de cristal de Purificación en el que cayó agua, se invalida”. 3. KERUYÁ. Calabaza hueca y seca que se utiliza para extraer agua. SI SE SUMERGE. Este es un grado de exigencia innecesario que se le aplica a la Purificación pero no para evitar impurezas, ya que se trata de una calabaza nueva que nunca fue utilizada. LA PODRÁ USAR. Después de secarla. No se teme que haya absorbido agua y que después la expulse. NO SE PUEDE UTILIZAR. Aunque vuelva a sumergirla, porque se sospecha que expulse líquidos impuros absorbidos en su interior. Según los Rabinos, se debe ser más exigente con aguas impuras que con aquellas que no son aptas para purificar. Rabí Yehoshúa las trata por igual. Él dice: si sospechas, hazlo aquí y allí. Si no sospechas, no lo hagas ni aquí ni allí. De cualquier forma, tanto si primeramente la sumergió en aguas no aptas para purificar y la secó, como si al final, se impurificó y la sumergió en aguas aptas para purificar y la secó, no podrá agregar agua purificada, por miedo a que la expulse. La Ley no coincide con Rabí Yehoshúa. 4. UN TUBO. De caña. CORTADO. Lo preparó para introducir agua de purificación o ceniza para purificación, aunque lo haya cortado algún compañero que no es sospechoso de impureza. SUMERGIDO ENSEGUIDA. Para que quede constancia de cara a los saduceos que para ser utilizado en el servicio de la Vaca Bermeja solo es necesaria la inmersión del mismo día, y no como defienden ellos como requisito indispensable, que el sol se ponga para completar la purificación. La ablución del día se llevaba a cabo por miedo a que hubiese quedado en algún recipiente algo de saliva de cualquier profano que la haya manipulado, y que, siendo aún húmeda, impurifique dicho recipiente antes de finalizar su fabricación. Por otro lado, a pesar de que este tipo de impureza no necesita de la puesta del sol, sino solamente la inmersión, aún así, hay distinción de cara a los saduceos. En relación con la Vaca de Purificación, le dieron el mismo tratamiento que al “impurificado por contacto con cadáver en su día séptimo”, lo que ocasiona la impurificación de hombres y utensilios. Si la Vaca hubiera necesitado la puesta del sol para completar su purificación, también el tubo lo habría necesitado. RABÍ YEHOSHÚA DICE: De no ser así, no habría distinción para los saduceos, siendo que los utensilios acabados con pureza no necesitan de la puesta del sol para ser utilizado en la purificación. La Ley es como Rabí Yehoshúa. TODA PERSONA ES APTA. Toda persona, si es pura, puede arrojar la ceniza al agua, incluyendo mujeres. Está escrito en relación con la recogida de la ceniza de la Vaca: “Recogerá un hombre puro la ceniza de la Vaca, y la dejará fuera del campamento” (Números 19:9). Sobre esto explicaron: “Hombre”, engloba incluso a un extraño. “Puro”, incluye también a las mujeres. “La dejará”, aquel que posee capacidad para colocar, con lo que se excluye al sordomudo, al deficiente mental y al niño (T. Yomá 43a). Detrás de la descripción del servicio de la recogida de la ceniza, aparece, sin mediar nada más, el tema de la purificación. De aquí se aprende que quien es apto para la recogida de cenizas es apto para la purificación. CONSIDERA APTOS A LOS NIÑOS. Si es cierto que todo el que es apto para la recogida es apto para purificar, debería estar escrito “Y tomará”, en singular, hablando de la purificación, igual que está escrito en singular “Y recogerá”, hablando de la recogida, entonces, ¿por qué está escrito: “Y tomarán”, en plural? Esto enseña, que aquellos que no son aptos allí, si lo son aquí, tal como el niño. Sin embargo, la mujer no, porque está escrito: “Y pondrá él”, y no, “y pondrá ella” (Ibíd.). 5. TODO... RECIPIENTE. Incluso de madera, de hueso, o de vidrio. E incluso de estiércol, aunque no se considera recipiente con respecto a la impureza, ni según la Torá ni según los Sabios. Así está escrito: “Pondrá aguas vivas en un recipiente”, no dice “en el recipiente”, porque se trata de un recipiente indeterminado. CÁNTAROS. Son recipientes grandes de arcilla llamados en árabe “Almajguir”. En uno de sus lados se encuentra una especie de asa. A esto se refiere la Mishná según la apreciación de Rambam. NI ROCIAR. O sea, que las aguas deben estar contenidas en el recipiente a la hora del rociamiento. SOLO LOS RECIPIENTES. Con tapadera previenen del contagio de impurezas dentro de una cabaña donde haya un cadáver. LOS RECIPIENTES. Si se introduce un insecto en el hueco de un recipiente de arcilla, impurifica todo lo que se encuentre en el interior aunque no entren en contacto, a excepción de aquellos que no se impurifican, al como versa: “Todo lo que esté en su interior se volverá impuro...de cualquier alimento que sea comestible” (Levítico 11:33-34). Los alimentos sólidos y líquidos se impurifican en el interior del recipiente de arcilla, no así un recipiente de arcilla dentro de otro. SOLO PREVIENE. Es como si dijera: “Y previene”. Este estilo lo encontramos igualmente en el primer capítulo del Tratado Betsá (2a): “solo la ceniza del hornillo está preparada”. Dijo Rabá en la Guemará (8a): Y la ceniza del hornillo está preparada. 6. EL HUEVO. Cuando el alfarero comienza la fabricación de una vasija, primeramente crea una base con forma de huevo, con su correspondiente oquedad. KASHER. Para purificar con el Agua de Purificación. RABÍ YOSÉ. Lo invalida porque no lo considera como recipiente. La Halajá no sigue su opinión. DE UNA GALLINA. Se refiere a la cáscara exterior. LOS SABIOS, NO. No le dan calidad de recipiente. Así es la Halajá. 7. LA OQUEDAD. Se refiere a una roca ahuecada que se encuentra junto al manantial, en cuya pared lateral se encuentra un agujero por donde penetra el agua para dar de beber a los animales (Abrevadero). Aquí se trata de una roca adosada y tallada. TAPADERA AJUSTADA. De encontrarse en el interior de una cabaña contaminada con un cadáver, previene de impureza a lo que se encuentra en su interior aunque la tapadera no esté ajustada. Esto es así porque no se considera recipiente, sino cisterna, que previene de impureza a lo que contiene, con solo una tapadera sin ajustar, tal como se estudia en el capítulo 5 de “Aholot” (Mishná 6). NO INVALIDA. Si cayó de ella agua de lluvia hacia un Mikvé, dichas aguas no se consideran manipuladas, y por tanto, no lo invalidan. LO CIERRA. Lo tapona. RODEADAS DE AGUA. Porque el agua no contacta totalmente con toda la pared del recipiente, y el agujero inferior le anula su calidad de recipiente. El trozo de trapo no lo completa y lo convierte nuevamente en recipiente. Sin embargo, si el agujero se encuentra en un lado, sí se considera recipiente y se puede utilizar para purificar. Aquí si se considera que las aguas lo rodean del todo. UNA CORONA. Una cubeta que se separó de la roca y la volvió a pegar con cal (Un recipiente de cal tiene ley de recipiente para estos efectos), y le adosó una corona de arcilla en la parte superior, si la llenó de agua hasta tocar el borde, las aguas no sirven para purificar, ya que la corona agregada no sirve como recipiente. MUY BIEN PEGADA. Si la corona estaba muy bien pegada, hasta el punto de que si lo alza agarrando la corona, todo el recipiente se levanta, es Kasher. 8. DOS CUBETAS. Se habla aquí de piedras sueltas no de una roca pegada. Estas dos cubetas se consideran recipiente. CON UNA DE ELLAS. Colocó ceniza de la Vaca en una de ellas. LA BOCA DE UN ODRE. Se refiere a un tubo colocado en la boca del odre que sirve para llenarlo de vino o de agua. El ancho de este tubo debe ser suficiente como para que puedan introducirse en él cómodamente los dos primeros dedos de la palma de la mano. No desde el pulgar, sino los dos siguientes. REBOSABA. Como cuando las paredes que se comunican son más bajas que las otras. LA CÁSCARA DEL AJO. El ancho de la capa de agua que se desborda es igual a la anchura de la cáscara del ajo. 9. QUE SE JUNTAN. La acercó una a otra. Esta misma expresión es empleada en: “Juntan dos barriles”, en el tratado Betsá (32b). LAS AGUAS. Que se encontraran entre ambas o en alguna grieta de la cubeta. El resto de las aguas sí son consideradas puras, mientras que las de las grietas no lo son. ESCAYOLA. Es un tipo de cal, más blanca que esta. En lengua extranjera se llama “Yeso”, “Yips”.

    Guemará

    Tratado Nidá, 17b

    Dijo Ribí Shimón Bar-Yojai: Existen cuatro cosas que el Santo, Bendito Él, y que a mí no me gustan. Entrar a su propia casa sin avisar y, ni que decir tiene, a casa de un amigo. Quien, al orinar, agarra el miembro. Quien orina, estando desnudo, junto a su cama. Quien mantiene relaciones conyugales delante de un ser vivo. Le preguntó Rabí Yehudá a Ribí Shemuel: ¿Ni siquiera delante de ratones? Le respondió: ¡Listo! ¡Se refiere a mantener relaciones delante de sus sirvientes o sirvientas, tal como lo suelen hacer en casa de fulano! ¿En qué se basaban aquellos para actuar así? En el versículo que reza: “Quédense aquí junto al asno” (Génesis 22:5), refiriéndose a aquella gente que se comporta como asnos. Rabá Bar Rab Huná solía tocar campanas antes de comenzar sus relaciones conyugales. Abayé solía espantar las moscas. Rabá, espantaba mosquitos.

    TOCAR. Campanas colgadas en el mosquitero alrededor de su cama. Las hacía sonar para alertar y alejar a los de la casa antes de mantener relaciones conyugales. ESPANTAR MOSCAS. Las espantaba para no mantener relaciones delante de seres vivos. MOSQUITOS. Zancudos.

    Zóhar

    Yitró 88b

    (Comentario sobre el Zóhar)

    Preguntó Rabí Elazar a su padre: ¿Estos banquetes cómo se realizan? Le respondió: Sobre la noche del Shabat se dijo: “Te haré cabalgar sobre las alturas de la tierra” (Isaías 58:14). Durante esta noche se bendice a la Reina y a todo el “Campo de Manzanas”. También, la mesa de los hombres. A éste, se le adosa otra alma, siendo dicha noche de alegría para la Reina. Cada cual debe alegrarse con su alegría y comer del banquete de la Reina. Respecto del segundo banquete del día del Shabat está escrito: “Entonces te regocijarás junto al Eterno” (Ibíd.). ¡Por supuesto, junto al Eterno! En esa hora se revela el Santo Antiquísimo y todos los mundos se alegran. A la alegría y a la perfección del Antiquísimo contribuimos, y, por supuesto, ¡este es Su banquete! En relación con el tercer banquete de Shabat se dijo: “Y te daré de comer de la heredad de tu Patriarca Yaacob” (Ibíd.). Este es el banquete del Zeer Anpín en la perfección, y, el resto de los seis días, de él toman su bendición. Cada cual debe regocijarse en Su banquete y completar estas tres comidas, que son las comidas de la Fe íntegra de la sagrada simiente de Israel. Esta fe suprema es de ellos y no de los pueblos idólatras. Por esta razón está dicho: “Entre Mí y los Hijos de Israel” (Éxodo 31:17).

    Halajá

    Rambam, Leyes de Shabat, capítulo 5

    1. Objetos, que por ser muy parecidos no pueden distinguirse a menos que sean revisados meticulosamente, no deben ser acercados a la luz de la lámpara para diferenciarlos, por miedo a que se olvide y la incline. Por dicha razón, el encargado del Bet-Hakeneset, si no trabaja allí de forma usual, no deberá revisar los vasos y bandejas a la luz de la vela, pues no sabría reconocerlos bien. No importa si la lámpara es de aceite de oliva o de queroseno, que posee mejor luz. No obstante, si el encargado es fijo, puede revisar tales vasos o bandejas al no necesitar gran fijación. Si la lámpara utilizaba aceite de oliva, no se le alienta a revisar (aunque esté permitido), no sea que saque parte de él. 2. Si la lámpara se encuentra tras la puerta, procurará no cerrarla y abrirla como hace habitualmente, porque podría apagarla. Por tanto, tendrá cuidado al abrir y al cerrar. Igualmente, está prohibido abrir la puerta si se encuentra frente al hogar, por miedo a provocar que el aire sople en él, aunque en aquel momento no había viento. Por otro lado, se permite colocar una lámpara sobre un árbol arraigado en el suelo, sin temer de nada.

    Moral

    Yosef Le’Jok, Compendio

    Así como el enfermo se cuida y se apresura en encontrar un médico experto, rogándole que lo cure y prometiéndole mucho dinero y honrándole en demasía, y todo por lo mucho que teme a su enfermedad, cuánto más debe preocuparse por curar su alma, cuya enfermedad es verdaderamente importante y cuya dolencia es verdaderamente poderosa. Procurará su curación a través de las personas instruidas y de sus libros. Así sanará lo que pervirtió, enmendará lo que destruyó, y reunirá lo que desparramó. Deberá sentirse afligido y escuchará la voz de sus maestros. Después de reflexionar en las enfermedades de su alma, en todas las máculas con las que la manchó, irá a contarle sus dolencias al sanador de las almas. Él lo conducirá por senderos justos y se apresurará en someterse al régimen de las Leyes del arrepentimiento tanto como le exijan. El Eterno, que es bondadoso, y que ansía el arrepentimiento, lo ayudará a causa del Honor de Su Nombre. Quien habita en los cielos será su ayuda y hará brillar Su Luz para él.
