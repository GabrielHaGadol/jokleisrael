וּבְלֵילְיָא דְשַׁבַּתָּא מְזָרְזֵי גַרְמַיְיהוּ בְּזִוּוּגָא דִילְהוֹן מִשּׁוּם דְּיַדְעֵי רָזָא עִלָּאָה בְּשַׁעְתָּא דְמַטְרוֹנִיתָא אִזְדַוְוגַת בְּמַלְכָּא. וְאִינוּן חַבְרַיָּיא דְּיַדְעִין רָזָא דָא מְכַוְּונִין לִבַּיְיהוּ לִמְהֵימְנוּתָא דְמָארֵיהוֹן וּמִתְבָּרְכָאן בְּאִבָּא דְמֵעֵיהוֹן בְּהַהוּא לֵילְיָא וְדָא הוּא דִכְתִיב אֲשֶׁר יִשְׁמְרוּ כְּמָה דְאַתְּ אָמַר וְאָבִיו שָׁמַר אֶת הַדָּבָר. וְאִקְרוּן סָרִיסִים וַדַּאי בְּגִין לְחַכָּאָה לְשַׁבַּתָּא לְאַשְׁכָּחָא רַעֲוָא דְמָארֵיהוֹן דִּכְתִיב וּבָחֲרוּ בַּאֲשֶׁר חָפַצְתִּי. מָאי בַּאֲשֶׁר חָפַצְתִּי דָּא זִוּוּגָא דְמַטְרוֹנִיתָא וּמַחֲזִיקִים בִּבְרִיתִי כֹּלָּא חָד בִּבְרִיתִי סְתָם. זַכָּאָה חוּלְקֵיהּ דְּמָאן דְּאִתְקַדַּשׁ בִּקְדוּשָׁה דָא וְיֵדַע רָזָא דָא. תָּא חֲזֵי כְּתִיב שֵׁשֶׁת יָמִים תַּעֲבֹד וְעָשִׂיתָ כָּל מְלַאכְתֶּךָ וְיוֹם הַשְּׁבִיעִי שַׁבָּת לַה' אֱלֹהֶיךָ וגו' כָּל מְלַאכְתֶּךָ בְּאִינוּן שִׁתָּא יוֹמֵי עֲבִידְתַּיְיהוּ דִבְנֵי נְשָׁא. וּבְגִין הָאי מִלָּה לָא מִזְדַוְּוגֵי חַבְרַיָּיא בַּר בְּזִמְנָא דְלָא יִשְׁתַּכַּח מֵעֲבִידְתַּיְיהוּ דִבְנֵי נְשָׁא אֶלָּא עֲבִידְתֵּיה דְּקב''ה. וּמָאי עֲבִידְתֵּיהּ זִוּוּגָא דְמַטְרוֹנִיתָא לְאַפָּקָא נִשְׁמָתִין קַדִּישִׁין לְעָלְמָא וּבְגִין כָּךְ בְּהָאי לֵילְיָא חַבְרַיָּיא מִתְקַדְּשֵׁי בִקְדוּשָׁה דְמָארֵיהוֹן וּמְכַוְונֵי לִבַּיְיהוּ וְנַפְקֵי בָּנֵי מַעֲלֵי בְּנִין קַדִּישִׁין דְּלָא סָאטָן לִימִינָא וְלִשְׂמָאלָא בְּנִין דְּמַלְכָּא וּמַטְרוֹנִיתָא. וְעַל אִלֵּין כְּתִיב בָּנִים אַתֶּם לַה' אֱלֹהֵיכֶם. לַה' אֱלֹהֵיכֶם וַדַּאי. בְּגִין דְּאִלֵּין אִקְרוּן בְּנִין דִּילֵיהּ בְּנִין לְמַלְכָּא וּלְמַטְרוֹנִיתָא וְהָא דַעְתַּיְיהוּ דְחַבְרַיָּיא דְּיַדְעִין רָזָא דָא בְדָא מִתְדַבְּקָן. וּבְגִין כָּךְ אִקְרוּן בְּנִין לְקוּדְשָׁא בְרִיךְ הוּא. וְהַנֵּי אִינוּן דְּעָלְמָא מִתְקַיְימָא בְגִינַיְיהוּ. וְכַד סָלִיק עָלְמָא בְדִינָא אִסְתַּכַּל קב''ה בְּאִינוּן בְּנוֹי וּמְרַחֵם עַל עָלְמָא. וְעַל דָּא כְתִיב כֻּלֹה זֶרַע אֱמֶת. זֶרַע אֱמֶת וַדַּאי:     תרגום הזוהר

    כֹּה אָמַר ה' לַסָּרִיסִים. מִי הֵם סָרִיסִים, אֵלּוּ הֵם הַחֲבֵרִים הָעוֹסְקִים בַּתּוֹרָה וּמְסָרְסִים עַצְמָם כָּל שֵׁשֶׁת יְמֵי הַשַּׁבָּת וְעוֹסְקִים בַּתּוֹרָה, וּבְלֵיל שַׁבָּת מְזָרְזִים עַצְמָם בַּזִּוּוּג שֶׁלָּהֶם, מִשּׁוּם שֶׁיּוֹדְעִים סוֹד הָעֶלְיוֹן, בְּשָׁעָה שֶׁהַמַּטְרוּנִיתָא מִתְחַבֶּרֶת בַּמֶּלֶךְ. וְאֵלּוּ הַחֲבֵרִים הַיּוֹדְעִים סוֹד הַזֶּה, מְכַוְּנִים אֶת לִבָּם לֶאֱמוּנַת אֲדוֹנָם, וּמִתְבָּרְכִים בִּפְרִי בִּטְנָם בַּלַּיְלָה הַהוּא. וְזֶהוּ שֶׁכָּתוּב, אֲשֶׁר יִשְׁמְרוּ, כְּדִכְתִיב, וְאָבִיו שָׁמַר אֶת הַדָּבָר. וְנִקְרָאִים וַדַּאי סָרִיסִים, מִשּׁוּם שֶׁמְּחַכִּים עַד הַשַּׁבָּת לִמְצֹא רָצוֹן שֶׁל אֲדוֹנָם, שֶׁכָּתוּב, וּבָחֲרוּ בַּאֲשֶׁר חָפָצְתִּי. מַהוּ בַּאֲשֶׁר חָפָצְתִּי. זֶהוּ הַזִּוּוּג שֶׁל מַטְרוּנִיתָא. וּמַחֲזִיקִים בִּבְרִיתִי, הַכֹּל אֶחָד, בִּבְרִיתִי סְתָם, אַשְׁרֵי חֶלְקוֹ שֶׁל מִי שֶׁמִּתְקַדֵּשׁ בִּקְדֻשָּׁה זוֹ, וְיוֹדֵעַ סוֹד הַזֶּה. בֹּא וּרְאֵה, כָּתוּב, שֵׁשֶׁת יָמִים תַּעֲבוֹד וְעָשִׂיתָ כָל מְלַאכְתֶּךְ וְיוֹם הַשְּׁבִיעִי שַׁבָּת לַה' אֱלֹקֶיךְ וְגוֹ'. כָּל מְלַאכְתֶּךְ, שֶׁבְּאֵלּוּ שִׁשָּׁה יָמִים מַעֲשֵׂיהֶם שֶׁל בְּנֵי אָדָם, וּבִשְׁבִיל דָּבָר זֶה אֵינָם מִזְדַוְּגִים הַחֲבֵרִים רַק בַּזְּמָן שֶׁלֹּא נִמְצָא מַעֲשֵׂיהֶם שֶׁל בְּנֵי אָדָם אֶלָּא מַעֲשָׂיו שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא. דְּהַיְנוּ בַּשַּׁבָּת, וּמַהוּ מַעֲשָׂיו, הוּא הַזִּוּוּג עִם מַטְרוּנִיתָא, לְהוֹצִיא נְשָׁמוֹת קְדוֹשׁוֹת לְעוֹלָם. וּמִשּׁוּם זֶה בַּלַּיְלָה הַזֶּה, הַחֲבֵרִים מִתְקַדְּשִׁים בִּקְדֻשַּׁת אֲדוֹנָם, וּמְכַוְּנִים לִבָּם, וְיוֹצְאִים בָּנִים טוֹבִים בָּנִים קְדוֹשִׁים שֶׁאֵינָם סָרִים יָמִין וּשְׂמֹאל, בָּנִים שֶׁל הַמֶּלֶךְ וְהַמַּטְרוּנִיתָא. וְעַל אֵלּוּ כָּתוּב, בָּנִים אַתֶּם לַה' אֱלֹקֵיכֶם. לה' אֱלֹקֵיכֶם וַדַּאי, מִשּׁוּם שֶׁאֵלּוּ נִקְרָאִים בָּנִים שֶׁלּוֹ, בָּנִים שֶׁל הַמֶּלֶךְ וּמַטְרוּנִיתָא. וַהֲרֵי דַּעַת הַחֲבֵרִים הַיּוֹדְעִים סוֹד הַזֶּה, מִתְדַבְּקִים בָּזֶה, וּמִשּׁוּם זֶה נִקְרָאִים בָּנִים לְהַקָּדוֹשׁ בָּרוּךְ הוּא. וְאֵלּוּ הֵם שֶׁהָעוֹלָם מִתְקַיֵּם בִּשְׁבִילָם. וּכְשֶׁהָעוֹלָם עוֹלֶה בַּדִּין, מִסְתַּכֵּל הַקָּדוֹשׁ בָּרוּךְ הוּא בְּבָנָיו אֵלּוּ, וּמְרַחֵם עַל הָעוֹלָם. וְעַל זֶה כָּתוּב, כֻּלּוֹ זֶרַע אֱמֶת, זֶרַע אֱמֶת וַדַּאי:     א. בְּרָכוֹת אֵלּוּ עִם שְׁאָר כָּל הַבְּרָכוֹת הָעֲרוּכוֹת בְּפִי כָל יִשְׂרָאֵל עֶזְרָא הַסוֹפֵר וּבֵית דִּינוֹ תִּקְנוּם וְאֵין אָדָם רַשָּׁאי לִפְחוֹת מֵהֶם וְלֹא לְהוֹסִיף עֲלֵיהֶם. מָקוֹם שֶׁהִתְקִינוּ לַחֲתוֹם בְּבָרוּךְ אֵינוֹ רַשָּׁאי שֶׁלֹּא לַחֲתוֹם וּמָקוֹם שֶׁהִתְקִינוּ שֶׁלֹּא לַחֲתוֹם אֵינוֹ רַשָּׁאי לַחֲתוֹם. מָקוֹם שֶׁהִתְקִינוּ שֶׁלֹּא לִפְתּוֹחַ בְּבָרוּךְ אֵינוֹ רַשָּׁאי לִפְתּוֹחַ מָקוֹם שֶׁהִתְקִינוּ לִפְתּוֹחַ אֵינוֹ רַשָּׁאי שֶׁלֹּא לִפְתּוֹחַ כְּלָלוֹ שֶׁל דָּבָר כָּל הַמְּשַׁנֶּה מִמַּטְבֵּעַ שֶׁטָּבְעוּ חֲכָמִים בִּבְרָכוֹת הֲרֵי זֶה טוֹעֶה וְחוֹזֵר וּמְבָרֵךְ כַּמַּטְבֵּעַ. וְכָל שֶׁאֵינוֹ אוֹמֵר אֱמֶת וְיַצִיב בְשַׁחֲרִית וֶאֱמֶת וֶאֱמוּנָה בַּעֲרָבִית לֹא יָצָא יְדֵי חוֹבָתוֹ: ב. הִקְדִּים בְּרָכָה שְׁנִיָּה לִבְרָכָה רִאשׁוֹנָה בֵּין בַּיּוֹם בֵּין בַּלַּיְלָה בֵּין לְפָנֶיהָ בֵּין לְאַחֲרֶיהָ יָצָא לְפִי שֶׁאֵין סֵדֶר בִּבְרָכוֹת. בְּשַׁחֲרִית פָּתַח יוֹצֵא אוֹר וְסִיֵּים מַעֲרִיב עֲרָבִים לֹא יָצָא פָּתַח בְּמַעֲרִיב עֲרָבִים וְסִיֵּים בְּיוֹצֵר אוֹר יָצָא וּבָעֶרֶב פָּתַח בְּמַעֲרִיב עֲרָבִים וְסִיֵּים בְּיוֹצֵר אוֹר לֹא יָצָא פָּתַח בְּיוֹצֵר אוֹר וְסִיֵּים בְּמַעֲרִיב עֲרָבִים יָצָא שֶׁכָּל הַבְּרָכוֹת הוֹלְכוֹת אַחַר חֲתִימָתָן:                                                                                                                                                                                                                                                                                                           ן

La Escritura hace referencia a Yitró otorgándole un título honroso, como diciendo que era “el suegro del rey”. 18. נָבֹל תִּבֹּל / CIERTAMENTE TE DESGASTARÁS. Esta expresión debe ser entendida tal como la traduce el Targum.  Esta expresión significa “marchitarse”.  Significa lo mismo que flestre en francés antiguo, y es similar en significado a los términos: “Y la hoja está marchita [נבל]”; “como se marchita [כנבול] una hoja de la viña”, la cual se marchita por efecto del sol y de la escarcha, por lo que su fuerza se debilita y se desgasta. גַּם אַתָּה  / TANTO TÚ COMO... El peso de ello es superior a lo que tu fuerza puede soportar.  19. אִיעָצְךָ וִיהִי אֱלֹהִים עִמָּךְ / TE ACONSEJARÉ Y QUE DI-S ESTÉ CONTIGO. Es decir, que esté contigo en este consejo que te doy.  Yitró quiso decirle: “Vé y consulta con el Omnipotente con respecto a lo que te digo”.  הֱיֶה אַתָּה לָעָם מוּל הָאֱלֹהִים / SÉ TÚ PARA EL PUEBLO [INTERMEDIARIO] ANTE DI-S. Es decir, sé tú un emisario y un intermediario entre ellos y el Omnipresente, y sé tú el que pregunte los juicios de Él para el pueblo.  אֶת הַדְּבָרִים / LOS ASUNTOS. Es decir, hombres ricos, para que no tengan necesidad de adular a otros ni de hacer favores.  אַנְשֵׁי אֱמֶת / HOMBRES DE VERDAD. Ellos son los que inspiran confianza, quienes son dignos de que se confíe en sus palabras. Escoge a gente así, porque gracias a esa cualidad sus palabras serán escuchadas.  שֹׂנְאֵי בָצַע / QUE ABORREZCAN LA GANANCIA. Es decir, que odien que su dinero esté en litigio.  Esto se refiere a la misma idea que expresamos en el Talmud: “Cualquier juez a quien se le extrae dinero a través de un litigio no es un juez.”  שָׂרֵי אֲלָפִים / JEFES DE MILLARES.     Tratado Babá Metsiá, capítulo 7

    (1) Quien contrata trabajadores (sin especificar condiciones) y les exige que madruguen y que trabajen hasta tarde, si en aquel lugar no se acostumbra a madrugar y salir tarde, no los puede obligar a hacerlo. Donde se acostumbre a darles de comer, les dará. Si es la costumbre dar entremeses, lo hará. Todo depende de la costumbre del país. Sucedió cierta vez que Rabí Yojanán Ben-Matyá le dijo a su hijo: Ve a contratar trabajadores. Fue y acordó con ellos darles de comer. Al regresar junto a su padre, éste le dijo: Hijo mío, ahora, aunque les prodigues tanta comida como la que se servía en los banquetes del rey Shelomó en su mejor época, no habrás cumplido con tu obligación hacia ellos, ya que son descendientes de Abraham, Yitzjak y Yaacob. Por tanto, antes de que comiencen su trabajo ve y diles que te comprometes con ellos solo a pan y legumbres. Rabí Shimón Ben-Gamliel aclaró: No había necesidad de decirles nada, ya que todo depende de la costumbre del país. (2) Los siguientes podrán comer de los frutos que trabajan, pues así lo contempla la Torá, en los siguientes casos: Cuando se trata de frutos no cosechados, después de terminar la labor. Si ya estaban recolectados, sólo podrá comer de ellos antes de estar listos para ser diezmados. Lo antedicho se refiere únicamente a alimentos que nacen de la tierra. Los frutos siguientes no podrán ser ingeridos por los trabajadores: Quien trabaja con la cosecha, antes de ser recolectada. Si trabaja con frutos cosechados, después de estar listos para diezmar. Tampoco podrá comer los alimentos que no brotan de la tierra. (3) Tanto si trabaja con las manos pero no con los pies, o con los pies, sin utilizar las manos, incluso que sólo utilice su espalda para realizar la labor, podrá comer de los frutos que trabaja. Rabí Yosé Bar-Yehudá dijo: Sólo podrá comer si trabaja tanto con los pies como con las manos. (4) Si lo contrataron para trabajar con higos no podrá comer uvas. Con uvas, no podrá comer higos. Podrá abstenerse de comer hasta que llegue al lugar donde se encuentre las mejores frutas y comerlas. Siempre que pueda comer lo hará sólo mientras trabaja. Sin embargo, para evitarle al propietario pérdidas, dijeron los Sabios que los obreros podrán comer solamente entre fila y fila, o cuando regresan del lagar. Un asno puede comer de su carga mientras que no la descargó. (5) El trabajador podrá comer calabacinos aunque valga un dinar, y dátiles, también aunque valga un dinar. Rabí Elazar Ben-Jismá dijo: El obrero no podrá comer frutos con un valor mayor que su salario. No obstante, el resto de los Sabios lo permiten, pero le educan a que no se comporte como un hambriento. Sin embargo, si se comporta así le cerrarán la puerta en la cara. (6) El obrero podrá prescindir de su derecho a comer, o de su hijo o hija, mayores, o de su siervo o sierva adultos, o de su esposa, con su consentimiento, pues sabrán entenderlo. Sin embargo, no podrá prescindir del derecho a comer de su hijo o hija pequeños, o de su siervo o sierva pequeños, o de su animal, ya que éstos no tienen suficiente juicio para entenderlo (, y por tanto, no podrán consentirlo). (7) Los trabajadores contratados para recolectar frutos en el cuarto año, no podrán comer de ellos. Si previamente no les advirtió (que los frutos pertenecían al cuarto año), deberá redimirlos para que los puedan comer. Si se despegaron los panes de higo o se abrieron los barriles, no podrán comer de ellos. Si no les advirtió con anterioridad, retirará el diezmo y les permitirá a los obreros que coman de lo que trabajan. (8) Los que trabajan como vigilantes de frutas podrán comer de ellas puesto que así se acostumbró, aunque, la Torá no lo permite. Existen cuatro tipos de vigilantes: El no asalariado, quien pide prestado, el asalariado y quien paga un alquiler. El vigilante que no cobra su trabajo debe jurar por todo lo que le ocurra. El que pide prestado, pagará por todo. El vigilante remunerado y el que paga un alquiler, deberán jurar por los casos de daño, captura o muerte, mientras que deberán pagar por los casos de pérdida o robo. (9) El ataque de un solo lobo no se computa como inevitable. El de dos, sí. Rabí Yehudá dijo: Cuando los lobos aparecen en manada, aún el ataque de uno de ellos no se puede repeler. El ataque de dos perros no es inevitable. Yadúa el babilonio dijo en nombre de Rabí Meír: Si atacan ambos por un solo flanco, se podrá repeler. Por dos flancos distintos, será considerado inevitable. El ataque realizado por asaltantes armados se considera no evitable. El ataque realizado por un león, oso, tigre, pantera o serpiente, no son evitables. ¿En qué caso? Cuando se acercan por sí mismos, no obstante, si el rebaño se acerca a estos animales o a los asaltantes, no será computado como caso inevitable (sino como negligencia). (10) Si el animal a cuidar muere por causas naturales, se llama caso inevitable. Si lo descuida y muere, no será conceptuado como inevitable. Si después de escapar y escalar montes rocosos, cae y muere, se llamará inevitable. Sin embargo, si hace subir al animal hasta allí, y cae y muere, no será computable como muerte inevitable. Un vigilante no asalariado puede plantear la condición de no tener que jurar obligatoriamente en caso de siniestro, y el que pide prestado, puede hacer lo propio para eludir el pago igualmente en caso de siniestro. También, el vigilante asalariado y el que paga un alquiler podrán condicionar no tener que pagar o jurar en caso de que ocurra algo a lo que vigila. (11) Cualquier cláusula que condiciona lo que la Torá ordena se considera nula. No solo eso, sino que cualquier clausula condicionante queda anulada si previamente se inició alguna acción. No obstante, si dicha condición puede materializarse después de cualquier acción emprendida, mantendrá su vigencia.

    1. NO LOS PUEDE OBLIGAR. Aunque les aumente el salario por encima del resto de los trabajadores. Por tanto, no les podrá decir: Esto que os aumenté es para que madruguéis y salgáis más tarde del trabajo. Ante esto los trabajadores podrán argüir: Esto que nos quieres pagar no es por las horas extras trabajadas sino por la calidad de nuestro trabajo.  ENTREMESES. Guarnición para el plato principal.  LA COSTUMBRE DEL PAÍS. Esto incluye aquellos lugares donde se acostumbró que los obreros acudan a casa del contratista para comer y beber, antes de comenzar el trabajo por la mañana.  En este caso, si el patrón les dice: Madruguen para trabajar en el campo que yo les llevaré allí la comida, ellos le podrán decir: Comeremos aquí en la casa antes de salir a trabajar, puesto que así es la costumbre en este país.  SUCEDIÓ… La Guemará comenta que el texto aparece incompleto, y que el siguiente es el correcto: “Si concreta con los obreros darles de comer, siendo que tácitamente lo debe hacer por ser la costumbre del país, esta condición supone un aumento sobre lo usual. Cierta vez ocurrió que Rabí Yojanán…  EN SU MEJOR ÉPOCA. O sea, cuando reinaba, pues vivió dos épocas en su vida, una como súbdito y otra como rey.  DESCENDIENTES DE ABRAHAM… Donde los banquetes de Abraham superaban a los del rey Shelomó, tal como vimos que Abraham preparó tres vacas para dar de comer solamente a tres personas.  Sin embargo, con respecto a los banquetes del rey Shelomó está escrito: “Yehudá e Israel son numerosos como la arena de la orilla del mar”.  ANTES DE QUE COMIENCEN. Mientras no comiences, solo hay palabras, después, será imposible retractarse de lo dicho.  RABÍ SHIMÓN… La Halajá concuerda con su parecer. 2. LOS SIGUIENTES. Obreros.  CONTEMPLA LA TORÁ. Que podrán comer de lo que trabajan.  DESPUÉS DE TERMINAR. La cosecha, tal como versa: “No lo pondrás en tu recipiente” (Deuteronomio 23:25), es decir, sólo cuando los puedas colocar en los recipientes del patrón los podrás colocar en el tuyo, o sea, sólo cuando estén cosechados.  RECOLECTADOS. Si se ocupa en frutos ya recolectados podrá comer de ellos mientras no esté listo para diezmarlos, en caso de que estén sujeto al diezmo.  En otros, podrá comer de ellos antes de estar listos para apartar de ellos la Jalá, en caso de que lo requiera.  Sobre esto está escrito: “No pondrás bozal al buey mientras trilla” (Ibíd. 25:4).  Siendo que no se dijo: “No trillarás mientras tenga el bozal puesto”, entendemos que la Torá quiso comparar al que pone el bozal con el animal al que se lo ponen, y viceversa.  Al igual que el buey, al cual le colocan el bozal, sólo come de lo que está despegado de la tierra mientras trabaja, así, quien pone el bozal al buey, el hombre, podrá comer, mientras trabaja, de lo que ya está recolectado, y viceversa.  Además, al igual que trillar, trata de plantas que crecen en la tierra, que mientras no se acabó el trabajo necesario para poder diezmar está permitido comer de ello, todo lo que crece de la tierra y aún no quedó listo para diezmar, también el obrero podrá comer de ello.  Esto excluye al ordeñador y al quesero, que trabaja en productos que no crecen de la tierra. Igualmente se excluye a los que separan los dátiles e higos pegados, pues en este caso, ya están listos los productos para ser diezmados. Por otro lado, quedan excluidos también los que se ocupan de arrancar los ajos y cebollas pequeños que nunca crecerán, para dejar espacio a los que sí tienen posibilidades de hacerlo, porque este tipo de trabajos aún no son concluyentes para que puedan ser diezmados. En todos estos casos el trabajador no podrá comer de ellos.  3. TANTO CON LOS PIES… Al igual que el buey trabaja con los pies y con las manos, el trabajador deberá trabajar con los pies y con las manos, pues como vimos, fueron comparados quien pone el bozal a quien se la ponen. La Halajá no concuerda con la opinión de Rabí Yosé hijo de Rabí Yehudá.  4. PARA EVITARLE PÉRDIDAS. Para que no pierdan tiempo de su trabajo mientras comen. ENTRE FILA Y FILA. Después de finalizar el trabajo de una fila y antes de comenzar la siguiente. Aunque ese tiempo no es tiempo de trabajo, el patrón prefiere que lo hagan en aquel momento. UN ASNO… Mientras camina va comiendo de la carga que porta antes de bajarla.  5. UN DINAR. Cada unidad.  MAYOR QUE SU SALARIO. El texto bíblico dijo: “Según tu deseo” (Deuteronomio 23:25). Al igual que cada cual se sacrifica en su trabajo según el salario que recibe, incluyendo casos en los que peligra su vida, así comerá de acuerdo con dicho salario.  LE EDUCAN. Le enseñan la mejor forma de comportarse. Sin embargo, el primer exponente de la Mishná no está de acuerdo con esta opinión mayoritaria de los sabios. La Ley concuerda con los sabios.  CERRARÁN LA PUERTA. Se abstendrán de contratarlo.  6. PODRÁ PRESCINDIR. Prefiere recibir dinero en lugar de la comida.  DE SU DERECHO. Por sí mismo.  SABRÁN ENTENDERLO. Y darán su consentimiento.  7. CUARTO AÑO. Los frutos de los árboles durante el cuarto año de su plantación. Éstos solo pueden ser ingeridos en Yerushalayim, o bien, redimirlos con dinero, para llevarlo a Yerushalayim.  PANES DE HIGO. Si los panes de higo se deshacen y contrata nuevos trabajadores para que los vuelvan a componer, o si se abrieron barriles de vino y contrata obreros para que los cierren.  NO PODRÁ COMER. Siendo que estaban listos para ser diezmados está prohibido ingerirlos. En hebreo estos frutos se denominan “Tébel”. 8. VIGILANTES DE FRUTAS. Se refiere a los vigilantes de los lagares, graneros y frutos cosechados. Sin embargo, a los cuidadores de huertos y jardines no les permiten comer, ni las leyes del país ni las de la Torá. Este tipo de guardián no tienen el mismo tratamiento que los trabajadores manuales. LEYES DEL PAÍS. Aquellas normas que se practican asiduamente.  JURAR POR TODO. Ante todos aquellos siniestros que obligan a pagar al resto de los guardianes. Deberá jurar que ocurrió tal como él afirme y quedará exento ante cualquier pago.  PAGARÁ POR TODO. O sea, por robo, pérdida o cualquier siniestro fortuito.  REMUNERADO… Todas estas leyes las aprendemos en el texto de la Perashá Veele Hamishpatim.  La primera de ellas: “Si un hombre entrega al prójimo dinero u objetos” (Éxodo 22:6), habla del vigilante no remunerado, pues así se le excluye tanto del robo como de la pérdida. La segunda, “Si un hombre entrega a su amo un asno o un toro” (Ibíd. 9), se refiere al vigilante remunerado. Allí se le obliga a pagar tanto el robo como la pérdida, puesto que está escrito: “Si robar le roba, pagará a su dueño” (Ibíd. 11). De aquí se aprende qué pasa cuando le roba, pero ¿qué hay de la pérdida? De la duplicidad “robar - roba”, se entiende que pagará en todos los casos. También se puede aprender utilizando la inferencia lógica “a fortiori”, así: Si el robo, que es casi inevitable, se castiga pagándolo, cuánto más, la pérdida, que es casi como una negligencia. El que alquila, siendo que él también disfruta de lo que guarda, tiene el mismo tratamiento que el vigilante remunerado. La Ley sobre el que pide prestado, aparece en tercer lugar: “Si un hombre pide (un objeto o un animal) prestado a su prójimo, y se rompe o muere, no estando su dueño con él, pagar pagará” (Ibíd. 13).  9. UN SOLO LOBO. Por tanto, el vigilante remunerado y el que alquila deben pagar, tal como versa: “El animal devorado no pagará” (Ibíd. 12), de lo que se deduce que hay animales devorados que se deben pagar y otros que no se deben pagar.  EN MANADA. Cuando una bestia se encuentra junto a su manada es probable que ataque individualmente a una persona. La Ley no concuerda ni con Rabí Yehudá ni con Yadúa el babilonio.  ASALTANTES. Aunque se trate de uno solo se conceptúa como irremediable.  10. SI LO DESCUIDA. La dejó de hambre, o la situó debajo del sol durante el verano, o pasando frío durante el invierno.  MONTES ROCOSOS. Se escapó y se subió a altas y escabrosas cimas.  NO TENER QUE JURAR. Aunque no esté condicionado algo que prescribe la Torá, sino que le dice: No puedo ser tu vigilante, sino atendiendo a las siguientes condiciones. Un vigilante no entra en funciones mientras que no arrastró mínimamente al animal a custodiar. Si aquel que tiró del animal especificó previamente que en caso de siniestro no jurará, estará sujeto a leyes especiales de acuerdo con sus condiciones y no como un vigilante corriente.  11. CUALQUIER CLAUSULA... Esta Mishná refleja enteramente la opinión de Rabí Meir, quien enseña que quien condiciona algo en contra de lo que especificó la Torá, incluso en temas económicos, no tendrá vigencia. No obstante, así no se dictaminó la Ley, sino que en temas económicos se permite establecer condiciones previamente.  ALGUNA ACCIÓN. Si adelanta en el tiempo una acción a la condición que desea imponer, la condición no tendrá vigencia. Esto no se parece al episodio de los Hijos de Gad y los Hijos de Reubén. La Torá dijo: “Si los Hijos de Gad y los Hijos de Reubén cruzan el Jordán junto a ustedes… y la tierra es conquistada, entonces, les daréis por heredad la tierra de Guilad” (Números 32:29), donde la condición precedió a la acción.  QUEDA ANULADA. Lo que hizo, hecho está, a pesar de que no cumplió su condición aquel que la estableció.  MATERIALIZARSE DESPUÉS. Entonces, la condición precede a la acción.  MANTENDRÁ SU VIGENCIA. Si no la pudiera materializar, entonces la condición pierde su validez, frente a la acción emprendida. Aparentemente, solo pronunció palabras vanas cuando no tenía ninguna intención de cumplir lo que pactó en un principio. Quería ganar tiempo o eludir su responsabilidad.

    Guemará

    Tratado Babá Metsiá 71a

    Rabí Yosef aprendió: “Cuando prestes dinero a (alguien de) Mi pueblo, a los pobres que están contigo” (Éxodo. 22:24). “Mi pueblo” y un pagano, la prioridad es de quien pertenece a "Mi pueblo". Entre un pobre y un rico, se prefiere al pobre. Tus pobres y los pobres del pueblo primero son “tus pobres”. Entre los pobres de tu ciudad y los de otro pueblo, tienen prioridad los pobres de tu pueblo. Dijo un Maestro: ¿“Mi pueblo” y un pagano, tiene prioridad “Mi pueblo”? ¡Esto es obvio! Rabí Najmán contestó: Huná me dijo que era necesario especificarlo para el caso en que se prestó al pagano con interés, y al israelita sin él. Se enseñó en una Baraita: Rabí Yosé dijo: ¡Ven y observa qué gran ceguera padecen los que prestan con interés! Si un hombre llama a su vecino malvado, entrará en disputas con él, como si lo hubiese querido pegar o matar. Sin embargo, ellos mismos, (cuando prestan con interés en contra de lo que estipula la Torá) llevan testigos, un escribano, pluma y tinta, y registran y firman: Este fulano (el prestamista) ha renegado del Di-s de Israel (es decir, se llaman a sí mismos “malvados” sin percatarse de ello). Se estudió en una Baraita: Rabí Shimón ben Elazar dice: Sobre quien dispone de dinero y lo presta sin interés, dice la Escritura: “Su dinero no dio con interés, y no tomó soborno contra un inocente. Quien así actúa nunca tropezará” (Salmos 15:5). De aquí se desprende que quien presta con interés, pierde su riqueza. Pero ¿acaso no vemos que hay personas que pierden su riqueza sin prestar con intereses? Rabí Elazar responde: Éstas últimas caen y se recuperan, mientras que las anteriores caen y no se recuperan. Nuestros Maestros enseñaron: “No tomarás de él interés o usura” (Levítico 25:36). No obstante puedes actuar con él como garante. ¿Con quién? ¿Garante de un Israelita? ¡Pero aprendimos: Los siguientes violan preceptos pasivos: El prestatario, el prestamista, el fiador y los testigos! Entonces, ¿con un pagano? Sí, se refiere a ser garante ante un gentil (que prestó a un judío con interés) ¡Esto es imposible!, ya que según la Ley no judía, el prestamista va directamente a cobrar al garante y éste, a su vez, se preocupa de cobrar al prestatario, por lo que incurriría en una falta, de todas formas, al cobrarle la deuda con el interés establecido. Contestó Rab Shéshet: Se trata de un caso en el que el gentil se compromete previamente a actuar según la Ley Judía, o sea, primero intentará cobrar al prestatario y después, si no paga éste, irá al garante. Si se trata de un gentil que actúa según la Ley Judía, ¡tampoco él debería cobrar interés! Respondió finalmente Rab Shéshet: Se trata de un caso en que aceptó esta parte de la ley pero no la otra.

    Rashí

    CUANDO PRESTES DINERO A MI PUEBLO. Cuando un pagano y uno de mi pueblo vienen a pedir prestado, quien pertenece a mi pueblo tiene prioridad. TUS POBRES Y LOS POBRES DEL PUEBLO. Persona pobre que es pariente del prestamista frente a otros pobres. PRIMERO SON “TUS POBRES”. Como versa el versículo: "Mi pueblo", quien es parte de ti. ENTRARÁ EN DISPUTAS. Comenzará a pelearse con él como si lo hubiera golpeado o quisiera matarle. Por otro lado, escuché la siguiente explicación; Se refiere a que puede hacerle la competencia en su trabajo para intentar que disminuyan sus ingresos. Sin embargo, a mí me parece esta explicación difícil de aceptar porque, ¿cómo van a permitir los Sabios que uno se vengue de su prójimo? Además, en el Tratado Ketubot (50a), se dijo: “Hasta los doce años, el padre deberá compartir con su hijo lo que tenga, a partir de allí, entrará con él en disputas”. ¿Acaso esto quiere decir que el padre deberá competir con el hijo en los negocios? No. Por tanto, se refiere a que entrará con él en disputas en cuanto a inducirle a que cumpla todos los preceptos de la Torá. HA RENEGADO. Al haber prestado al prójimo con interés.  NO TROPEZARÁ. Si el versículo dijo: “Quien así actúa nunca tropezará”, se entiende, que quien se comporta de forma diferente sí lo hará.  NO SE RECUPERAN. Así se desprende del versículo: “Quien así actúa, nunca tropezará”. Es decir, si tropieza, no será para siempre, mientras que presta con interés, siempre tropezará.  COMO GARANTE. Fiador del prestatario frente al prestamista.  ¿CON QUIÉN? ¿Quién es este prestamista para poder saber si puedo actuar como fiador?  LOS SIGUIENTES. Transgreden la Ley de la Torá que dice: “No le impondrán interés” (Éxodo 22:24).  Siendo que los gentiles primeramente cobran al garante, sólo a él le exigirán el pago.  INCURRIRÍA EN UNA FALTA. Se considera como si el fiador tomara prestado del prestamista para prestárselo, a su vez, con interés, al prestatario.  SEGÚN LA LEY JUDÍA. O sea, primeramente demandará al prestatario y después al fiador. Siendo así, no se le considera al fiador como si él hubiese tomado prestado en primer lugar. Cuando el prestamista le cobra a él, la misma cantidad se la cobra al prestatario.

    Yitró 89a

    (Comentario sobre el Zóhar)

    “Así le dijo el Eterno a los eunucos” (Isaías 56:4). ¿Quiénes son éstos? Son los eruditos que están ocupados con la Torá y se comportan como eunucos durante los seis días de la semana, mientras que en la noche de Shabat se apresuran para estar con sus mujeres, porque conocen el secreto supremo: Es el momento en que la Reina se apega al Rey. Los eruditos que conocen este secreto dirigen sus corazones hacia la Fe en su Creador, y en esa noche se bendice su simiente. Así dice el Texto: “Aquellos que guardaron Mis preceptos” (Ibíd.), como también en: “Pero su padre aguardó el asunto” (Génesis 37:11). Se denominan eunucos porque aguardan al Shabat para cumplir la Voluntad de su Creador, tal como versa: “Escogen aquello que Me complace” (Isaías 56:4). ¿Qué significa lo que Me complace? Se refiere a cohabitar con la Reina. “Y son fieles a Mi pacto" (Ibíd.), es todo uno, sólo en Mi pacto. Dichosa aquella persona que se santifica con esta santidad, y conoce este secreto. Ven y observa: Está escrito: “Seis días trabajaras y harás toda tu labor, pero el día séptimo, Shabat, lo consagrarás al Eterno, tu Di-s” (Éxodo 20:9). “Toda tu labor”, estos seis días son para efectuar las labores humanas. Por esta causa los eruditos no tienen relación con sus mujeres en el tiempo que las personas laboran y sí en los días que el Eterno trabaja. ¿Cuál es Su labor? Cohabitar con la Reina, para hacer descender almas santas al mundo. Por esta razón, en esta noche, los eruditos se santifican con la santidad de su Creador, concentran sus corazones en su Creador, y así engendran hijos buenos y santos, que no se apartan ni a diestra ni a siniestra, hijos del Rey y la Reina. Sobre ellos fue escrito: “Hijos sois del Eterno vuestro Di-s” (Deuteronomio 14:1). Hijos del Eterno, ya que estos son considerados Sus hijos, hijos del Rey y de la Reina. Los eruditos que conocen este secreto se apegan a esto, y por esta causa son llamados Hijos del Santo, Bendito Él. Por ellos el mundo se sostiene, y cuando el mundo es juzgado, el Santo, Bendito Él se fija en Sus hijos, y se apiada del mundo. Referente a ellos el versículo dice: “Todos son de buena semilla” (Jeremías 2:21), de verdadera buena semilla.

    Halajá

    Rambam, Leyes sobre la Lectura de la Shemá, Capítulo 1

    1. Se debe pronunciar la Shemá dos veces al día, por la noche y por la mañana, como está escrito: “Al acostarte y al levantarte” (Deuteronomio. 6:7), o sea, por la noche, cuando la gente suele ir a dormir, y cuando se levanta, que es en el día. 2. ¿Qué se lee? Tres pasajes, que son: “Escucha, Oh Israel” (Deuteronomio. 6:4-9). “Y sucederá que si cumpliereis permanentemente” (Ibíd. 11:13-21). “Y dijo el Eterno” (Números 15:37-41). Se adelanta la lectura de Primero se lee el párrafo: “Escucha, Oh Israel”, porque contiene el principio de Unicidad del Eterno, el mandamiento de Amarlo, y el precepto de estudiar Torá, que son los principios fundamentales del judaísmo y de lo cual depende todo lo demás. Posteriormente, se pronuncia: “Y sucederá, que si cumpliereis permanentemente”, porque contiene un mandamiento para recordar todos los demás. Finalmente, se recita el texto concerniente al mandamiento de Tzitzit, puesto que también incluye el mandamiento de recordar todas las demás Mitzvot. 3. Aunque no estamos obligados a colocar los Tzitzit durante la noche, se recita este texto porque recuerda la salida de Egipto, la cual es obligación mencionar, tanto de día como de noche, como versa: “Para que recuerdes la salida de Egipto todos los días de tu vida” (Deuteronomio 16:3). La lectura de estos tres textos en su orden preceptivo se denomina: “La lectura de la Shemá”. 4. Después de recitar el primer versículo de la Shemá, dirá en voz baja: “Bendito sea el nombre de Su glorioso Reinado por toda la eternidad”. Luego continúa leyendo normalmente: “Amarás al Eterno tu Di-s”, y así sucesivamente hasta el final. Si en el servicio matutino alguien comenzó con “Que creó la luz” y terminó con “Que oscureció las tardes”, no habrá cumplido con su obligación. Sin embargo, si comienza diciendo: “Que oscureció las tardes” y termina con “Que creó la luz”, habrá cumplido con su obligación. Si en el servicio vespertino comienza diciendo “Que oscureció las tardes” y termina con “Que creó la luz”, no habrá cumplido con su obligación. No obstante, si comienza con “Que creó la luz” y termina con “Que oscureció las tardes”, habrá cumplido con su obligación. Esto se debe a que la conclusión es la parte determinante de una bendición.

    Moral

    Compendio

    Aprendimos (Avot 1:5): “No converses mucho con las mujeres”. Aquí se refiere a su propia esposa, ¡cuánto y más con la de otro hombre! De aquí aprendieron los Sabios: quien conversa mucho con una mujer se hace daño y descuida sus estudios de Torá, y al final heredará el infierno. ¡Ay de los tontos y necios que escuchan o leen esta Mishná en la sinagoga, haciendo caso omiso de lo que advirtieron los Sabios. ¡Cuánto más advierten, menos casos hacen! Nuestros santos Maestros dicen que se hace mal a sí mismo, o sea, que el Otro Lado, llamado también el Mal, se apodera de él. En consecuencia, descuida sus estudios de Torá, lo cual es peor que cometer idolatría, mantener relaciones prohibidas o el asesinato. Y al final, heredará el infierno. Nuestros sagrados Maestros no se equivocaron al decir: “Y al final”, porque cuando muera, será lanzado de un ángel malo a otro, sin descanso. Y después de muchos años, lo llevarán a recibir su herencia en el infierno, y la parte de los otros, puesto que el inicuo hereda, tanto su propia parte como la de los justos. Debe asombrarnos la ceguera de la gente necia que ignora las palabras de nuestros Maestros. Cuando les preguntas si de verdad creen en sus palabras, dirán: por supuesto que sí, pero sus acciones demuestran lo contrario. O quizás, por su arrogancia, no les preocupan los castigos. ¡Que el Eterno los salve!
