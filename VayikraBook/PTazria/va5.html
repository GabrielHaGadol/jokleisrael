יָבִיא בְדָמָיו אָיִל. רַבִּי אוֹסֵר. הָאוֹמֵר אֶחָד מִכְּבָשַׂי הֶקְדֵּשׁ, וְאֶחָד מִשְּׁוָרַי הֶקְדֵּשׁ, הָיוּ לוֹ שְׁנַיִם, הַגָּדוֹל שֶׁבָּהֶן הֶקְדֵּשׁ. שְׁלשָׁה, הַבֵּינוֹנִי שֶׁבָּהֶן הֶקְדֵּשׁ. פֵּרַשְׁתִּי וְאֵינִי יוֹדֵעַ מַה פֵּרַשְׁתִּי, אוֹ שֶׁאָמַר, אָמַר לִי אַבָּא וְאֵינִי יוֹדֵעַ מָה, הַגָּדוֹל שֶׁבָּהֶן הֶקְדֵּשׁ: י. הֲרֵי עָלַי עוֹלָה, יַקְרִיבֶנָּה בַמִּקְדָּשׁ. וְאִם הִקְרִיבָהּ בְּבֵית חוֹנְיוֹ, לֹא יָצָא. שֶׁאַקְרִיבֶנָּה בְּבֵית חוֹנְיוֹ, יַקְרִיבֶנָּה בַמִּקְדָּשׁ. וְאִם הִקְרִיבָהּ בְּבֵית חוֹנְיוֹ, יָצָא. רַבִּי שְׁמְעוֹן אוֹמֵר, אֵין זוֹ עוֹלָה. הֲרֵינִי נָזִיר, יְגַלַּח בַּמִּקְדָּשׁ. וְאִם בְּבֵית חוֹנְיוֹ, יָצָא. רַבִּי שִׁמְעוֹן אוֹמֵר, אֵין זֶה נָזִיר. הַכֹּהֲנִים שֶׁשִּׁמְּשׁוּ בְּבֵית חוֹנְיוֹ, לֹא יְשַׁמְּשׁוּ בַמִּקְדָּשׁ בִּירוּשָׁלַיִם, וְאֵין צָרִיךְ לוֹמַר לְדָבָר אַחֵר, שֶׁנֶּאֱמַר (מלכים ב. כג), אַךְ לֹא יַעֲלוּ כֹּהֲנֵי הַבָּמוֹת אֶל מִזְבַּח יְיָ בִּירוּשָׁלָיִם כִּי אִם אָכְלוּ מַצוֹת בְּתוֹךְ אֲחֵיהֶם, הֲרֵי הֵם כְּבַעֲלֵי מוּמִין, חוֹלְקִין וְאוֹכְלִין, אֲבָל לֹא מַקְרִיבִין: יא. נֶאֱמַר בְּעוֹלַת הַבְּהֵמָה אִשֵּׁה רֵיחַ נִיחוֹחַ, (וּבְעוֹלַת הָעוֹף אִשֵּׁה רֵיחַ נִיחוֹחַ), וּבַמִּנְחָה אִשֵּׁה רֵיחַ נִיחוֹחַ,     ר' עובדיה מברטנורה

    כ(א) הרי עלי עשרון. פירשתי. כמה עשרונים אביא, ואיני יודע כמה אמרתי: יביא ששים עשרונים. דאי בציר מהכי נדר לא איכפת [לן] דמתנה ואומר מה שפירשתי יהא לנדרי והשאר יהא לנדבה. ובטפי מהכי ליכא לספוקי, דאין מנחה אחת יתירה מששים עשרון: הרי עלי מנחה. האומר הרי עלי מנחה סתם: יביא איזו שירצה. מחמש מנחות: מיוחדת. שנקראת מנחה סתם ואין לה שם לווי. וכל שאר מנחות יש להם שם לווי, מנחת מחבת, מנחת מרחשת, מנחת מאפה. ואין הלכה כר' יהודה: (ב) מנחה מין המנחה. האומר הרי עלי מנחה, או מין מנחה: יביא אחת. מן המנחות האמורות בפרשה: מנחות. עלי, או שאומר מין מנחות עלי, יביא שתי מנחות ממין אחד: פירשתי. מיני מנחות שאביא. ואיני יודע כמה מינים הם שנדרתי להביא: מביא חמשתן. מנחת סולת, מנחת מרחשת, מנחת מחבת, מנחת מאפה תנור והיא באה שני מינים, חלות ורקיקין: פירשתי מנחה של עשרונים בכלי אחד, ואיני יודע כמה עשרונים קבעתי בה: יביא ששים עשרון. דטפי מהכי ליכא לספוקי. ואי בציר מהכי נדר, לא איכפת לן דמתנה ואומר כמה שפירשתי יהא לנדבה: רבי אומר. כיון דאמר מנחה דמשמע חדא, אלמא בכלי אחד קבעה, ואם יביא ששים בכלי אחד, שמא בציר מהכי נדר והויא מנחה יתירה, דאית ליה לרבי דקביעותא דמנא מלתא היא, הלכך יביא ששים מנחות מאחת ועד ששים, חדא של עשרון אחד וחדא של שנים, וחדא של שלשה, וחדא של ארבעה, עד ששים, שנמצא מביא בין הכל אלף שמונה מאות ושלשים עשרון, דודאי חדא מנייהו נדר, ואינך נדבה. ואין הלכה כרבי: (ג) לא יפחות משני גזירין. שתי בקעיות גדולות. דמיעוט עצים שנים: המעלה את הקומץ בחוץ חייב. כרת. דהעלאה היא. ובהאי כללא הוי נמי המעלה קומץ בפנים דהקטרה מעלייתא היא, הלכך חשיב חמשה קומצים ותו לא, ולא חשיב להאי דהמעלה קומץ בפנים לקומץ ששית: ושני בזיכים. של לחם הפנים: (ד) לא יפחות מדינר זהב. והוא שיאמר מטבע של זהב. דאי לא, דלמא נסכא דדהבא קאמר, דהיינו חתיכה של זהב: נחושת לא יפחות ממעה כסף. שיביא נחושת ששוה מעה כסף: פירשתי. כך וכך זהב: ואיני יודע כמה פירשתי. יביא כל כך עד שידע בעצמו שמעולם לא נתכוין לכל כך: (ה) הרי עלי יין. לנסכים: לא יפחות משלשה לוגין. שהן פחותים שבנסכים, רביעית ההין לכבש תלתא לוגין הוו. שהין שנים עשר לוגין: שמן לא יפחות מלוג. שהפחות שבמנחה, עשרון סולת, והיא טעונה לוג שמן: רבי אומר שלשה לוגין. כפחות שבמנחת נסכים, עשרון לכבש בלול ברביעית ההין שמן. ואין הלכה כרבי: כיום המרובה. כיום טוב של חג הסוכות כשחל להיות בשבת. שאותו היום מרובה בנסכים לקרבנות של חובת היום מכל שאר ימות השנה. דהוו שלשה עשר פרים וארבעה עשר כבשים וארבעה מוספים, שנים מוסף דשבת ושנים מוסף דחג, ואילים שנים ושעיר אחד. והנסכים הצריכים לכולם, מאה וארבעים לוג: (ו) הרי עלי עולה יביא כבש. ר' אלעזר בן עזריה אומר או תור או בן יונה. באתריה דתנא קמא לא היו קורים עולה סתם אלא לעולת בהמה, ופחותה שבעולת בהמה הוא כבש, הלכך יביא כבש. ובאתריה דרבי אלעזר בן עזריה היו קורים עולה סתם גם לעולת העוף, הלכך יביא תור או בן יונה. ומר כי אתריה, ומר כי אתריה: יביא פר ועגל. זכרים ולא נקבות, דעולה ליכא לספוקי אלא בזכרים. ומתניתין רבי היא דאמר לקמן המתנדב קטן והביא גדול לא יצא. אבל רבנן פליגי עליה ואמרי המתנדב קטן והביא גדול יצא, דיש בכלל מרובה מועט. והלכה כחכמים: פירשתי. אחד מקרבנות הבהמה: ואיני יודע מאיזה מהן פירשתי. יביא מכל מיני בהמה הזכרים גדולים וקטנים, דהיינו פר ועגל איל ושעיר וגדי וטלה: פירשתי. את המין: ואיני יודע מה פירשתי. אם מין הבהמה או מין העוף: (ז) תודה ושלמים. תודה או שלמים: יביא כבש. פחות שבתודה ושלמים: יביא פר ופרה. דתודה ושלמים איכא לספוקי בזכרים ונקבות: רחל. נקבה בת שתי שנים: גדי. בן שנה מן העזים: שעיר. עז בן שתי בַעֲבוֹדָה זָרָה וַהֲווֹ בָנִין בִּנְיָין לְפַרְצוּפַיְיהוּ (ס''א לְטִנּוּפַיְיהוּ) וּלְגִיעוּלַיְיהוּ לִסְטַר מְסָאֲבָא לַעֲבוֹדָה זָרָה וְכַד שָׁרָאן לְמִבְנֵי הֲווֹ אַמְרֵי מִלָּה וְכֵיוַן דְּאִתְדְּכָר בְּפוּמַיְיהוּ סָלִיק עֲלֵיהּ רוּחַ מְסָאֲבָא וְכַד אִסְתְּלִיק עֲבִידְתָּא בְרוּחַ מְסָאֲבָא אִסְתְּלִיק. כֵּיוַן דְּעָאלוּ יִשְׂרָאֵל לְאַרְעָא בָעָא קֻדְשָׁא בְּרִיךְ הוּא לְדַכָּאָה לוֹן וּלְקַדְּשָׁא לוֹן אַרְעָא וּלְאַפְנָאָה אֲתַר לִשְׁכִינְתָּא דְלָא תִשְׁרֵי שְׁכִינְתָּא גּוֹ מְסָאֲבָא וְעַל דָּא בְהַהוּא נֶגַע צָרַעַת הֲווֹ סַתְרִין בִּנְיָינִין דְּאָעִין וַאֲבָנִין דְּאִתְעֲבִידוּ בִמְסָאֲבוּ. תָּא חֲזֵי אִי עוֹבָדָא דָא הָוָה לְאַשְׁכְּחָא מַטְמוֹנִין בִּלְחוֹדוֹי יֵהַדְרוּן אֲבָנִין לְבָתַר כְּמָה דְאִינוּן לְאַתְרַיְיהוּ וְעַפְרָא לְאַתְרֵיהּ אֲבָל קְרָא כְּתִיב וְחִלְּצוּ אֶת הָאֲבָנִים וּכְתִיב וְעָפָר אַחֵר יִקַּח. בְּגִין דְּיִתְעֲבָר רוּחַ מְסָאֲבָא וְיִתְפְּנֵי וְיִתְקַדֵּשׁ הַשְׁתָּא כְמִלְּקַדְמִין וְיִשְׁתְּכָחוּ יִשְׂרָאֵל בִּקְדוּשָׁה וּבְדִיּוּרָא קַדִּישָׁא לְמִשְׁרֵי בֵּינַיְיהוּ שְׁכִינְתָּא וְעַל דָּא מָאן דְּבָנֵי בִנְיָין כַּד שָׁארֵי לְמִבְנֵי בָּעֵי לְאַדְכְּרָא בְפוּמָה דְּהָא לְפוּלְחָנָא דְקֻדְשָׁא בְּרִיךְ הוּא הוּא בָנֵי בְגִין דִּכְתִיב (ירמיה כ''ב) הוֹי בּוֹנֶה בֵיתוֹ בְלֹא צֶדֶק. וְגוֹמֵר. וּכְדֵין סִיַּיעְתָּא דִשְׁמַיָּיא שָׁארֵי עֲלוֹי וְקֻדְשָׁא בְּרִיךְ הוּא זַמִּין עֲלֵיהּ קְדוּשָׁתָא וְקָארֵי עֲלֵיהּ שְׁלָם הֲדָא הוּא דִכְתִיב (איוב ה') וְיָדַעְתָּ כִּי שָׁלוֹם אָהֳלֶךָ וְגוֹמֵר:     תרגום הזוהר

    ככָּתוּב, כִּי תָבֹאוּ אֶל אֶרֶץ כְּנַעַן וְגוֹ'. וְנָתַתִּי נֶגַע צָרַעַת בְּבֵית אֶרֶץ אֲחוּזַתְכֶם. שׁוֹאֵל, וְכִי הוּא זֶה שָׂכָר טוֹב שֶׁיִּמָּצְאוּ נְגָעִים בַּבָּתִּים שֶׁל אֵלּוּ שֶׁזָּכוּ לִכָּנֵס לָאָרֶץ. וּמֵשִׁיב, אֶלָּא הֲרֵי הֶעֱמִידוּהוּ, שֶׁהוּא, כִּי אַחַר שֶׁיִּתְּצוּ הַבָּתִּים הַמְּנוּגָעִים, יִמְצְאוּ אוֹצָרוֹת שֶׁהִצְפִּינוּ הַכְּנַעֲנִים בְּבָתֵּיהֶם, וְיֶהֱנוּ מֵהֶם יִשְׂרָאֵל. אֲבָל בֹּא וּרְאֵה, אַשְׁרֵיהֶם יִשְׂרָאֵל שֶׁהֵם דְּבֵקִים בַּקָּדוֹשׁ בָּרוּךְ הוּא, וְהַקָּדוֹשׁ בָּרוּךְ הוּא אוֹהֵב אוֹתָם, שֶׁכָּתוּב, אָהַבְתִי אֶתְכֶם אָמַר ה'. וּמִתּוֹךְ אַהֲבָתוֹ, הִכְנִיסָם לָאָרֶץ הַקְּדוֹשָׁה, לְהַשְּׁרוֹת שְׁכִינָתוֹ בֵּינֵיהֶם, וְלָדוּר עִמָּהֶם, וְשֶׁיִּמָּצְאוּ יִשְׂרָאֵל קְדוֹשִׁים מִכָּל בְּנֵי הָעוֹלָם. בֹּא וּרְאֵה, כָּתוּב, וְכָל הַנָּשִׁים אֲשֶׁר נָשָׂא לִבָּן וְגוֹ'. הַיְנוּ, בְּשָׁעָה שֶׁהָיוּ עוֹשׂוֹת מְלַאכְתָּן, הָיוּ אוֹמְרוֹת, זֶה לַמִּקְדָּשׁ, זֶה לַמִּשְׁכָּן, זֶה לַפָּרֹכֶת, וְכֵן עָשׂוּ כָּל הָאוּמָנִים, כְּדֵי שֶׁתִּשְׁרֶה הַקְּדֻשָּׁה עַל יְדֵיהֶם, וְתִתְקַדֵּשׁ הַמְּלָאכָה הַהִיא, וְכַאֲשֶׁר הֶבִיאוּהָ לִמְקוֹמָהּ עָלְתָה וְהָיְתָה בִּקְדֻשָּׁה. כְּעֵין זֶה מִי שֶׁעוֹבֵד לַעֲבוֹדָה זָרָה, אוֹ לְצַד הָאַחֵר שֶׁאֵינוֹ קָדוֹשׁ, כֵּיוָן שֶׁמַּזְכִּירָהּ עַל עֲבוֹדָה הַהִיא, הִנֵּה רוּחַ הַטּוּמְאָה שׁוֹרֶה עָלָיו, וּכְשֶׁעוֹלֶה הָעֲבוֹדָה, הִיא עוֹלָה בְּטוּמְאָה. הַכְּנַעֲנִים, עוֹבְדֵי עֲבוֹדָה זָרָה הָיוּ, וְכֻלָּם יַחַד הָיוּ דְּבוּקִים בְּרוּחַ הַטּוּמְאָה בַּעֲבוֹדָה זָרָה, וְהָיוּ בּוֹנִים בִּנְיָנִים לְפַרְצוּפֵיהֶם וּלְתוֹעַבְתָּם שֶׁל צַד הַטּוּמְאָה, לַעֲבוֹדָה זָרָה, וּכְשֶׁהִתְחִילוּ לִבְנוֹת, הָיוּ אוֹמְרִים אֵיזֶה דָּבָר, וְכֵיוָן שֶׁנִּזְכָּר בְּפִיהֶם, עָלָה עַל הַבִּנְיָן רוּחַ הַטּוּמְאָה. וּכְשֶׁעָלְתָה הָעֲבוֹדָה, עָלְתָה בְּרוּחַ הַטּוּמְאָה. כֵּיוָן שֶׁבָּאוּ יִשְׂרָאֵל לָאָרֶץ, רָצָה הַקָּדוֹשׁ בָּרוּךְ הוּא לְטַהֵר לְמַעֲנָם אֶת הָאָרֶץ, וּלְפַנּוֹת מָקוֹם לַשְּׁכִינָה, כְּדֵי שֶׁלֹּא תִּשְׁרֶה הַשְּׁכִינָה בְּמָקוֹם טָמֵא. וְעַל כֵּן בְּנֶגַע צָרַעַת הַהוּא הָיוּ הוֹרְסִים הַבִּנְיָנִים שֶׁל עֵצִים וַאֲבָנִים שֶׁנַּעֲשׂוּ בְּטוּמְאָה. בֹּא וּרְאֵה, אִם מַעֲשֶׂה זֶה, שֶׁל נְתִיצַת הַבָּתִּים הַמְּנוּגָעִים, הָיָה כְּדֵי לִמְצֹא אוֹצָרוֹת בִּלְבָד, הָיוּ צְרִיכִים אַחַר כָּךְ לְהַחֲזִיר הָאֲבָנִים לִמְקוֹמָן כְּמוֹ שֶׁהָיוּ, וְכֵן הָיוּ צְרִיכִים לְהַחֲזִיר הֶעָפָר לִמְקוֹמוֹ. אֲבָל הַמִּקְרָא כָּתוּב, וְחִלְצוּ אֶת הָאֲבָנִים. וְכָתוּב וְעָפָר אַחֵר יִקָּח. שֶׁהוּא, כְּדֵי לְהַעֲבִיר אֶת רוּחַ הַטּוּמְאָה, וּתְפוּנֶה, וְתִתְקַדֵּשׁ עַתָּה הָאָרֶץ כְּמִקֹּדֶם, וְיִמָּצְאוּ יִשְׂרָאֵל בִּקְדֻשָּׁה, וּבְדִיּוּר קָדוֹשׁ, שֶׁתִּשְׁרֶה בֵּינֵיהֶם הַשְּׁכִינָה. וְעַל כֵּן מִי שֶׁבּוֹנֶה בִּנְיָן, כְּשֶׁמַּתְחִיל לִבְנוֹתוֹ, צָרִיךְ לְהַזְכִּיר בְּפִיו, שֶׁבּוֹנֶה אוֹתוֹ לַעֲבוֹדַת הַקָּדוֹשׁ בָּרוּךְ הוּא, מִשּׁוּם שֶׁכָּתוּב, הוֹי בּוֹנֶה בֵּיתוֹ בְלֹא צֶדֶק, וְאָז עֶזְרַת הַשָּׁמַיִם שׁוֹרֶה עָלָיו, עַל בֵּיתוֹ, וְהַקָּדוֹשׁ בָּרוּךְ הוּא מֵכִין עָלָיו קְדֻשָּׁתוֹ, וְקוֹרֵא עָלָיו שָׁלוֹם. זֶה שֶׁנֶּאֱמַר וְיָדַעְתָּ כִּי שָׁלוֹם אָהֳלֶךְ וגו'.     Rashí

    14. וּבְיוֹם הֵרָאוֹת בּוֹ בָּשָׂר חַי / PERO EN EL DÍA QUE EN ELLA SE OBSERVE CARNE SANA. Si en la llaga apareció carne sana, la Torá ya ha explicado al respecto que ello constituye un signo de impureza. Sin embargo, esta idea ha sido repetida aquí para enseñar lo siguiente: en caso de que la afección cutánea se halle en uno de las veinticuatro puntas de los miembros principales del cuerpo, los cuales no transmiten impureza por la aparición de carne sana –pues la afección no puede ser vista enteramente de un solo vistazo, ya que se sesga hacia un lado y hacia otro, y luego la punta del órgano cambia y su sesgo queda expuesto por la grasa acumulada en dicha área en caso de que dicha área haya engordado y se haya hinchado, haciéndose visible la carne sana–, se pudo haber pensado que la afección sigue siendo pura. Acerca de este caso, este versículo nos enseña que lo hace impuro.  וּבְיוֹם / PERO EN EL DÍA. ¿Qué es lo que esto nos enseña? Enseña que hay un día en que observas el área afectada para determinar si es pura o impura y hay un día en que no la observas. A partir de aquí, nuestros Maestros afirmaron que a un recién casado se le conceden los siete días enteros de la celebración nupcial sin que sea observado (para no estropear la alegría nupcial), tanto a él como a su traje, a su demás ropa y a su casa, si en cualquiera de estos objetos brota tzaráat. Y de igual modo, en una festividad se le conceden todos los días de la festividad.  15. צָרַעַת הוּא / TZARÁAT ES. Esa carne. La palabra בָּשָׂר es masculina.  18. שְׁחִין / INFLAMACIÓN. Este término significa calentamiento. Se aplica a esta erupción cutánea porque la carne se calienta por la infección que le sobrevino a causa de un golpe, pero no a causa de la quemadura de un fuego.  וְנִרְפָּא / Y SE HUBIESE CURADO. La inflamación se mejoró y en su lugar apareció otra afección cutánea.

    Mishná

    Tratado Menajot, capítulo 13

    (1) (Quien dice) “Me comprometo a traer un décimo”: Debe traer uno. “Décimos”: Debe traer dos. “Dije cuántos, pero no me acuerdo lo que dije”: Llevará sesenta décimos. “Me comprometo a ofrendar una Minjá”: Llevará la que quiera. Dijo Rabí Yehudá: Llevará la Minjá de harina de flor que es la única especificada.  (2) “Una Minjá”, o “Parte de una Minjá”, llevará sólo una. “Varias Minjá”, o “Parte de varias Minjá”, llevará dos. “Dije cuántas, pero no me acuerdo lo que dije”: Llevará las cinco. “Dije específicamente que llevaría una Minjá con varios décimos, pero no me acuerdo lo que dije exactamente”: Llevará una Minjá de sesenta décimos. Dijo Rabí: Llevará tantas Minjá como décimos, desde uno hasta sesenta.  (3) (Quien dice) “Me comprometo a traer leña”: Llevará, como mínimo, dos leños. “Olíbano”: Llevará, como mínimo, un puñado. En cinco ocasiones se fijó la medida de un puñado: (Aquel que dijo) “Me comprometo a llevar olíbano”: Llevará, como mínimo, un puñado. Quien se compromete a llevar una Minjá, la acompañará con un puñado de olíbano. Quien incinera el puñado (que acompaña a la Minjá) en el exterior, se condena. Las dos escudillas requieren sendos puñados.  (4) “Me comprometo a llevar oro”: Llevará, al menos, un denario de oro. “Plata”: Llevará, como mínimo, un denario de plata. “Cobre”: Llevará, al menos, una moneda de plata. “Prometí algo, pero no me acuerdo qué”: Empezará a llevar hasta que diga: ¡Tanto no prometí!  (5) “Me comprometo a llevar vino”: Llevará, como mínimo, tres Log. “Aceite”: Al menos, un Log. Entonces, si es que prometió menos de esta cantidad, no importa, pues proferirá la siguiente condición: lo que no prometí sea considerado como una donación.  DIJO RABÍ. Siendo que dijo “Minjá”, en singular, se entiende: En un solo recipiente. Es decir: Una Minjá con un décimo, otra con dos, otra con tres, y así sucesivamente hasta sesenta. De hacerlo así estaría ofrendando en total mil ochocientos treinta décimos. ¡Seguro que una de esas ofrendó! El resto se considerarán donaciones. La Halajá no coincide con la opinión de Rabí.  3. DOS LEÑOS. Dos troncos grandes, porque el mínimo de “leños” es dos. SE CONDENA. A Karet, por haberlo quemado. Esta regla incluye a quien quemó el puñado dentro del Atrio, y por eso se contaron cinco y no seis. Este versículo habla de los animales que adquirieron un defecto y que fueron reemplazados, tal como se explica en el capítulo anterior.  REEMPLAZARLO POR DOS. Esta Ley no es igual a aquello que se dijo al principio: “Un toro de cien: Si lleva dos que valen conjuntamente cien no cumple con su obligación”, porque allí especificó que debía ser cualquier toro que valga cien, pero aquí se habla de “Este”. Por tanto, si adquiere un defecto se anula la promesa, pues ya nunca más podrá cumplir con su promesa.  RABÍ LO PROHÍBE. En un principio, pero si lo llevó, le sirve. Siendo que dijo “Este”, no es responsable por él. La Ley no coincide con la opinión de Rabí.  CONSAGRARÁ EL MEDIANO. Es decir, también consagrará el mediano, porque cuando tiene dos, el mayor se consagra, pues quien lo ofrece, ofrece lo mejor, tal como versa (Deuteronomio 12:11): “Lo mejor de vuestras promesas”. Cuando hay tres, sospechamos que también desea consagrar el mediano, porque no sabemos cuál de ellos exactamente desea consagrar. Por un lado está el grande, que es el mejor, y por otro, tenemos el mediano, que es el mejor con respecto al pequeño. Por esta razón se prohíben ambos. Sin embargo, sólo se sacrifica uno. ¿Cómo debe proceder para deshacerse de uno? Espera hasta que el mediano adquiera un defecto, y sacrifica el grande. La lógica es la siguiente: Si el mediano es el que hay que sacrificar, el defecto lo invalida, y debe consagrar el grande desde un principio, el mediano es profano desde un principio.  ALGO PROMETÍ. Uno de ellos pero no recuerdo cuál.  ME ORDENÓ MI PADRE. Cuando estaba moribundo dijo: He ofrendado uno de mis toros, pero no recuerdo cuál me dijo. EL MAYOR. Al decir “Prometí”, no cabe duda de que prometió el más grande.  10.     (Comentario sobre el Zóhar)

    Está escrito (Levítico 14:34): “Cuando lleguen a la tierra de Kenaan… y Yo ponga afección de tzaráat sobre una casa en la tierra de su posesión”. ¿Acaso esta es la buena recompensa que se van a encontrar aquellos que merezcan llegar a la Tierra? No, se refiere a que encontrarán los tesoros que escondieron los kenaanitas en sus casas (después de derrumbar las casas afectadas) y de los cuales se van a beneficiar los israelitas. Pero, ven y observa: Bienaventurado sea Israel por apegarse al Santo, Bandito Él, pues el Santo, Bendito Él, los ama, ya que está escrito (Malají 1:2): “¡Os amo!, dijo el Eterno”. Gracias a este amor los hizo que entraran en la Tierra Sagrada, para depositar su Presencia Divina sobre ellos, y para habitar entre ellos, y para lograr que el Pueblo de Israel sea el más santo de todo el mundo. Ven y observa: Está escrito (Éxodo 35:26): “Y todas las mujeres cuyo corazón las inspiró con sabiduría”. Cuando realizaban su trabajo, decían: Esto es para el Santuario, esto es para el Tabernáculo, esto es para la Cortina, y así sucesivamente decían todos los artesanos, para que la santidad se posara sobre sus manos. Así su trabajo quedaba santificado, y cuando lo llevaban a su lugar correspondiente, llegaba con santidad. Asimismo ocurre con aquel que practica la idolatría, o al Otro Lado que no es santo. Cuando menciona este servicio el espíritu impuro se posa sobre él, y cuando el servicio se eleva, se eleva en la impureza. Los kenaanitas eran idólatras, y todos juntos estaban apegados al espíritu de la impureza, a la idolatría, y les construían edificios para colocar sus bustos (sus suciedades) y para sus abominaciones, que estaban del lado de la impureza y de la idolatría. Cuando empezaban a construirlas decían ciertas palabras, que al sacarlas de sus bocas hacían que dicha impureza se apoderara de esos edificios. Así, cuando comenzaban a servir, aparecía el espíritu de la impureza. Al entrar Israel a la Tierra, quiso el Santo, Bendito Él, purificarles y santificarles la Tierra, y dejar un espacio para la Shejiná, para que la Presencia Divina no se depositara en un lugar impuro. Por esta razón, gracias a dichas afecciones de tzaráat tenían que derrumbar los edificios de madera y piedra que construyeron con impureza. Ven y observa: Si todo esto lo hizo sólo con la intención de que encontraran los tesoros, es porque después podrían devolver las piedras a su lugar tal como estaban anteriormente, y asimismo deberían devolver la tierra a su lugar. Sin embargo, está escrito (´Levítico 14:40): “Quitarán las piedras”, y también (Ibíd. 42): “Y tomarán otra tierra”. Todo esto para que saliera el espíritu de impureza, para que dejara espacio y para que la Tierra se santificara como antaño. De esta forma el Pueblo de Israel podría santificarse y vivir con santidad, y facilitar que se pose entre ellos la Shejiná. Entonces, la ayuda celestial se posa sobre él, y el Santo, Bendito Él, hace morar Su Santidad sobre él, y lo llama “Shalom”. Sobre esto está dicho (Job 5:24): “Sabrás que tu morada está en paz…”.

    Halajá

    Rambam, Leyes sobre las Ofrendas, capítulo 3

    1. Todo aquello que se ofrenda sólo de día, como por ejemplo, el puñado, el olíbano, el incienso y las Minjá que se incineran, se permite ofrendarlas con la puesta del sol, aunque se consuman en el fuego toda la noche como ocurre con los miembros de la Ofrenda de Ascensión.  Las libaciones que acompañan a los sacrificios se ofrendan sólo de día, tal como versa (Números 29:39): “Para vuestras libaciones y para vuestros Sacrificios de Paz”. El Texto comparó las libaciones a los Sacrificios de Paz: Al igual que los Sacrificios de Paz se ofrendan sólo de día, igualmente las libaciones deben ofrendarse de día. No obstante, las libaciones que se ofrendan de forma individual se consagran de noche y se ofrecen de noche. Por tanto, si le dan libaciones que se ofrecen separadamente, los consagran y los ofrecerá de noche. El alba los anula, tal como ocurre con los miembros de las Ofrendas de Ascensión. 3. Durante todo el día se puede realizar la apoyatura de las manos, el degüello, el desnuque, la quema del incienso, la presentación de la sangre, el rociamiento, el alzamiento, la separación del puñado, y el Sacrificio Adicional (Musaf). Por otro lado, durante toda la noche se pueden incinerar las vísceras y los miembros. La regla es así: Todo lo que se debe hacer de día, vale todo el día, y todo lo que se debe hacer de noche, vale toda la noche. Sin embargo, los que se esmeran, cumplen los preceptos lo antes que pueden. 

    Moral

    Shaaré Teshuvá, Rabenu Yoná, página 2, capítulo 4

    Lo esencial en el arrepentimiento es abandonar el pecado. Que abandone los malos caminos, y resuelva, con todo su corazón, que no volverá a conducirse de esa manera. Si obró de modo execrable, no continuará haciendo lo mismo, tal como está escrito (Yejezquel 33:11): “Volved, volved de vuestros malos caminos”. También se dijo (Yeshayahu 55:7): “¡Abandone el malvado su camino! Debes saber que quien pecó de forma accidental porque sintió pasión y no pudo controlar su instinto, no tuvo ideas, le traicionaron sus sentidos y no lo pudo doblegar, ni tampoco se apresuró a gritar en el mar del deseo, el instinto lo conquistará con sus armas y caerá en sus trampas en ese preciso momento. No porque él deseaba pecar sino, porque el instinto malo lo asustó para que haga lo que él desea. Lo primero que debe hacer la persona que quiere enmendar su camino es arrepentirse, y acongojar su corazón por el pecado cometido, y que el alma se apene amargamente como la amargura del ajenjo. Luego, deberá agregar a su alma, diariamente, temor al Creador, y su corazón sentirá miedo del Todopoderoso, en todo momento, hasta que consiga que su corazón confíe en el Eterno completamente. Así, cuando pase de nuevo el Mal Instinto por delante de él y se lo encuentre de cara, como tantas otras veces, y sienta el deseo tan fuerte como la primera vez, no se dejará embaucar para abandonar el buen camino, tal como versa (Proverbios 28:13): “Pero el que los reconoce (a los pecados) y los abandona, será apiadado”. Primero menciona que se deben reconocer los pecados, y sólo después, abandonarlos. Sin embargo, la persona que se mantiene en el mal camino y continúa diariamente por el camino errado, se precipita muchas veces, en todo momento, deseando el mal, afrontando constantemente el deseo desenfrenado y sus instintos, sin poner medios para impedir satisfacer todo lo que su corazón anhela. El primer paso que debe dar este hombre en su deseo de mejorar es dejar el mal camino y los malos pensamientos, además de resolver que ya no va a pecar más. Después, se arrepentirá de todas sus malas acciones, hará penitencia y se dirigirá de nuevo hacia el Eterno, tal como versa (Yeshayahu 55:7): “El malvado abandonará su camino, y el inicuo, sus pensamientos. Regresará hacia Di-s y Él se apiadará”. Esto se asemeja a quien desea sumergirse en un baño ritual (Mikve) con un reptil en sus manos. Por supuesto que primero deberá soltar el reptil antes de entrar a purificarse, porque mientras sostenga en sus manos dicho reptil seguirá siendo impuro, y el baño no le servirá de nada. En este caso, abandonar los malos pensamientos es sinónimo de soltar el reptil, y el arrepentimiento por lo que pecó, la penitencia y el rezo, sinónimo de la ablución en el baño. Acontecerá que cuando aquel hombre comience a sufrir y padecer por lo malo que hizo y por su empeño de andar por el mal camino, se volverá atrás y no cometerá las iniquidades que pretendía. Un ejemplo de esto que decimos: Como aquella persona que golpea a su becerro con la vara que suele usar para hacer surcos, así aquel que persiste en el mal camino, deberá aprender de la moral en primer lugar, para después abandonar el camino de la muerte para encaminarse por el camino recto.
